59 inline void CReliableTimer::End()
60 {
61  m_nPerformanceCounterEnd = GetPerformanceCountNow();
62 
63  // enforce that we've advanced at least one cycle
64  if ( m_nPerformanceCounterEnd < m_nPerformanceCounterStart )
65  {
66 #ifdef _SERVER
67  if ( m_nPerformanceCounterEnd+10000 < m_nPerformanceCounterStart )
68  AssertMsgOnce( false, CDbgFmtMsg( "CReliableTimer went backwards - start:%lld end:%lld", m_nPerformanceCounterStart, m_nPerformanceCounterEnd ).ToString() );
69 #endif
70  m_nPerformanceCounterEnd = m_nPerformanceCounterStart + 1;
71  }
72 }

110 //-----------------------------------------------------------------------------
111 // Purpose: Returns if previously set limit has been reached
112 //-----------------------------------------------------------------------------
113 inline bool CReliableTimer::BLimitReached()
114 {
115  DbgAssert( m_nPerformanceCounterStart ); // SetLimit must have been called
116  DbgAssert( m_nPerformanceCounterLimit ); // SetLimit must have been called
117  int64 nPerformanceCountNow = GetPerformanceCountNow();
118 
119  // make sure time advances
120  if ( nPerformanceCountNow < m_nPerformanceCounterStart )
121  {
122 #ifdef _SERVER
123  if ( nPerformanceCountNow+10000 < m_nPerformanceCounterStart )
124  AssertMsgOnce( false, CDbgFmtMsg( "CReliableTimer went backwards - start:%lld end:%lld", m_nPerformanceCounterStart, m_nPerformanceCounterEnd ).ToString() );
125 #endif
126  // reset the limit to be lower, to match our new clock
127  m_nPerformanceCounterLimit = nPerformanceCountNow + (m_nPerformanceCounterLimit - m_nPerformanceCounterStart);
128  }
129 
130  return ( nPerformanceCountNow >= m_nPerformanceCounterLimit );
131 }
