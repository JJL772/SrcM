20 //-----------------------------------------------------------------------------
21 // Templatized helper class to deal with the kinds of things that spatial
22 // partition code always seems to have; buckets with lists of lots of elements
23 // and elements that live in lots of buckets. This makes it really quick to
24 // add and remove elements, and to iterate over all elements in a bucket.
25 //
26 // For this to work, you must initialize the set with two functions one that
27 // maps from bucket to the index of the first element in that bucket, and one
28 // that maps from element to the index of the first bucket that element lies in.
29 // The set will completely manage the index, it's just expected that those
30 // indices will be stored outside the set.
31 //
32 // S is the storage type of the index; it is the type that you may use to
33 // save indices into memory. I is the local iterator type, which you should
34 // use in any local scope (eg, inside a for() loop.) The reason for this is
35 // that you may wish to use unsigned shorts inside the structs you are
36 // saving with a CBidirectionalSet; but 16-bit arithmetic is catastrophically
37 // slow on a PowerPC -- during testing we saw CBidirectionalSet:: operations
38 // consume as much as 8% of the frame.
39 //
40 // For this reason, on the 360, the handles have been typedef'd to native
41 // register types (U32) which are accepted as parameters by the functions.
42 // The implicit assumption is that CBucketHandle and CElementHandle can
43 // be safely cast to ints! You can increase to U64 without performance
44 // penalty if necessary; the PowerPC is a 64-bit processor.
45 //-----------------------------------------------------------------------------
46 template< class CBucketHandle, class CElementHandle, class S, class I = S >

157 void CBidirectionalSet<CBucketHandle,CElementHandle,S,I>::ValidateAddElementToBucket( CBucketHandlePram bucket, CElementHandlePram element )
158 {
159 #ifdef _DEBUG
160  // Make sure that this element doesn't already exist in the list of elements in the bucket
161  I elementInBucket = m_FirstElement( bucket );
162  while( elementInBucket != m_ElementsInBucket.InvalidIndex() )
163  {
164  // If you hit an Assert here, fix the calling code. It's too expensive to ensure
165  // that each item only shows up once here. Hopefully you can do something better
166  // outside of here.
167  Assert( m_ElementsInBucket[elementInBucket].m_Element != element );
168  elementInBucket = m_ElementsInBucket.Next( elementInBucket );
169  }
170  // Make sure that this bucket doesn't already exist in the element's list of buckets.
171  I bucketInElement = m_FirstBucket( element );
172  while( bucketInElement != m_BucketsUsedByElement.InvalidIndex() )
173  {
174  // If you hit an Assert here, fix the calling code. It's too expensive to ensure
175  // that each item only shows up once here. Hopefully you can do something better
176  // outside of here.
177  Assert( m_BucketsUsedByElement[bucketInElement].m_Bucket != bucket );
178  bucketInElement = m_BucketsUsedByElement.Next( bucketInElement );
179  }
180 #endif

184 //-----------------------------------------------------------------------------
185 // Adds an element to the bucket
186 //-----------------------------------------------------------------------------
187 template< class CBucketHandle, class CElementHandle, class S, class I >
188 void CBidirectionalSet<CBucketHandle,CElementHandle,S,I>::AddElementToBucket( CBucketHandlePram bucket, CElementHandlePram element )
189 {
190  Assert( m_FirstBucket && m_FirstElement );
191 
192  // Allocate new element + bucket entries
193  I idx = m_ElementsInBucket.Alloc(true);
194  I list = m_BucketsUsedByElement.Alloc( true );
195 
196  // Store off the element data
197  m_ElementsInBucket[idx].m_Element = element;
198  m_ElementsInBucket[idx].m_BucketListIndex = list;
199 
200  // Here's the bucket data
201  m_BucketsUsedByElement[list].m_Bucket = bucket;
202  m_BucketsUsedByElement[list].m_ElementListIndex = idx;
203 
204  // Insert the element into the list of elements in the bucket
205  S& firstElementInBucket = m_FirstElement( bucket );
206  if ( firstElementInBucket != m_ElementsInBucket.InvalidIndex() )
207  m_ElementsInBucket.LinkBefore( firstElementInBucket, idx );
208  firstElementInBucket = idx;
209 
210  // Insert the bucket into the element's list of buckets
211  S& firstBucketInElement = m_FirstBucket( element );
212  if ( firstBucketInElement != m_BucketsUsedByElement.InvalidIndex() )
213  m_BucketsUsedByElement.LinkBefore( firstBucketInElement, list );
214  firstBucketInElement = list;
215 }

218 // Test if an element is in a particular bucket.

221 template< class CBucketHandle, class CElementHandle, class S, class I >
222 bool CBidirectionalSet<CBucketHandle,CElementHandle,S,I>::IsElementInBucket( CBucketHandlePram bucket, CElementHandlePram element )
223 {
224  // Search through all elements in this bucket to see if element is in there.
225  I elementInBucket = m_FirstElement( bucket );
226  while( elementInBucket != m_ElementsInBucket.InvalidIndex() )
227  {
228  if( m_ElementsInBucket[elementInBucket].m_Element == element )
229  {
230  return true;
231  }
232  elementInBucket = m_ElementsInBucket.Next( elementInBucket );
233  }
234  return false;
235 }

249 //-----------------------------------------------------------------------------
250 // Removes an element from all buckets
251 //-----------------------------------------------------------------------------
252 template< class CBucketHandle, class CElementHandle, class S, class I >
253 void CBidirectionalSet<CBucketHandle,CElementHandle,S,I>::RemoveElement( CElementHandlePram element )
254 {
255  Assert( m_FirstBucket && m_FirstElement );
256 
257  // Iterate over the list of all buckets the element is in
258  I i = m_FirstBucket( element );
259  while (i != m_BucketsUsedByElement.InvalidIndex())
260  {
261  CBucketHandlePram bucket = m_BucketsUsedByElement[i].m_Bucket;
262  I elementListIndex = m_BucketsUsedByElement[i].m_ElementListIndex;
263 
264  // Unhook the element from the bucket's list of elements
265  if (elementListIndex == m_FirstElement(bucket))
266  m_FirstElement(bucket) = m_ElementsInBucket.Next(elementListIndex);
267  m_ElementsInBucket.Free(elementListIndex);
268 
269  I prevNode = i;
270  i = m_BucketsUsedByElement.Next(i);
271  m_BucketsUsedByElement.Free(prevNode);
272  }
273 
274  // Mark the list as empty
275  m_FirstBucket( element ) = m_BucketsUsedByElement.InvalidIndex();
276 }

278 //-----------------------------------------------------------------------------
279 // Removes a bucket from all elements
280 //-----------------------------------------------------------------------------
281 template< class CBucketHandle, class CElementHandle, class S, class I >
282 void CBidirectionalSet<CBucketHandle,CElementHandle,S,I>::RemoveBucket( CBucketHandlePram bucket )
283 {
284  // Iterate over the list of all elements in the bucket
285  I i = m_FirstElement( bucket );
286  while (i != m_ElementsInBucket.InvalidIndex())
287  {
288  CElementHandlePram element = m_ElementsInBucket[i].m_Element;
289  I bucketListIndex = m_ElementsInBucket[i].m_BucketListIndex;
290 
291  // Unhook the bucket from the element's list of buckets
292  if (bucketListIndex == m_FirstBucket(element))
293  m_FirstBucket(element) = m_BucketsUsedByElement.Next(bucketListIndex);
294  m_BucketsUsedByElement.Free(bucketListIndex);
295 
296  // Remove the list element
297  I prevNode = i;
298  i = m_ElementsInBucket.Next(i);
299  m_ElementsInBucket.Free(prevNode);
300  }
301 
302  // Mark the bucket list as empty
303  m_FirstElement( bucket ) = m_ElementsInBucket.InvalidIndex();
304 }
