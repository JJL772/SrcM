70  //
71  // Returns a string representation of the value without modifying the variant.
72  //
73  const char *ToString( void ) const
74  {

77  switch (fieldType)

83 
84  case FIELD_BOOLEAN:

89  }
90  else
91  {
92  Q_strncpy(szBuf, "true",sizeof(szBuf));
93  }
94  return(szBuf);
95  }
96 
97  case FIELD_INTEGER:
98  {
99  Q_snprintf( szBuf, sizeof( szBuf ), "%i", iVal );
100  return(szBuf);
101  }
102 
103  case FIELD_FLOAT:
104  {

106  return(szBuf);
107  }
108 
109  case FIELD_COLOR32:
110  {

112  return(szBuf);
113  }
114 
115  case FIELD_VECTOR:
116  {
117  Q_snprintf(szBuf,sizeof(szBuf), "[%g %g %g]", (double)vecVal[0], (double)vecVal[1], (double)vecVal[2]);
118  return(szBuf);
119  }
120 
121  case FIELD_VOID:
122  {
123  szBuf[0] = '\0';
124  return(szBuf);
125  }
126  }
127 
128  return("No conversion to string");
129  }
