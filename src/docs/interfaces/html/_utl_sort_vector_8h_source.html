19 //-----------------------------------------------------------------------------
20 // class CUtlSortVector:
21 // description:
22 // This in an sorted order-preserving vector. Items may be inserted or removed
23 // at any point in the vector. When an item is inserted, all elements are
24 // moved down by one element using memmove. When an item is removed, all
25 // elements are shifted back down. Items are searched for in the vector
26 // using a binary search technique. Clients must pass in a Less() function
27 // into the constructor of the vector to determine the sort order.
28 //-----------------------------------------------------------------------------

175 template <class T, class LessFunc, class BaseVector>
176 int CUtlSortVector<T, LessFunc, BaseVector>::Insert( const T& src )
177 {
178  AssertFatal( !m_bNeedsSort );
179 
180  int pos = FindLessOrEqual( src ) + 1;
181  this->GrowVector();
182  this->ShiftElementsRight(pos);
183  CopyConstruct<T>( &this->Element(pos), src );
184  return pos;
185 }

277 template <class T, class LessFunc, class BaseVector>
278 template < typename TKey >
279 int CUtlSortVector<T, LessFunc, BaseVector>::Find( const TKey& src ) const
280 {
281  AssertFatal( !m_bNeedsSort );
282 
283  LessFunc less;
284 
285  int start = 0, end = this->Count() - 1;
286  while (start <= end)
287  {
288  int mid = (start + end) >> 1;
289  if ( less.Less( this->Element(mid), src, m_pLessContext ) )
290  {
291  start = mid + 1;
292  }
293  else if ( less.Less( src, this->Element(mid), m_pLessContext ) )
294  {
295  end = mid - 1;
296  }
297  else
298  {
299  return mid;
300  }
301  }
302  return -1;
303 }
