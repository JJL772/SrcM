70  Iterator_t First() const { return m_pBlocks ? Iterator_t( m_pBlocks, 0 ) : InvalidIterator(); }
71  Iterator_t Next( const Iterator_t &it ) const
72  {
73  Assert( IsValidIterator( it ) );
74  if ( !IsValidIterator( it ) )
75  return InvalidIterator();
76 
77  BlockHeader_t * RESTRICT pHeader = it.m_pBlockHeader;
78  if ( it.m_nIndex + 1 < pHeader->m_nBlockSize )
79  return Iterator_t( pHeader, it.m_nIndex + 1 );
80 
81  return pHeader->m_pNext ? Iterator_t( pHeader->m_pNext, 0 ) : InvalidIterator();
82  }
83  intp GetIndex( const Iterator_t &it ) const
84  {
85  Assert( IsValidIterator( it ) );
86  if ( !IsValidIterator( it ) )
87  return InvalidIndex();
88 
89  return ( intp )( HeaderToBlock( it.m_pBlockHeader ) + it.m_nIndex );
90  }
91  bool IsIdxAfter( intp i, const Iterator_t &it ) const
92  {
93  Assert( IsValidIterator( it ) );
94  if ( !IsValidIterator( it ) )
95  return false;
96 
97  if ( IsInBlock( i, it.m_pBlockHeader ) )
98  return i > GetIndex( it );
99 
100  for ( BlockHeader_t * RESTRICT pbh = it.m_pBlockHeader->m_pNext; pbh; pbh = pbh->m_pNext )
101  {
102  if ( IsInBlock( i, pbh ) )
103  return true;
104  }
105  return false;
106  }
107  bool IsValidIterator( const Iterator_t &it ) const { return it.m_pBlockHeader && it.m_nIndex >= 0 && it.m_nIndex < it.m_pBlockHeader->m_nBlockSize; }
108  Iterator_t InvalidIterator() const { return Iterator_t( NULL, INVALID_INDEX ); }

264 template< class T >
265 void CUtlFixedMemory<T>::Grow( int num )
266 {
267  if ( num <= 0 )
268  return;
269 
270  int nBlockSize = m_nGrowSize;
271  if ( nBlockSize == 0 )
272  {
273  if ( m_nAllocationCount )
274  {
275  nBlockSize = m_nAllocationCount;
276  }
277  else
278  {
279  // Compute an allocation which is at least as big as a cache line...
280  nBlockSize = ( 31 + sizeof( T ) ) / sizeof( T );
281  Assert( nBlockSize );
282  }
283  }
284  if ( nBlockSize < num )
285  {
286  int n = ( num + nBlockSize -1 ) / nBlockSize;
287  Assert( n * nBlockSize >= num );
288  Assert( ( n - 1 ) * nBlockSize < num );
289  nBlockSize *= n;
290  }
291  m_nAllocationCount += nBlockSize;
292 
293  MEM_ALLOC_CREDIT_CLASS();
294  BlockHeader_t * RESTRICT pBlockHeader = ( BlockHeader_t* )malloc( sizeof( BlockHeader_t ) + nBlockSize * sizeof( T ) );
295  if ( !pBlockHeader )
296  {
297  Error( "CUtlFixedMemory overflow!\n" );
298  }
299  pBlockHeader->m_pNext = NULL;
300  pBlockHeader->m_nBlockSize = nBlockSize;
301 
302  if ( !m_pBlocks )
303  {
304  m_pBlocks = pBlockHeader;
305  }
306  else
307  {
308 #if 1 // IsIdxAfter assumes that newly allocated blocks are at the end
309  BlockHeader_t * RESTRICT pbh = m_pBlocks;
310  while ( pbh->m_pNext )
311  {
312  pbh = pbh->m_pNext;
313  }
314  pbh->m_pNext = pBlockHeader;
315 #else
316  pBlockHeader = m_pBlocks;
317  pBlockHeader->m_pNext = m_pBlocks;
318 #endif
319  }
320 }
