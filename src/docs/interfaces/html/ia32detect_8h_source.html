276  void init2 (byte count)
277  {
278  uint32 d[4];
279  bool c[256];
280 
281  for (int ci1 = 0; ci1 < 256; ci1++)
282  c[ci1] = false;
283 
284  for (int i = 0; i < count; i++)
285  {
286 #ifdef COMPILER_MSVC64
287  __cpuid((int *) d, 2);
288 #else
289  __asm
290  {
291  mov eax, 2;
292  lea esi, d;
293  cpuid;
294  mov [esi + 0x0], eax;
295  mov [esi + 0x4], ebx;
296  mov [esi + 0x8], ecx;
297  mov [esi + 0xC], edx;
298  }
299 #endif
300 
301  if (i == 0)
302  d[0] &= 0xFFFFFF00;
303 
304  process2(d[0], c);
305  process2(d[1], c);
306  process2(d[2], c);
307  process2(d[3], c);
308  }
309 
310  int m = 0;
311 
312  for (int ci2 = 0; ci2 < 256; ci2++)
313  if (c[ci2])
314  m++;
315 
316  cache = new byte[m];
317 
318  m = 0;
319 
320  for (int ci3 = 1; ci3 < 256; ci3++)
