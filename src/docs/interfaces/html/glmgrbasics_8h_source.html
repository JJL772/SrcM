161 class GLMFuncLogger
162 {
163  public:
164 
165  // simple function log
166  GLMFuncLogger( const char *funcName )
167  {
168  m_funcName = funcName;
169  m_earlyOut = false;
170 
171  GLMPrintf( ">%s", m_funcName );
172  };
173 
174  // more advanced version lets you pass args (i.e. called parameters or anything else of interest)
175  // no macro for this one, since no easy way to pass through the args as well as the funcname
176  GLMFuncLogger( const char *funcName, char *fmt, ... )
177  {
178  m_funcName = funcName;
179  m_earlyOut = false;
180 
181  // this acts like GLMPrintf here
182  // all the indent policy is down in GLMPrintfVA
183  // which means we need to inject a ">" at the front of the format string to make this work... sigh.
184 
185  char modifiedFmt[2000];
186  modifiedFmt[0] = '>';
187  strcpy( modifiedFmt+1, fmt );
188 
189  va_list vargs;
190  va_start(vargs, fmt);
191  GLMPrintfVA( modifiedFmt, vargs );
192  va_end( vargs );
193  }
194 
195  ~GLMFuncLogger( )
196  {
197  if (m_earlyOut)
198  {
199  GLMPrintf( "<%s (early out)", m_funcName );
200  }
201  else
202  {
203  GLMPrintf( "<%s", m_funcName );
204  }
205  };
206 
207  void EarlyOut( void )
208  {
209  m_earlyOut = true;
210  };
211 
212  const char *m_funcName; // set at construction time
213  bool m_earlyOut;
214 };

253 // class based on the file mirror, that makes it easy to edit them outside the app.

255 // it receives an initial block of text from the engine, and hashes it. ("orig")
256 // it munges it by duplicating all the text after the "!!" line, and appending it in commented form. ("munged")
257 // a mirror file is activated, using a filename based on the hash from the orig text.
258 // if there is already content on disk matching that filename, use that content *unless* the 'blitz' parameter is set.
259 // (i.e. engine is instructing this subsystem to wipe out any old/modified variants of the text)
