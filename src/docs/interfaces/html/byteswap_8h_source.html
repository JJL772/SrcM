104  //-----------------------------------------------------------------------------
105  // IsByteSwapped()
106  //
107  // When supplied with a chunk of input data and a constant or magic number
108  // (in native format) determines the endienness of the current machine in
109  // relation to the given input data.
110  //
111  // Returns:
112  // 1 if input is the same as nativeConstant.
113  // 0 if input is byteswapped relative to nativeConstant.
114  // -1 if input is not the same as nativeConstant and not byteswapped either.
115  //
116  // ( This is useful for detecting byteswapping in magic numbers in structure
117  // headers for example. )
118  //-----------------------------------------------------------------------------
119  template<typename T> inline int SourceIsNativeEndian( T input, T nativeConstant )
120  {
121  // If it's the same, it isn't byteswapped:
122  if( input == nativeConstant )
123  return 1;
124 
125  int output;
126  LowLevelByteSwap<T>( &output, &input );
127  if( output == nativeConstant )
128  return 0;
129 
130  assert( 0 ); // if we get here, input is neither a swapped nor unswapped version of nativeConstant.
131  return -1;
132  }

206  //-----------------------------------------------------------------------------
207  // The lowest level byte swapping workhorse of doom. output always contains the
208  // swapped version of input. ( Doesn't compare machine to target endianness )
209  //-----------------------------------------------------------------------------
210  template<typename T> static void LowLevelByteSwap( T *output, T *input )
211  {
212  T temp = *output;
213 #if defined( _X360 )
214  // Intrinsics need the source type to be fixed-point
215  DWORD* word = (DWORD*)input;
216  switch( sizeof(T) )
217  {
218  case 8:
219  {
220  __storewordbytereverse( *word, 0, &temp );
221  __storewordbytereverse( *(word+1), 4, &temp );
222  }
223  break;
224 
225  case 4:
226  __storewordbytereverse( *word, 0, &temp );
227  break;
228 
229  case 2:
230  __storeshortbytereverse( *input, 0, &temp );
231  break;
232 
233  default:
234  Assert( "Invalid size in CByteswap::LowLevelByteSwap" && 0 );
235  }
236 #else
237  for( int i = 0; i < sizeof(T); i++ )
238  {
239  ((unsigned char* )&temp)[i] = ((unsigned char*)input)[sizeof(T)-(i+1)];
240  }
