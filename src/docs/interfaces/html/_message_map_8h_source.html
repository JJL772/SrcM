81  { \
82  vgui::PanelMessageMap *map = vgui::FindOrAddPanelMessageMap( GetPanelClassName() ); \
83  \

88  entry.firstParamType = (vgui::DataType_t)p1type; \

99  static void ChainToMap( void ) \
100  { \
101  static bool chained = false; \
102  if ( chained ) \
103  return; \
104  chained = true; \
105  vgui::PanelMessageMap *map = vgui::FindOrAddPanelMessageMap( GetPanelClassName() ); \
106  map->pfnClassName = &GetPanelClassName; \
107  if ( map && GetPanelBaseClassName() && GetPanelBaseClassName()[0] ) \
108  { \
109  map->baseMap = vgui::FindOrAddPanelMessageMap( GetPanelBaseClassName() ); \
110  } \
111  } \
