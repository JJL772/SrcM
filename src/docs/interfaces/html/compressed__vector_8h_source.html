55 inline Vector32& Vector32::operator=(const Vector &vOther)
56 {
57  CHECK_VALID(vOther);
58 
59  static float expScale[4] = { 4.0f, 16.0f, 32.f, 64.f };
60 
61  float fmax = Max( fabs( vOther.x ), fabs( vOther.y ) );
62  fmax = Max( fmax, (float)fabs( vOther.z ) );
63 
64  for (exp = 0; exp < 3; exp++)
65  {
66  if (fmax < expScale[exp])
67  break;
68  }
69  Assert( fmax < expScale[exp] );
70 
71  float fexp = 512.0f / expScale[exp];
72 
73  x = Clamp( (int)(vOther.x * fexp) + 512, 0, 1023 );
74  y = Clamp( (int)(vOther.y * fexp) + 512, 0, 1023 );
75  z = Clamp( (int)(vOther.z * fexp) + 512, 0, 1023 );
76  return *this;
77 }

367  // 0x0001 - 0x03ff
368  static unsigned short ConvertFloatTo16bits( float input )
369  {
370  if ( input > maxfloat16bits )
371  input = maxfloat16bits;
372  else if ( input < -maxfloat16bits )
373  input = -maxfloat16bits;
374 
375  float16bits output;
376  float32bits inFloat;
377 
378  inFloat.rawFloat = input;
379 
380  output.bits.sign = inFloat.bits.sign;
381 
382  if ( (inFloat.bits.biased_exponent==0) && (inFloat.bits.mantissa==0) )
383  {
384  // zero
385  output.bits.mantissa = 0;
386  output.bits.biased_exponent = 0;
387  }
388  else if ( (inFloat.bits.biased_exponent==0) && (inFloat.bits.mantissa!=0) )
389  {
390  // denorm -- denorm float maps to 0 half
391  output.bits.mantissa = 0;
392  output.bits.biased_exponent = 0;
393  }
394  else if ( (inFloat.bits.biased_exponent==0xff) && (inFloat.bits.mantissa==0) )
395  {
396 #if 0
397  // infinity
398  output.bits.mantissa = 0;
399  output.bits.biased_exponent = 31;
400 #else
401  // infinity maps to maxfloat
402  output.bits.mantissa = 0x3ff;
403  output.bits.biased_exponent = 0x1e;
404 #endif
405  }
406  else if ( (inFloat.bits.biased_exponent==0xff) && (inFloat.bits.mantissa!=0) )
407  {
408 #if 0
409  // NaN
410  output.bits.mantissa = 1;
411  output.bits.biased_exponent = 31;
412 #else
413  // NaN maps to zero
414  output.bits.mantissa = 0;
415  output.bits.biased_exponent = 0;
416 #endif
417  }
418  else
419  {
420  // regular number
421  int new_exp = inFloat.bits.biased_exponent-127;
422 
423  if (new_exp<-24)
424  {
425  // this maps to 0
426  output.bits.mantissa = 0;
427  output.bits.biased_exponent = 0;
428  }
429 
430  if (new_exp<-14)
431  {
432  // this maps to a denorm
433  output.bits.biased_exponent = 0;
434  unsigned int exp_val = ( unsigned int )( -14 - ( inFloat.bits.biased_exponent - float32bias ) );
435  if( exp_val > 0 && exp_val < 11 )
436  {
437  output.bits.mantissa = ( 1 << ( 10 - exp_val ) ) + ( inFloat.bits.mantissa >> ( 13 + exp_val ) );
438  }
439  }
440  else if (new_exp>15)
441  {
442 #if 0
443  // map this value to infinity
444  output.bits.mantissa = 0;
445  output.bits.biased_exponent = 31;
446 #else
447  // to big. . . maps to maxfloat
448  output.bits.mantissa = 0x3ff;
449  output.bits.biased_exponent = 0x1e;
450 #endif
451  }
452  else
453  {
454  output.bits.biased_exponent = new_exp+15;
455  output.bits.mantissa = (inFloat.bits.mantissa >> 13);
456  }
457  }
458  return output.rawWord;
459  }

461  static float Convert16bitFloatTo32bits( unsigned short input )
462  {
463  float32bits output;
464  const float16bits &inFloat = *((float16bits *)&input);
465 
466  if( IsInfinity( inFloat ) )
467  {
468  return maxfloat16bits * ( ( inFloat.bits.sign == 1 ) ? -1.0f : 1.0f );
469  }
470  if( IsNaN( inFloat ) )
471  {
472  return 0.0;
473  }
474  if( inFloat.bits.biased_exponent == 0 && inFloat.bits.mantissa != 0 )
475  {
476  // denorm
477  const float half_denorm = (1.0f/16384.0f); // 2^-14
478  float mantissa = ((float)(inFloat.bits.mantissa)) / 1024.0f;
479  float sgn = (inFloat.bits.sign)? -1.0f :1.0f;
480  output.rawFloat = sgn*mantissa*half_denorm;
481  }
482  else
483  {
484  // regular number
485  unsigned mantissa = inFloat.bits.mantissa;
486  unsigned biased_exponent = inFloat.bits.biased_exponent;
487  unsigned sign = ((unsigned)inFloat.bits.sign) << 31;
488  biased_exponent = ( (biased_exponent - float16bias + float32bias) * (biased_exponent != 0) ) << 23;
489  mantissa <<= (23-10);
490 
491  *((unsigned *)&output) = ( mantissa | biased_exponent | sign );
492  }
493 
494  return output.rawFloat;
495  }
