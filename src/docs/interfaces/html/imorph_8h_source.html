132 //-----------------------------------------------------------------------------

135 inline void CMorphBuilder::Begin( IMorph *pMorph, float flFloatToFixedScale )
136 {
137  Assert( pMorph && !m_pMorph );
138  m_pMorph = pMorph;
139  m_pMorph->Lock( flFloatToFixedScale );
140 
141 #ifdef _DEBUG

148 }
