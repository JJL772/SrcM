138  // Handles receiving a new connection from a Steam user. This call will ask the Steam
139  // servers to validate the users identity, app ownership, and VAC status. If the Steam servers
140  // are off-line, then it will validate the cached ticket itself which will validate app ownership
141  // and identity. The AuthBlob here should be acquired on the game client using SteamUser()->InitiateGameConnection()
142  // and must then be sent up to the game server for authentication.
143  //
144  // Return Value: returns true if the users ticket passes basic checks. pSteamIDUser will contain the Steam ID of this user. pSteamIDUser must NOT be NULL
145  // If the call succeeds then you should expect a GSClientApprove_t or GSClientDeny_t callback which will tell you whether authentication
146  // for the user has succeeded or failed (the steamid in the callback will match the one returned by this call)
147  virtual bool SendUserConnectAndAuthenticate( uint32 unIPClient, const void *pvAuthBlob, uint32 cubAuthBlobSize, CSteamID *pSteamIDUser ) = 0;
148 
149  // Creates a fake user (ie, a bot) which will be listed as playing on the server, but skips validation.
150  //
151  // Return Value: Returns a SteamID for the user to be tracked with, you should call HandleUserDisconnect()
152  // when this user leaves the server just like you would for a real user.
153  virtual CSteamID CreateUnauthenticatedUserConnection() = 0;
154 
155  // Should be called whenever a user leaves our game server, this lets Steam internally
156  // track which users are currently on which servers for the purposes of preventing a single
157  // account being logged into multiple servers, showing who is currently on a server, etc.
158  virtual void SendUserDisconnect( CSteamID steamIDUser ) = 0;
159 
160  // Update the data to be displayed in the server browser and matchmaking interfaces for a user
161  // currently connected to the server. For regular users you must call this after you receive a
162  // GSUserValidationSuccess callback.
163  //
164  // Return Value: true if successful, false if failure (ie, steamIDUser wasn't for an active player)
165  virtual bool BUpdateUserData( CSteamID steamIDUser, const char *pchPlayerName, uint32 uScore ) = 0;

203 // These are in GameSocketShare mode, where instead of ISteamGameServer creating its own
204 // socket to talk to the master server on, it lets the game use its socket to forward messages
205 // back and forth. This prevents us from requiring server ops to open up yet another port
206 // in their firewalls.
207 //
208 // the IP address and port should be in host order, i.e 127.0.0.1 == 0x7f000001
209 
210  // These are used when you've elected to multiplex the game server's UDP socket
211  // rather than having the master server updater use its own sockets.
212  //
213  // Source games use this to simplify the job of the server admins, so they
214  // don't have to open up more ports on their firewalls.
215 
216  // Call this when a packet that starts with 0xFFFFFFFF comes in. That means
217  // it's for us.
218  virtual bool HandleIncomingPacket( const void *pData, int cbData, uint32 srcIP, uint16 srcPort ) = 0;
219 
220  // AFTER calling HandleIncomingPacket for any packets that came in that frame, call this.
221  // This gets a packet that the master server updater needs to send out on UDP.
222  // It returns the length of the packet it wants to send, or 0 if there are no more packets to send.
223  // Call this each frame until it returns 0.
224  virtual int GetNextOutgoingPacket( void *pOut, int cbMaxOut, uint32 *pNetAdr, uint16 *pPort ) = 0;
