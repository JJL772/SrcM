102 inline void CBaseMeshReader::BeginRead(
103  IMesh* pMesh,
104  int firstVertex,
105  int numVertices,
106  int firstIndex,
107  int numIndices )
108 {
109  Assert( pMesh && (!m_pMesh) );
110 
111  if ( numVertices < 0 )
112  {
113  numVertices = pMesh->VertexCount();
114  }
115 
116  if ( numIndices < 0 )
117  {
118  numIndices = pMesh->IndexCount();
119  }
120 
121  m_pMesh = pMesh;
122  m_MaxVertices = numVertices;
123  m_MaxIndices = numIndices;
124 
125  // UNDONE: support reading from compressed VBs if needed
126  VertexCompressionType_t compressionType = CompressionType( pMesh->GetVertexFormat() );
127  Assert( compressionType == VERTEX_COMPRESSION_NONE );
128  if ( compressionType != VERTEX_COMPRESSION_NONE )
129  {
130  Warning( "Cannot use CBaseMeshReader with compressed vertices! Will get junk data or a crash.\n" );
131  }
132 
133  // Locks mesh for modifying
134  pMesh->ModifyBeginEx( true, firstVertex, numVertices, firstIndex, numIndices, *this );
135 
136  // Point to the start of the buffers..
137  Reset();
138 }
139 
140 inline void CBaseMeshReader::EndRead()
141 {
142  Assert( m_pMesh );
143  m_pMesh->ModifyEnd( *this );
144  m_pMesh = NULL;
145 }
