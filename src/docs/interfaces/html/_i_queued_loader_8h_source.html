68 abstract_class IResourcePreload
69 {
70 public:
71  // Called during preload phase for ALL the resources expected by the level.
72  // Caller should not do i/o but generate AddJob() requests. Resources that already exist
73  // and are not referenced by this function would be candidates for purge.
74  virtual bool CreateResource( const char *pName ) = 0;
75 
76  // Sent as an event hint during preload, that creation has completed, AddJob() i/o is about to commence.
77  // Caller should purge any unreferenced resources before the AddJobs are performed.
78  // "Must Complete" data will be guaranteed finished, at preload conclusion, before the normal load phase commences.
79  virtual void PurgeUnreferencedResources() = 0;
80 
81  // Sent as an event hint that gameplay rendering is imminent.
82  // Low priority jobs may still be in async flight.
83  virtual void OnEndMapLoading( bool bAbort ) = 0;
84 
85  virtual void PurgeAll() = 0;
86 };
