56 // Same as CaselessStringLessThan, but it ignores differences in / and \.
57 inline bool CaselessStringLessThanIgnoreSlashes( const char * const &lhs, const char * const &rhs )
58 {
59  const char *pa = lhs;
60  const char *pb = rhs;
61  while ( *pa && *pb )
62  {
63  char a = *pa;
64  char b = *pb;
65 
66  // Check for dir slashes.
67  if ( a == '/' || a == '\\' )
68  {
69  if ( b != '/' && b != '\\' )
70  return ('/' < b);
71  }
72  else
73  {
74  if ( a >= 'a' && a <= 'z' )
75  a = 'A' + (a - 'a');
76 
77  if ( b >= 'a' && b <= 'z' )
78  b = 'A' + (b - 'a');
79 
80  if ( a > b )
81  return false;
82  else if ( a < b )
83  return true;
84  }
85  ++pa;
86  ++pb;
87  }
88 
89  // Filenames also must be the same length.
90  if ( *pa != *pb )
91  {
92  // If pa shorter than pb then it's "less"
93  return ( !*pa );
94  }
95 
96  return false;
97 }

664 //-----------------------------------------------------------------------------
665 // Allocates/ deallocates nodes
666 //-----------------------------------------------------------------------------
667 #pragma warning(push)
668 #pragma warning(disable:4389) // '==' : signed/unsigned mismatch
669 template < class T, class I, typename L, class M >
670 I CUtlRBTree<T, I, L, M>::NewNode()
671 {
672  I elem;
673 
674  // Nothing in the free list; add.
675  if ( m_FirstFree == InvalidIndex() )
676  {
677  Assert( m_Elements.IsValidIterator( m_LastAlloc ) || m_NumElements == 0 );
678  typename M::Iterator_t it = m_Elements.IsValidIterator( m_LastAlloc ) ? m_Elements.Next( m_LastAlloc ) : m_Elements.First();
679  if ( !m_Elements.IsValidIterator( it ) )
680  {
681  MEM_ALLOC_CREDIT_CLASS();
682  m_Elements.Grow();
683 
684  it = m_Elements.IsValidIterator( m_LastAlloc ) ? m_Elements.Next( m_LastAlloc ) : m_Elements.First();
685 
686  Assert( m_Elements.IsValidIterator( it ) );
687  if ( !m_Elements.IsValidIterator( it ) )
688  {
689  Error( "CUtlRBTree overflow!\n" );
690  }
691  }
692  m_LastAlloc = it;
693  elem = m_Elements.GetIndex( m_LastAlloc );
694  Assert( m_Elements.IsValidIterator( m_LastAlloc ) );
695  }
696  else
697  {
698  elem = m_FirstFree;
699  m_FirstFree = Links( m_FirstFree ).m_Right;
700  }
701 
702 #ifdef _DEBUG
703  // reset links to invalid....
704  Links_t &node = Links( elem );
705  node.m_Left = node.m_Right = node.m_Parent = InvalidIndex();
706 #endif
707 
708  Construct( &Element( elem ) );
709  ResetDbgInfo();
710 
711  return elem;
712 }
