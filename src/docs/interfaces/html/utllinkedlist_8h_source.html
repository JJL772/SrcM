32 // A lovely index-based linked list! T is the class type, I is the index
33 // type, which usually should be an unsigned short or smaller. However,
34 // you must avoid using 16- or 8-bit arithmetic on PowerPC architectures;

53 // Class S is the storage type; the type you can use to save off indices in
54 // persistent memory. Class I is the iterator type, which is what should be used
55 // in local scopes. I defaults to be S, but be aware that on the 360, 16-bit
56 // arithmetic is catastrophically slow. Therefore you should try to save shorts
57 // in memory, but always operate on 32's or 64's in local scope.
58 // The ideal parameter order would be TSMI (you are more likely to override M than I)

145  template < typename List_t >
146  class _CUtlLinkedList_constiterator_t
147  {
148  public:
149  typedef typename List_t::ElemType_t ElemType_t;
150  typedef typename List_t::IndexType_t IndexType_t;
151 
152  // Default constructor -- gives a currently unusable iterator.
153  _CUtlLinkedList_constiterator_t()
154  : m_list( 0 )
155  , m_index( List_t::InvalidIndex() )
156  {
157  }
158  // Normal constructor.
159  _CUtlLinkedList_constiterator_t( const List_t& list, IndexType_t index )
160  : m_list( &list )
161  , m_index( index )
162  {
163  }
164 
165  // Pre-increment operator++. This is the most efficient increment
166  // operator so it should always be used.
167  _CUtlLinkedList_constiterator_t& operator++()
168  {
169  m_index = m_list->Next( m_index );
170  return *this;
171  }
172  // Post-increment operator++. This is less efficient than pre-increment.
173  _CUtlLinkedList_constiterator_t operator++(int)
174  {
175  // Copy ourselves.
176  _CUtlLinkedList_constiterator_t temp = *this;
177  // Increment ourselves.
178  ++*this;
179  // Return the copy.
180  return temp;
181  }
182 
183  // Pre-decrement operator--. This is the most efficient decrement
184  // operator so it should always be used.
185  _CUtlLinkedList_constiterator_t& operator--()
186  {
187  Assert( m_index != m_list->Head() );
188  if ( m_index == m_list->InvalidIndex() )
189  {
190  m_index = m_list->Tail();
191  }
192  else
193  {
194  m_index = m_list->Previous( m_index );
195  }
196  return *this;
197  }
198  // Post-decrement operator--. This is less efficient than post-decrement.
199  _CUtlLinkedList_constiterator_t operator--(int)
200  {
201  // Copy ourselves.
202  _CUtlLinkedList_constiterator_t temp = *this;
203  // Decrement ourselves.
204  --*this;
205  // Return the copy.
206  return temp;
207  }
208 
209  bool operator==( const _CUtlLinkedList_constiterator_t& other) const
210  {
211  Assert( m_list == other.m_list );
212  return m_index == other.m_index;
213  }
214 
215  bool operator!=( const _CUtlLinkedList_constiterator_t& other) const
216  {
217  Assert( m_list == other.m_list );
218  return m_index != other.m_index;
219  }
220 
221  const ElemType_t& operator*() const
222  {
223  return m_list->Element( m_index );
224  }
225 
226  const ElemType_t* operator->() const
227  {
228  return (&**this);
229  }
230 
231  protected:
232  // Use a pointer rather than a reference so that we can support
233  // assignment of iterators.
234  const List_t* m_list;
235  IndexType_t m_index;
236  };

548 #pragma warning(push)
549 #pragma warning( disable: 4310 ) // Allows "(I)(S)M::INVALID_INDEX" below
550 template <class T, class S, bool ML, class I, class M>
551 inline bool CUtlLinkedList<T,S,ML,I,M>::IndexInRange( I index ) // Static method
552 {
553  // Since S is not necessarily the type returned by M, we need to check that M returns indices
554  // which are representable by S. A common case is 'S === unsigned short', 'I == int', in which
555  // case CUtlMemory will have 'InvalidIndex == (int)-1' (which casts to 65535 in S), and will
556  // happily return elements at index 65535 and above.
557 
558  // Do some static checks here:
559  // 'I' needs to be able to store 'S'
560  COMPILE_TIME_ASSERT( sizeof(I) >= sizeof(S) );
561  // 'S' should be unsigned (to avoid signed arithmetic errors for plausibly exhaustible ranges)
562  COMPILE_TIME_ASSERT( ( sizeof(S) > 2 ) || ( ( (S)-1 ) > 0 ) );
563  // M::INVALID_INDEX should be storable in S to avoid ambiguities (e.g. with 65536)
564  COMPILE_TIME_ASSERT( ( M::INVALID_INDEX == -1 ) || ( M::INVALID_INDEX == (S)M::INVALID_INDEX ) );
565 
566  return ( ( (S)index == index ) && ( (S)index != InvalidIndex() ) );
567 }

654 //-----------------------------------------------------------------------------
655 // Node allocation/deallocation
656 //-----------------------------------------------------------------------------
657 template <class T, class S, bool ML, class I, class M>
658 I CUtlLinkedList<T,S,ML,I,M>::AllocInternal( bool multilist )
659 {
660  Assert( !multilist || ML );
661 #ifdef MULTILIST_PEDANTIC_ASSERTS
662  Assert( multilist == ML );
663 #endif
664  I elem;
665  if ( m_FirstFree == InvalidIndex() )
666  {
667  Assert( m_Memory.IsValidIterator( m_LastAlloc ) || m_ElementCount == 0 );
668 
669  typename M::Iterator_t it = m_Memory.IsValidIterator( m_LastAlloc ) ? m_Memory.Next( m_LastAlloc ) : m_Memory.First();
670 
671  if ( !m_Memory.IsValidIterator( it ) )
672  {
673  MEM_ALLOC_CREDIT_CLASS();
674  m_Memory.Grow();
675  ResetDbgInfo();
676 
677  it = m_Memory.IsValidIterator( m_LastAlloc ) ? m_Memory.Next( m_LastAlloc ) : m_Memory.First();
678 
679  Assert( m_Memory.IsValidIterator( it ) );
680  if ( !m_Memory.IsValidIterator( it ) )
681  {

887 void CUtlLinkedList<T,S,ML,I,M>::RemoveAll()
888 {

890  // valid elements for the multilist case (since we don't have all elements
891  // connected to each other in a list).
892 
893  if ( m_LastAlloc == m_Memory.InvalidIterator() )
894  {
895  Assert( m_Head == InvalidIndex() );
896  Assert( m_Tail == InvalidIndex() );
897  Assert( m_FirstFree == InvalidIndex() );
898  Assert( m_ElementCount == 0 );
899  return;

902  if ( ML )
903  {
904  for ( typename M::Iterator_t it = m_Memory.First(); it != m_Memory.InvalidIterator(); it = m_Memory.Next( it ) )

907  if ( IsValidIndex( i ) ) // skip elements already in the free list
908  {

910  Destruct( &internalElem.m_Element );
911  internalElem.m_Previous = i;
912  internalElem.m_Next = m_FirstFree;
913  m_FirstFree = i;
914  }
915

917  break; // don't destruct elements that haven't ever been constructed
918  }
919  }

923  I next;

928  Destruct( &internalElem.m_Element );
929  internalElem.m_Previous = i;
930  internalElem.m_Next = next == InvalidIndex() ? m_FirstFree : next;
931  i = next;
932  }

934  {
935  m_FirstFree = Head();
936  }
937  }

940  m_Head = InvalidIndex();
941  m_Tail = InvalidIndex();

950 template <class T, class S, bool ML, class I, class M>
951 void CUtlLinkedList<T,S,ML,I,M>::LinkBefore( I before, I elem )
952 {
953  Assert( IsValidIndex(elem) );
954 
955  // Unlink it if it's in the list at the moment
956  Unlink(elem);
957 
958  ListElem_t * RESTRICT pNewElem = &InternalElement(elem);
959 
960  // The element *after* our newly linked one is the one we linked before.
961  pNewElem->m_Next = before;
962 
963  S newElem_mPrevious; // we need to hang on to this for the compairson against InvalidIndex()
964  // below; otherwise we get a a load-hit-store on pNewElem->m_Previous, even
965  // with RESTRICT
966  if (before == InvalidIndex())
967  {
968  // In this case, we're linking to the end of the list, so reset the tail
969  newElem_mPrevious = m_Tail;
970  pNewElem->m_Previous = m_Tail;
971  m_Tail = elem;
972  }
973  else
974  {
975  // Here, we're not linking to the end. Set the prev pointer to point to
976  // the element we're linking.
977  Assert( IsInList(before) );
978  ListElem_t * RESTRICT beforeElem = &InternalElement(before);
979  pNewElem->m_Previous = newElem_mPrevious = beforeElem->m_Previous;
980  beforeElem->m_Previous = elem;
981  }
982 
983  // Reset the head if we linked to the head of the list
984  if (newElem_mPrevious == InvalidIndex())
985  m_Head = elem;
986  else
987  InternalElement(newElem_mPrevious).m_Next = elem;
988 
989  // one more element baby
990  ++m_ElementCount;
991 }
