177 inline void *MemAlloc_AllocAligned( size_t size, size_t align )
178 {
179  unsigned char *pAlloc, *pResult;
180 
181  if (!IsPowerOfTwo(align))
182  return NULL;
183 
184  align = (align > sizeof(void *) ? align : sizeof(void *)) - 1;
185 
186  if ( (pAlloc = (unsigned char*)g_pMemAlloc->Alloc( sizeof(void *) + align + size ) ) == (unsigned char*)NULL)
187  return NULL;
188 
189  pResult = (unsigned char*)( (size_t)(pAlloc + sizeof(void *) + align ) & ~align );
190  ((unsigned char**)(pResult))[-1] = pAlloc;
191 
192  return (void *)pResult;
193 }

249 inline void *MemAlloc_ReallocAligned( void *ptr, size_t size, size_t align )
250 {
251  if ( !IsPowerOfTwo( align ) )
252  return NULL;
253 
254  // Don't change alignment between allocation + reallocation.
255  if ( ( (size_t)ptr & ( align - 1 ) ) != 0 )
256  return NULL;
257 
258  if ( !ptr )
259  return MemAlloc_AllocAligned( size, align );
260 
261  void *pAlloc, *pResult;
262 
263  // Figure out the actual allocation point
264  pAlloc = ptr;
265  pAlloc = (void *)(((size_t)pAlloc & ~( sizeof(void *) - 1 ) ) - sizeof(void *));
266  pAlloc = *( (void **)pAlloc );
267 
268  // See if we have enough space
269  size_t nOffset = (size_t)ptr - (size_t)pAlloc;
270  size_t nOldSize = g_pMemAlloc->GetSize( pAlloc );
271  if ( nOldSize >= size + nOffset )
272  return ptr;
273 
274  pResult = MemAlloc_AllocAligned( size, align );
275  memcpy( pResult, ptr, nOldSize - nOffset );
276  g_pMemAlloc->Free( pAlloc );
277  return pResult;
278 }
