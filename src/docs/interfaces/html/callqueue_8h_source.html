93  if ( m_bNoQueue == bDisable )

113  void CallQueued()
114  {
115  if ( !m_queue.Count() )
116  {
117  return;
118  }
119 
120  m_queue.PushItem( NULL );
121 
122  CFunctor *pFunctor;
123 
124  while ( m_queue.PopItem( &pFunctor ) && pFunctor != NULL )
125  {
126 #ifdef _DEBUG
127  if ( pFunctor->m_nUserID == m_nBreakSerialNumber)
128  {
129  m_nBreakSerialNumber = (unsigned)-1;
130  }
131 #endif
132  (*pFunctor)();
133  pFunctor->Release();
134  }
135 
136  }
137 
138  void QueueFunctor( CFunctor *pFunctor )
139  {
140  Assert( pFunctor );
141  QueueFunctorInternal( RetAddRef( pFunctor ) );
142  }
143 
144  void Flush()
145  {
146  m_queue.PushItem( NULL );
147 
148  CFunctor *pFunctor;
149 
150  while ( m_queue.PopItem( &pFunctor ) && pFunctor != NULL )
151  {
152  pFunctor->Release();
153  }
154  }
155 
156  FUNC_GENERATE_QUEUE_METHODS();
157 
158 private:
159  void QueueFunctorInternal( CFunctor *pFunctor )
160  {
161  if ( !m_bNoQueue )
162  {
163 #ifdef _DEBUG
164  pFunctor->m_nUserID = m_nCurSerialNumber++;
165 #endif
166  m_queue.PushItem( pFunctor );
167  }
168  else
169  {
170  (*pFunctor)();
171  pFunctor->Release();
172  }
173  }
174
