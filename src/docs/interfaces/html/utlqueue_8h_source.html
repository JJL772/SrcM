118 template <class T, class M>
119 inline bool CUtlQueue<T, M>::RemoveAtHead( T &removedElement )
120 {
121  Assert( m_head != QUEUE_ITERATOR_INVALID );
122  if ( m_head == QUEUE_ITERATOR_INVALID )
123  {
124  Construct( &removedElement );
125  return false;
126  }
127 
128  QueueIter_t it = m_head;
129  removedElement = m_memory[ it ];
130  Destruct( &m_memory[ it ] );
131  if ( m_head == m_tail )
132  {
133  m_head = m_tail = QUEUE_ITERATOR_INVALID;
134  }
135  else
136  {
137  m_head = Next_Unchecked( m_head );
138  }
139  return true;
140 }

199 template <class T, class M>
200 void CUtlQueue<T, M>::Insert( T const &element )
201 {
202  if ( m_tail == QUEUE_ITERATOR_INVALID )
203  {
204  // empty
205  m_memory.EnsureCapacity( 1 );
206  m_head = m_tail = QueueIter_t( 0 );
207  }
208  else
209  {
210  // non-empty
211  QueueIter_t nextTail = Next_Unchecked( m_tail );
212  if ( nextTail == m_head ) // if non-empty, and growing by 1 appears to make the queue of length 1, then we were already full before the Insert
213  {
214  int nOldAllocCount = m_memory.NumAllocated();
215  m_memory.Grow();
216  int nNewAllocCount = m_memory.NumAllocated();
217  int nGrowAmount = nNewAllocCount - nOldAllocCount;
218 
219  nextTail = Next_Unchecked( m_tail ); // if nextTail was 0, then it now should be nOldAllocCount
220 
221  if ( m_head != QueueIter_t( 0 ) )
222  {
223  // if the queue wraps around the end of m_memory, move the part at the end of memory to the new end of memory
224  Q_memmove( &m_memory[ m_head + nGrowAmount ], &m_memory[ m_head ], ( nOldAllocCount - m_head ) * sizeof( T ) );
225 #ifdef _DEBUG
226  Q_memset( &m_memory[ m_head ], 0xdd, nGrowAmount * sizeof( T ) );
227 #endif
228  m_head = QueueIter_t( m_head + nGrowAmount );
229  }
230  }
231  m_tail = nextTail;
232  }
233 
234  CopyConstruct( &m_memory[ m_tail ], element );
235 }

306 {

311  return false;
312
