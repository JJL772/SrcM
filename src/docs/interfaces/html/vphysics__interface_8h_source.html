22 // ------------------------------------------------------------------------------------
23 // UNITS:
24 // ------------------------------------------------------------------------------------
25 // NOTE: Coordinates are in HL units. 1 unit == 1 inch. X is east (forward), Y is north (left), Z is up (up)
26 // QAngle are pitch (around y), Yaw (around Z), Roll (around X)
27 // AngularImpulse are exponetial maps (an axis in HL units scaled by a "twist" angle in degrees)
28 // They can be transformed like normals/covectors and added linearly
29 // mass is kg, volume is in^3, acceleration is in/s^2, velocity is in/s
30 
31 // density is kg/m^3 (water ~= 998 at room temperature)
32 // preferably, these would be in kg/in^3, but the range of those numbers makes them not very human readable
33 // having water be about 1000 is really convenient for data entry.
34 // Since volume is in in^3 and density is in kg/m^3:
35 // density = (mass / volume) * CUBIC_METERS_PER_CUBIC_INCH
36 // Force is applied using impulses (kg*in/s)
37 // Torque is applied using impulses (kg*degrees/s)
38 // ------------------------------------------------------------------------------------
39 
40 #define METERS_PER_INCH (0.0254f)
41 #define CUBIC_METERS_PER_CUBIC_INCH (METERS_PER_INCH*METERS_PER_INCH*METERS_PER_INCH)
42 // 2.2 lbs / kg
43 #define POUNDS_PER_KG (2.2f)
44 #define KG_PER_POUND (1.0f/POUNDS_PER_KG)
45 
46 // convert from pounds to kg
47 #define lbs2kg(x) ((x)*KG_PER_POUND)
48 #define kg2lbs(x) ((x)*POUNDS_PER_KG)
