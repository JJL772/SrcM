397 {
398  if ( !m_pFirstFree )
399  {
400  FreeBlock_t *pNew = (FreeBlock_t *)m_Allocator.Alloc( CHUNK_SIZE );
401  Assert( (unsigned)pNew % ALIGNMENT == 0 );
402  m_Chunks.AddToTail( pNew );
403  m_nFree = CHUNK_SIZE / BLOCK_SIZE;
404  m_pFirstFree = pNew;
405  for ( int i = 0; i < m_nFree - 1; i++ )
406  {

408  pNew++;

410  pNew->pNext = NULL;
411  }

414  m_pFirstFree = m_pFirstFree->pNext;

420 template <int ITEM_SIZE, int ALIGNMENT, int CHUNK_SIZE, class CAllocator, int COMPACT_THRESHOLD >
421 inline void CAlignedMemPool<ITEM_SIZE, ALIGNMENT, CHUNK_SIZE, CAllocator, COMPACT_THRESHOLD>::Free( void *p )
422 {
423  // Insertion sort to encourage allocation clusters in chunks
424  FreeBlock_t *pFree = ((FreeBlock_t *)p);
425  FreeBlock_t *pCur = m_pFirstFree;
426  FreeBlock_t *pPrev = NULL;
427 
428  while ( pCur && pFree > pCur )
429  {
430  pPrev = pCur;
431  pCur = pCur->pNext;
432  }
433 
434  pFree->pNext = pCur;
435 
436  if ( pPrev )
437  {
438  pPrev->pNext = pFree;
439  }
440  else
441  {
442  m_pFirstFree = pFree;
443  }
