371 inline void CFastTimer::End()
372 {
373  CCycleCount cnt;
374  cnt.Sample();
375  if ( IsX360() )
376  {
377  // have to handle rollover, hires timer is only accurate to 32 bits
378  // more than one overflow should not have occurred, otherwise caller should use a slower timer
379  if ( (uint64)cnt.m_Int64 <= (uint64)m_Duration.m_Int64 )
380  {
381  // rollover occurred
382  cnt.m_Int64 += 0x100000000LL;
383  }
384  }
385 
386  m_Duration.m_Int64 = cnt.m_Int64 - m_Duration.m_Int64;
387 
388 #ifdef DEBUG_FASTTIMER
389  m_bRunning = false;
390 #endif
391 }

393 inline CCycleCount CFastTimer::GetDurationInProgress() const
394 {
395  CCycleCount cnt;

397  if ( IsX360() )

400  // more than one overflow should not have occurred, otherwise caller should use a slower timer
401  if ( (uint64)cnt.m_Int64 <= (uint64)m_Duration.m_Int64 )
402  {
403  // rollover occurred
404  cnt.m_Int64 += 0x100000000LL;
405  }
406  }

535 //-----------------------------------------------------------------------------
536 // Purpose: If we're over our specified time period, return the amount of the overage.
537 // Output: # of microseconds since we reached our specified time period.
538 //-----------------------------------------------------------------------------
539 inline int CLimitTimer::CMicroSecOverage() const
540 {
541  CCycleCount cycleCount;
542  cycleCount.Sample();
543  uint64 lcCycles = cycleCount.GetLongCycles();
544 
545  if ( lcCycles < m_lCycleLimit )
546  return 0;
547
