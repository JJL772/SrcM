289 // Bob Jenkins's 32-bit mix function.
290 inline unsigned int Mix32HashFunctor::operator()( uint32 n ) const
291 {
292  // Perform a mixture of the bits in n, where each bit
293  // of the input value has an equal chance to affect each
294  // bit of the output. This turns tightly clustered input
295  // values into a smooth distribution.
296  //
297  // This takes 16-20 cycles on modern x86 architectures;
298  // that's roughly the same cost as a mispredicted branch.
299  // It's also reasonably efficient on PPC-based consoles.
300  //
301  // If you're still thinking, "too many instructions!",
302  // do keep in mind that reading one byte of uncached RAM
303  // is about 30x slower than executing this code. It pays
304  // to have a good hash function which minimizes collisions
305  // (and therefore long lookup chains).
306  n = ( n + 0x7ed55d16 ) + ( n << 12 );
307  n = ( n ^ 0xc761c23c ) ^ ( n >> 19 );
308  n = ( n + 0x165667b1 ) + ( n << 5 );
309  n = ( n + 0xd3a2646c ) ^ ( n << 9 );
310  n = ( n + 0xfd7046c5 ) + ( n << 3 );
311  n = ( n ^ 0xb55a4f09 ) ^ ( n >> 16 );
312  return n;
313 }

315 // Based on the widely-used FNV-1A string hash with a final
316 // mixing step to improve dispersion for very small and very
317 // large hash table sizes.
318 inline unsigned int StringHashFunctor::operator()( const char* s ) const
319 {
320  uint32 h = 2166136261u;
321  for ( ; *s; ++s )
322  {
323  uint32 c = (unsigned char) *s;
324  h = (h ^ c) * 16777619;
325  }
326  return (h ^ (h << 17)) + (h >> 21);
327 }
328 
329 // Equivalent to StringHashFunctor on lower-case strings.
330 inline unsigned int CaselessStringHashFunctor::operator()( const char* s ) const
331 {
332  uint32 h = 2166136261u;
333  for ( ; *s; ++s )
334  {
335  uint32 c = (unsigned char) *s;
336  // Brutally fast branchless ASCII tolower():
337  // if ((c >= 'A') && (c <= 'Z')) c += ('a' - 'A');
338  c += (((('A'-1) - c) & (c - ('Z'+1))) >> 26) & 32;
339  h = (h ^ c) * 16777619;
340  }
341  return (h ^ (h << 17)) + (h >> 21);
342 }
