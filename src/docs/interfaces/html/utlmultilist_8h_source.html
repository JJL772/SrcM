325 template <class T, class I>
326 inline bool CUtlMultiList<T,I>::IndexInRange( int index ) // Static method
327 {
328  // Since I is not necessarily the type returned by M (int), we need to check that M returns
329  // indices which are representable by I. A common case is 'I === unsigned short', in which case
330  // case CUtlMemory will have 'InvalidIndex == (int)-1' (which casts to 65535 in I), and will
331  // happily return elements at index 65535 and above.
332 
333  // Do a couple of static checks here: the invalid index should be (I)~0 given how we use m_MaxElementIndex,
334  // and 'I' should be unsigned (to avoid signed arithmetic errors for plausibly exhaustible ranges).
335  COMPILE_TIME_ASSERT( (I)M::INVALID_INDEX == (I)~0 );
336  COMPILE_TIME_ASSERT( ( sizeof(I) > 2 ) || ( ( (I)-1 ) > 0 ) );
337 
338  return ( ( (I)index == index ) && ( (I)index != InvalidIndex() ) );
339 }

341 template <class T, class I>
342 inline bool CUtlMultiList<T,I>::IsValidIndex( I i ) const
343 {
344  // GCC warns if I is an unsigned type and we do a ">= 0" against it (since the comparison is always 0).
345  // We get the warning even if we cast inside the expression. It only goes away if we assign to another variable.
346  long x = i;
347 
348  return (i < m_MaxElementIndex) && (x >= 0) &&
349  ((m_Memory[i].m_Previous != i) || (m_Memory[i].m_Next == i));
350 }

390 //-----------------------------------------------------------------------------
391 // Node allocation/deallocation
392 //-----------------------------------------------------------------------------
393 template <class T, class I>
394 I CUtlMultiList<T,I>::Alloc( )
395 {
396  I elem;
397  if (m_FirstFree == InvalidIndex())
398  {
399  // We can overflow before the utlmemory overflows, since we have have I != int
400  if ( !IndexInRange( m_MaxElementIndex ) )
401  {
402  // We rarely if ever handle alloc failure. Continuing leads to corruption.
403  Error( "CUtlMultiList overflow! (exhausted index range)\n" );
404  return InvalidIndex();
405  }
406 
407  // Nothing in the free list; add.
408  // Since nothing is in the free list, m_TotalElements == total # of elements
409  // the list knows about.
410  if (m_MaxElementIndex == m_Memory.NumAllocated())
411  {
412  m_Memory.Grow();
413  ResetDbgInfo();
414 
415  if ( m_MaxElementIndex >= m_Memory.NumAllocated() )
416  {
417  // We rarely if ever handle alloc failure. Continuing leads to corruption.
418  Error( "CUtlMultiList overflow! (exhausted memory allocator)\n" );
419  return InvalidIndex();
420  }
421  }
422 
423  elem = (I)m_MaxElementIndex;
424  ++m_MaxElementIndex;
425  }
426  else
427  {
428  elem = m_FirstFree;
429  m_FirstFree = InternalElement(m_FirstFree).m_Next;
430  }
431 
432  // Mark the element as not being in a list
433  InternalElement(elem).m_Next = InternalElement(elem).m_Previous = elem;
434 
435  ++m_TotalElements;
436 
437  Construct( &Element(elem) );
438 
439  return elem;
440 }

466 //-----------------------------------------------------------------------------
467 // list modification
468 //-----------------------------------------------------------------------------
469 template <class T, class I>
470 void CUtlMultiList<T,I>::LinkBefore( ListHandle_t list, I before, I elem )
471 {
472  Assert( IsValidIndex(elem) && IsValidList(list) );
473 
474  // Unlink it if it's in the list at the moment
475  Unlink(list, elem);
476 
477  ListElem_t& newElem = InternalElement(elem);
478 
479  // The element *after* our newly linked one is the one we linked before.
480  newElem.m_Next = before;
481 
482  if (before == InvalidIndex())
483  {
484  // In this case, we're linking to the end of the list, so reset the tail
485  newElem.m_Previous = m_List[list].m_Tail;
486  m_List[list].m_Tail = elem;
487  }
488  else
489  {
490  // Here, we're not linking to the end. Set the prev pointer to point to
491  // the element we're linking.
492  Assert( IsInList(before) );
493  ListElem_t& beforeElem = InternalElement(before);
494  newElem.m_Previous = beforeElem.m_Previous;
495  beforeElem.m_Previous = elem;
496  }
497 
498  // Reset the head if we linked to the head of the list
499  if (newElem.m_Previous == InvalidIndex())
500  m_List[list].m_Head = elem;
501  else
502  InternalElement(newElem.m_Previous).m_Next = elem;
503 
504  // one more element baby
505  ++m_List[list].m_Count;
506 
507  // Store the element into the list
508  if (m_pElementList)
509  m_pElementList[elem] = list;
510 }

554 template <class T, class I>
555 void CUtlMultiList<T,I>::Unlink( ListHandle_t list, I elem )
556 {
557  Assert( IsValidIndex(elem) && IsValidList(list) );
558 
559  if (IsInList(elem))
560  {
561  // Make sure the element is in the right list
562  Assert( IsElementInList( list, elem ) );
563  ListElem_t& oldElem = InternalElement(elem);
564 
565  // If we're the first guy, reset the head
566  // otherwise, make our previous node's next pointer = our next
567  if (oldElem.m_Previous != InvalidIndex())
568  InternalElement(oldElem.m_Previous).m_Next = oldElem.m_Next;
569  else
570  m_List[list].m_Head = oldElem.m_Next;
571 
572  // If we're the last guy, reset the tail
573  // otherwise, make our next node's prev pointer = our prev
574  if (oldElem.m_Next != InvalidIndex())
575  InternalElement(oldElem.m_Next).m_Previous = oldElem.m_Previous;
576  else
577  m_List[list].m_Tail = oldElem.m_Previous;
578 
579  // This marks this node as not in the list,
580  // but not in the free list either
581  oldElem.m_Previous = oldElem.m_Next = elem;
582 
583  // One less puppy
584  --m_List[list].m_Count;
585 
586  // Store the element into the list
587  if (m_pElementList)
588  m_pElementList[elem] = m_List.InvalidIndex();
589  }
590 }

732 template <class T, class I>
733 void CUtlMultiList<T,I>::RemoveAll()
734 {
735  if (m_MaxElementIndex == 0)
736  return;
737 
738  // Put everything into the free list
739  I prev = InvalidIndex();
740  for (int i = (int)m_MaxElementIndex; --i >= 0; )
741  {
742  // Invoke the destructor
743  if (IsValidIndex((I)i))
744  Destruct( &Element((I)i) );
745 
746  // next points to the next free list item
747  InternalElement((I)i).m_Next = prev;
748 
749  // Indicates it's in the free list
750  InternalElement((I)i).m_Previous = (I)i;
751  prev = (I)i;
752  }
753 
754  // First free points to the first element
755  m_FirstFree = 0;
756 
757  // Clear everything else out
758  for (I list = m_List.Head(); list != m_List.InvalidIndex(); list = m_List.Next(list) )
