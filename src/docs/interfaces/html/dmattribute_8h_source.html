446 //-----------------------------------------------------------------------------
447 // helper function for determining which attributes/elements to traverse during copy/find/save/etc.
448 //-----------------------------------------------------------------------------
449 inline bool ShouldTraverse( const CDmAttribute *pAttr, TraversalDepth_t depth )
450 {
451  switch ( depth )
452  {
453  case TD_NONE:
454  return false;
455 
456  case TD_SHALLOW:
457  if ( !pAttr->IsFlagSet( FATTRIB_MUSTCOPY ) )
458  return false;
459  // fall-through intentional
460  case TD_DEEP:
461  if ( pAttr->IsFlagSet( FATTRIB_NEVERCOPY ) )
462  return false;
463  // fall-through intentional
464  case TD_ALL:
465  return true;
466  }
467 
468  Assert( 0 );
469  return false;
470 }

634 template< class T >
635 T *FindReferringElement( CDmElement *pElement, UtlSymId_t symAttrName, bool bMustBeInSameFile = true )
636 {
637  DmAttributeReferenceIterator_t i = g_pDataModel->FirstAttributeReferencingElement( pElement->GetHandle() );
638  while ( i != DMATTRIBUTE_REFERENCE_ITERATOR_INVALID )
639  {
640  CDmAttribute *pAttribute = g_pDataModel->GetAttribute( i );
641  CDmElement *pDmeParent = pAttribute->GetOwner();
642  if ( pDmeParent && pAttribute->GetNameSymbol() == symAttrName )
643  {
644  T *pParent = CastElement< T >( pDmeParent );
645  if ( pParent )
646  {
647  if ( !bMustBeInSameFile || ( pParent->GetFileId() == pElement->GetFileId() ) )
648  return pParent;
649  }
650  }
651  i = g_pDataModel->NextAttributeReferencingElement( i );
652  }
653 
654  return NULL;
655 }

657 template< class T >
658 T *FindAncestorReferencingElement( CDmElement *target )
659 {
660  if ( !target )
661  return NULL;
662 
663  for ( DmAttributeReferenceIterator_t it = g_pDataModel->FirstAttributeReferencingElement( target->GetHandle() );
664  it != DMATTRIBUTE_REFERENCE_ITERATOR_INVALID;
665  it = g_pDataModel->NextAttributeReferencingElement( it ) )
666  {
667  CDmAttribute *attr = g_pDataModel->GetAttribute( it );
668  Assert( attr );
669  CDmElement *element = attr->GetOwner();
670  Assert( element );
671  if ( !element )
672  continue;
673  T *t = CastElement< T >( element );
674  if ( !t )
675  continue;
676 
677  return t;
678  }
679  return NULL;
680 }

682 template< class T >
683 T *FindAncestorReferencingElement_R_Impl( CUtlRBTree< CDmElement * >& visited, CDmElement *check )
684 {
685  if ( visited.Find( check ) != visited.InvalidIndex() )
686  return NULL;
687 
688  visited.Insert( check );
689 
690  // Pass one, see if it's in this ancestor list
691  DmAttributeReferenceIterator_t it;
692  for ( it = g_pDataModel->FirstAttributeReferencingElement( check->GetHandle() );
693  it != DMATTRIBUTE_REFERENCE_ITERATOR_INVALID;
694  it = g_pDataModel->NextAttributeReferencingElement( it ) )
695  {
696  CDmAttribute *attr = g_pDataModel->GetAttribute( it );
697  Assert( attr );
698  CDmElement *element = attr->GetOwner();
699  Assert( element );
700  if ( !element )
701  continue;
702  T *t = CastElement< T >( element );
703  if ( !t )
704  continue;
705 
706  return t;
707  }
708 
709  for ( it = g_pDataModel->FirstAttributeReferencingElement( check->GetHandle() );
710  it != DMATTRIBUTE_REFERENCE_ITERATOR_INVALID;
711  it = g_pDataModel->NextAttributeReferencingElement( it ) )
712  {
713  CDmAttribute *attr = g_pDataModel->GetAttribute( it );
714  Assert( attr );
715  CDmElement *element = attr->GetOwner();
716  Assert( element );
717  if ( !element )
718  continue;
719 
720  T *found = FindAncestorReferencingElement_R_Impl< T >( visited, element );
721  if ( found )
722  return found;
723  }
724  return NULL;
725 }
