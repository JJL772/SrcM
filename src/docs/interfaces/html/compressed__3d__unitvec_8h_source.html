55  void packVector( const Vector& vec )
56  {
57  // convert from Vector to cUnitVector
58 
59  Assert( vec.IsValid());
60  Vector tmp = vec;
61 
62  // input vector does not have to be unit length
63  // Assert( tmp.length() <= 1.001f );
64 
65  mVec = 0;
66  if ( tmp.x < 0 ) { mVec |= XSIGN_MASK; tmp.x = -tmp.x; }
67  if ( tmp.y < 0 ) { mVec |= YSIGN_MASK; tmp.y = -tmp.y; }
68  if ( tmp.z < 0 ) { mVec |= ZSIGN_MASK; tmp.z = -tmp.z; }
69 
70  // project the normal onto the plane that goes through
71  // X0=(1,0,0),Y0=(0,1,0),Z0=(0,0,1).
72  // on that plane we choose an (projective!) coordinate system
73  // such that X0->(0,0), Y0->(126,0), Z0->(0,126),(0,0,0)->Infinity
74 
75  // a little slower... old pack was 4 multiplies and 2 adds.
76  // This is 2 multiplies, 2 adds, and a divide....
77  float w = 126.0f / ( tmp.x + tmp.y + tmp.z );
78  long xbits = (long)( tmp.x * w );
79  long ybits = (long)( tmp.y * w );
80 
81  Assert( xbits < 127 );
82  Assert( xbits >= 0 );
83  Assert( ybits < 127 );
84  Assert( ybits >= 0 );
85 
86  // Now we can be sure that 0<=xp<=126, 0<=yp<=126, 0<=xp+yp<=126
87  // however for the sampling we want to transform this triangle
88  // into a rectangle.
89  if ( xbits >= 64 )
90  {
91  xbits = 127 - xbits;
92  ybits = 127 - ybits;
93  }
94 
95  // now we that have xp in the range (0,127) and yp in
96  // the range (0,63), we can pack all the bits together
97  mVec |= ( xbits << 7 );
98  mVec |= ybits;
99  }

101  void unpackVector( Vector& vec )
102  {
103  // if we do a straightforward backward transform
104  // we will get points on the plane X0,Y0,Z0
105  // however we need points on a sphere that goes through
106  // these points. Therefore we need to adjust x,y,z so
107  // that x^2+y^2+z^2=1 by normalizing the vector. We have
108  // already precalculated the amount by which we need to
109  // scale, so all we do is a table lookup and a
110  // multiplication
111 
112  // get the x and y bits
113  long xbits = (( mVec & TOP_MASK ) >> 7 );
114  long ybits = ( mVec & BOTTOM_MASK );
115 
116  // map the numbers back to the triangle (0,0)-(0,126)-(126,0)
117  if (( xbits + ybits ) >= 127 )
118  {
119  xbits = 127 - xbits;
120  ybits = 127 - ybits;
121  }
122 
123  // do the inverse transform and normalization
124  // costs 3 extra multiplies and 2 subtracts. No big deal.
125  float uvadj = mUVAdjustment[mVec & ~SIGN_MASK];
126  vec.x = uvadj * (float) xbits;
127  vec.y = uvadj * (float) ybits;
128  vec.z = uvadj * (float)( 126 - xbits - ybits );
129 
130  // set all the sign bits
131  if ( mVec & XSIGN_MASK ) vec.x = -vec.x;
132  if ( mVec & YSIGN_MASK ) vec.y = -vec.y;
133  if ( mVec & ZSIGN_MASK ) vec.z = -vec.z;
134 
135  Assert( vec.IsValid());
136  }

138  static void initializeStatics()
139  {
140  for ( int idx = 0; idx < 0x2000; idx++ )
141  {
142  long xbits = idx >> 7;
143  long ybits = idx & BOTTOM_MASK;
144 
145  // map the numbers back to the triangle (0,0)-(0,127)-(127,0)
146  if (( xbits + ybits ) >= 127 )
147  {
148  xbits = 127 - xbits;
149  ybits = 127 - ybits;
150  }
151 
152  // convert to 3D vectors
153  float x = (float)xbits;
154  float y = (float)ybits;
155  float z = (float)( 126 - xbits - ybits );
156 
157  // calculate the amount of normalization required
158  mUVAdjustment[idx] = 1.0f / sqrtf( y*y + z*z + x*x );
159  Assert( _finite( mUVAdjustment[idx]));
