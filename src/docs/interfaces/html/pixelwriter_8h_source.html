410 FORCEINLINE_PIXEL void CPixelWriter::WritePixelNoAdvanceF( float r, float g, float b, float a )
411 {
412  Assert( IsUsingFloatFormat() );
413 
414  // X360TBD: Not ported
415  Assert( IsPC() );
416 
417  if (PIXELWRITER_USING_16BIT_FLOAT_FORMAT & m_nFlags)
418  {
419  float16 fp16[4];
420  fp16[0].SetFloat( r );
421  fp16[1].SetFloat( g );
422  fp16[2].SetFloat( b );
423  fp16[3].SetFloat( a );
424  // fp16
425  unsigned short pBuf[4] = { 0, 0, 0, 0 };
426  pBuf[ m_RShift >> 4 ] |= (fp16[0].GetBits() & m_RMask) << ( m_RShift & 0xF );
427  pBuf[ m_GShift >> 4 ] |= (fp16[1].GetBits() & m_GMask) << ( m_GShift & 0xF );
428  pBuf[ m_BShift >> 4 ] |= (fp16[2].GetBits() & m_BMask) << ( m_BShift & 0xF );
429  pBuf[ m_AShift >> 4 ] |= (fp16[3].GetBits() & m_AMask) << ( m_AShift & 0xF );
430  memcpy( m_pBits, pBuf, m_Size );
431  }
432  else
433  {
434  // fp32
435  int pBuf[4] = { 0, 0, 0, 0 };
436  pBuf[ m_RShift >> 5 ] |= (FloatBits(r) & m_RMask) << ( m_RShift & 0x1F );
437  pBuf[ m_GShift >> 5 ] |= (FloatBits(g) & m_GMask) << ( m_GShift & 0x1F );
438  pBuf[ m_BShift >> 5 ] |= (FloatBits(b) & m_BMask) << ( m_BShift & 0x1F );
439  pBuf[ m_AShift >> 5 ] |= (FloatBits(a) & m_AMask) << ( m_AShift & 0x1F );
440  memcpy( m_pBits, pBuf, m_Size );
441  }
442 }

473 //-----------------------------------------------------------------------------
474 // Writes a pixel without advancing the index
475 //-----------------------------------------------------------------------------
476 FORCEINLINE_PIXEL void CPixelWriter::WritePixelNoAdvance( int r, int g, int b, int a )
477 {
478  Assert( !IsUsingFloatFormat() );
479 
480  if ( m_Size <= 0 )
481  {
482  return;
483  }
484  if ( m_Size < 5 )
485  {
486  unsigned int val = (r & m_RMask) << m_RShift;
487  val |= (g & m_GMask) << m_GShift;
488  val |= (m_BShift > 0) ? ((b & m_BMask) << m_BShift) : ((b & m_BMask) >> -m_BShift);
489  val |= (a & m_AMask) << m_AShift;
490 
491  switch( m_Size )
492  {
493  default:
494  Assert( 0 );
495  return;
496  case 1:
497  {
498  m_pBits[0] = (unsigned char)((val & 0xff));
499  return;
500  }
501  case 2:
502  {
503  ((unsigned short *)m_pBits)[0] = (unsigned short)((val & 0xffff));
504  return;
505  }
506  case 3:
507  {
508  if ( IsPC() || !IsX360() )
509  {
510  ((unsigned short *)m_pBits)[0] = (unsigned short)((val & 0xffff));
511  m_pBits[2] = (unsigned char)((val >> 16) & 0xff);
512  }
513  else
514  {
515  m_pBits[0] = (unsigned char)(((val >> 16) & 0xff));
516  m_pBits[1] = (unsigned char)(((val >> 8 ) & 0xff));
517  m_pBits[2] = (unsigned char)(val & 0xff);
518  }
519  return;
520  }
521  case 4:
522  {
523  ((unsigned int *)m_pBits)[0] = val;
524  return;
525  }
526  }
527  }
528  else // RGBA32323232 or RGBA16161616 -- PC only.
529  {
530  AssertMsg(!IsX360(), "Unsupported lightmap format used in WritePixelNoAdvance(). This is a severe performance fault.\n");
531 
532  int64 val = ( ( int64 )(r & m_RMask) ) << m_RShift;
533  val |= ( ( int64 )(g & m_GMask) ) << m_GShift;
534  val |= (m_BShift > 0) ? ((( int64 )( b & m_BMask)) << m_BShift) : (((int64)( b & m_BMask)) >> -m_BShift);
535  val |= ( ( int64 )(a & m_AMask) ) << m_AShift;
536 
537  switch( m_Size )
538  {
539  case 6:
540  {
541  if ( IsPC() || !IsX360() )
542  {
543  ((unsigned int *)m_pBits)[0] = val & 0xffffffff;
544  ((unsigned short *)m_pBits)[2] = (unsigned short)( ( val >> 32 ) & 0xffff );
545  }
546  else
547  {
548  ((unsigned int *)m_pBits)[0] = (val >> 16) & 0xffffffff;
549  ((unsigned short *)m_pBits)[2] = (unsigned short)( val & 0xffff );
550  }
551  return;
552  }
553  case 8:
554  {
555  if ( IsPC() || !IsX360() )
556  {
557  ((unsigned int *)m_pBits)[0] = val & 0xffffffff;
558  ((unsigned int *)m_pBits)[1] = ( val >> 32 ) & 0xffffffff;
559  }
560  else
561  {
562  ((unsigned int *)m_pBits)[0] = ( val >> 32 ) & 0xffffffff;
563  ((unsigned int *)m_pBits)[1] = val & 0xffffffff;
564  }
565  return;
566  }
567  default:
568  Assert( 0 );
569  return;
570  }
571  }
572 }

633 // here are some explicit formats so we can avoid the switch:
634 FORCEINLINE void CPixelWriter::WritePixelNoAdvance_RGBA8888( FLTX4 rgba )
635 {
636  // it's easier to do tiered convert-saturates here
637  // than the d3d color convertor op
638 
639  // first permute
640  const static fltx4 permReverse = XMVectorPermuteControl(3,2,1,0);
641  fltx4 N = XMVectorPermute(rgba, rgba, permReverse);
642 
643  N = __vctuxs(N, 0); // convert to unsigned fixed point 0 w/ saturate
644  N = __vpkuwus(N, N); // convert to halfword saturate
645  N = __vpkuhus(N, N); // convert to byte saturate
646  N = __vspltw(N, 0); // splat w-word to all four
647 
648  __stvewx(N, m_pBits, 0); // store whatever word happens to be aligned with m_pBits to that word
649 }

656 FORCEINLINE void CPixelWriter::WritePixelNoAdvance_BGRA8888( FLTX4 rgba, void * RESTRICT pBits ) RESTRICT
657 {
658  // this happens to be in an order such that we can use the handy builtin packing op
659  // clamp to 0..255 (coz it might have leaked over)
660  static const fltx4 vTwoFiftyFive = {255.0f, 255.0f, 255.0f, 255.0f};
661  fltx4 N = MinSIMD(vTwoFiftyFive, rgba);
662 
663  // the magic number such that when mul-accummulated against rbga,
664  // gets us a representation 3.0 + (r)*2^-22 -- puts the bits at
665  // the bottom of the float
666  static CONST XMVECTOR PackScale = { (1.0f / (FLOAT)(1 << 22)), (1.0f / (FLOAT)(1 << 22)), (1.0f / (FLOAT)(1 << 22)), (1.0f / (FLOAT)(1 << 22))}; // 255.0f / (FLOAT)(1 << 22)
667  static const XMVECTOR Three = {3.0f, 3.0f, 3.0f, 3.0f};
668 
669  N = __vmaddfp(N, PackScale, Three);
670  N = __vpkd3d(N, N, VPACK_D3DCOLOR, VPACK_32, 3); // pack to X word
671  N = __vspltw(N, 0); // splat X
672
