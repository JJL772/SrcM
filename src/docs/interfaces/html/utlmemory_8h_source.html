189  // Can't grow a buffer whose memory was externally allocated
190  this->ConvertToGrowableMemory( m_nMallocGrowSize );
191  }
192 
193  BaseClass::EnsureCapacity( num );
194  }
195

197  int m_nMallocGrowSize;

483 {
484  if ( !IsExternallyAllocated() )
485  return;
486 
487  m_nGrowSize = nGrowSize;
488  if (m_nAllocationCount)

492 
493  int nNumBytes = m_nAllocationCount * sizeof(T);
494  T *pMemory = (T*)malloc( nNumBytes );
495  memcpy( (void*)pMemory, (void*)m_pMemory, nNumBytes );

497  }
498  else

500  m_pMemory = NULL;
501  }

659 //-----------------------------------------------------------------------------
660 // Grows the memory
661 //-----------------------------------------------------------------------------
662 inline int UtlMemory_CalcNewAllocationCount( int nAllocationCount, int nGrowSize, int nNewSize, int nBytesItem )
663 {
664  if ( nGrowSize )
665  {
666  nAllocationCount = ((1 + ((nNewSize - 1) / nGrowSize)) * nGrowSize);
667  }
668  else
669  {
670  if ( !nAllocationCount )
671  {
672  // Compute an allocation which is at least as big as a cache line...
673  nAllocationCount = (31 + nBytesItem) / nBytesItem;
674  }
675 
676  while (nAllocationCount < nNewSize)
677  {
678 #ifndef _X360
679  nAllocationCount *= 2;
680 #else
681  int nNewAllocationCount = ( nAllocationCount * 9) / 8; // 12.5 %
682  if ( nNewAllocationCount > nAllocationCount )
683  nAllocationCount = nNewAllocationCount;
684  else
685  nAllocationCount *= 2;
686 #endif
687  }
688  }
689 
690  return nAllocationCount;
691 }

694 void CUtlMemory<T,I>::Grow( int num )
695 {
696  Assert( num > 0 );
697 
698  if ( IsExternallyAllocated() )
699  {
700  // Can't grow a buffer whose memory was externally allocated
701  Assert(0);
702  return;
703  }
704 
705  // Make sure we have at least numallocated + num allocations.
706  // Use the grow rules specified for this memory (in m_nGrowSize)
707  int nAllocationRequested = m_nAllocationCount + num;
708 
709  UTLMEMORY_TRACK_FREE();
710 
711  int nNewAllocationCount = UtlMemory_CalcNewAllocationCount( m_nAllocationCount, m_nGrowSize, nAllocationRequested, sizeof(T) );
712 
713  // if m_nAllocationRequested wraps index type I, recalculate
714  if ( ( int )( I )nNewAllocationCount < nAllocationRequested )
715  {
716  if ( ( int )( I )nNewAllocationCount == 0 && ( int )( I )( nNewAllocationCount - 1 ) >= nAllocationRequested )
717  {
718  --nNewAllocationCount; // deal w/ the common case of m_nAllocationCount == MAX_USHORT + 1
719  }
720  else
721  {
722  if ( ( int )( I )nAllocationRequested != nAllocationRequested )
723  {
724  // we've been asked to grow memory to a size s.t. the index type can't address the requested amount of memory
725  Assert( 0 );
726  return;
727  }
728  while ( ( int )( I )nNewAllocationCount < nAllocationRequested )
729  {
730  nNewAllocationCount = ( nNewAllocationCount + nAllocationRequested ) / 2;
731  }
732  }
733  }
734 
735  m_nAllocationCount = nNewAllocationCount;
736 
737  UTLMEMORY_TRACK_ALLOC();
738 
739  if (m_pMemory)
740  {
741  MEM_ALLOC_CREDIT_CLASS();
742  m_pMemory = (T*)realloc( m_pMemory, m_nAllocationCount * sizeof(T) );
743  Assert( m_pMemory );
744  }
745  else
746  {
747  MEM_ALLOC_CREDIT_CLASS();
748  m_pMemory = (T*)malloc( m_nAllocationCount * sizeof(T) );
749  Assert( m_pMemory );
750  }
751 }

807 template< class T, class I >
808 void CUtlMemory<T,I>::Purge( int numElements )
809 {
810  Assert( numElements >= 0 );
811 
812  if( numElements > m_nAllocationCount )
813  {
814  // Ensure this isn't a grow request in disguise.
815  Assert( numElements <= m_nAllocationCount );
816  return;
817  }
818 
819  // If we have zero elements, simply do a purge:
820  if( numElements == 0 )
821  {
822  Purge();
823  return;
824  }
825 
826  if ( IsExternallyAllocated() )
827  {
828  // Can't shrink a buffer whose memory was externally allocated, fail silently like purge
829  return;
830  }
831 
832  // If the number of elements is the same as the allocation count, we are done.
833  if( numElements == m_nAllocationCount )
834  {
835  return;
836  }
837 
838 
839  if( !m_pMemory )
840  {
841  // Allocation count is non zero, but memory is null.
842  Assert( m_pMemory );
843  return;
844  }
845 
846  UTLMEMORY_TRACK_FREE();
847 
848  m_nAllocationCount = numElements;
849 
850  UTLMEMORY_TRACK_ALLOC();
851 
852  // Allocation count > 0, shrink it down.
853  MEM_ALLOC_CREDIT_CLASS();
854  m_pMemory = (T*)realloc( m_pMemory, m_nAllocationCount * sizeof(T) );
855 }
