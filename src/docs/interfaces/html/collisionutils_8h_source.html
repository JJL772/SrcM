48 //-----------------------------------------------------------------------------
49 //
50 // ComputeIntersectionBarycentricCoordinates
51 //
52 // Figures out the barycentric coordinates (u,v) where a ray hits a
53 // triangle. Note that this will ignore the ray extents, and it also ignores
54 // the ray length. Note that the edge from v1->v2 represents u (v2: u = 1),
55 // and the edge from v1->v3 represents v (v3: v = 1). It returns false
56 // if the ray is parallel to the triangle (or when t is specified if t is less
57 // than zero).
58 //
59 //-----------------------------------------------------------------------------
60 bool ComputeIntersectionBarycentricCoordinates( const Ray_t& ray,
61  const Vector& v1, const Vector& v2, const Vector& v3, float& u, float& v,
62  float *t = 0 );
