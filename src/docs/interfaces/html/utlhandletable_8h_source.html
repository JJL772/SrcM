26 //-----------------------------------------------------------------------------
27 // Purpose: This is a table used to allocate handles
28 // HandleBits specifies the max # of simultaneously allocated handles.
29 // An extra bit is used for the validity state
30 // The rest of the 32 bits are used for a serial number
31 //-----------------------------------------------------------------------------
32 template< class T, int HandleBits >
33 class CUtlHandleTable
34 {
35 public:
36  CUtlHandleTable();

46 
47  // Is a handle valid?
48  bool IsHandleValid( UtlHandle_t h ) const;
49 
50  // Iterate over handles; they may not be valid
51  unsigned int GetValidHandleCount() const;
52  unsigned int GetHandleCount() const;
53  UtlHandle_t GetHandleFromIndex( int i ) const;
54  int GetIndexFromHandle( UtlHandle_t h ) const;
55 
56  void MarkHandleInvalid( UtlHandle_t h );
57  void MarkHandleValid( UtlHandle_t h );
58 
59 private:
60  struct HandleType_t
61  {
62  HandleType_t( unsigned int i, unsigned int s ) : nIndex( i ), nSerial( s )
63  {
64  Assert( i < ( 1 << HandleBits ) );
65  Assert( s < ( 1 << ( 31 - HandleBits ) ) );
66  }
67  unsigned int nIndex : HandleBits;
68  unsigned int nSerial : 31 - HandleBits;
69  };

72  {
73  EntryType_t() : m_nSerial( 0 ), nInvalid( 0 ), m_pData( 0 ) {}
74  unsigned int m_nSerial : 31;
75  unsigned int nInvalid : 1;
76  T *m_pData;
77  };

99 //-----------------------------------------------------------------------------
100 // Allocate, deallocate handles
101 //-----------------------------------------------------------------------------
102 template< class T, int HandleBits >
103 UtlHandle_t CUtlHandleTable<T, HandleBits>::AddHandle()
104 {
105  unsigned int nIndex = ( m_unused.Count() > 0 ) ? m_unused.RemoveAtHead() : m_list.AddToTail();
106 
107  EntryType_t &entry = m_list[ nIndex ];
108  entry.nInvalid = 0;
109  entry.m_pData = NULL;
110 
111  ++m_nValidHandles;
112 
113  return CreateHandle( entry.m_nSerial, nIndex );
114 }
115 
116 template< class T, int HandleBits >
117 void CUtlHandleTable<T, HandleBits>::RemoveHandle( UtlHandle_t handle )
118 {
119  unsigned int nIndex = GetListIndex( handle );
120  Assert( nIndex < ( unsigned int )m_list.Count() );
121  if ( nIndex >= ( unsigned int )m_list.Count() )
122  return;
123 
124  EntryType_t &entry = m_list[ nIndex ];
125  ++entry.m_nSerial; // mark old serial# invalid
126  if ( !entry.nInvalid )
127  {
128  entry.nInvalid = 1;
129  --m_nValidHandles;
130  }
131  entry.m_pData = NULL;
132 
133 
134  // If a handle has been used this many times, then we need to take it out of service, otherwise if the
135  // serial # wraps around we'll possibly revalidate old handles and they'll start to point at the wrong objects. Unlikely, but possible.
136  bool bStopUsing = ( entry.m_nSerial >= ( (1 << ( 31 - HandleBits ) ) - 1 ) );
137  if ( !bStopUsing )
138  {
139  m_unused.Insert( nIndex );
140  }
141 }

179 //-----------------------------------------------------------------------------
180 // Is a handle valid?
181 //-----------------------------------------------------------------------------
182 template< class T, int HandleBits >
183 bool CUtlHandleTable<T, HandleBits>::IsHandleValid( UtlHandle_t handle ) const
184 {
185  if ( handle == UTLHANDLE_INVALID )
186  return false;
187 
188  unsigned int nIndex = GetListIndex( handle );
189  AssertOnce( nIndex < ( unsigned int )m_list.Count() );
190  if ( nIndex >= ( unsigned int )m_list.Count() )
191  return false;
192 
193  const EntryType_t &entry = m_list[ nIndex ];
194  if ( entry.m_nSerial != GetSerialNumber( handle ) )
195  return false;
196 
197  if ( 1 == entry.nInvalid )
198  return false;
199 
200  return true;
201 }
