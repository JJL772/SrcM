34 // Here, we show how to use this system (the ifdef UTLDELEGATE_USAGE_DEMONSTRATION is used to get syntax coloring).
35 //-----------------------------------------------------------------------------

37 // First, define the functions you wish to call.
38 int Test1( char *pString, float x );
39 class CTestClass
40 {
41 public:
42  void Test2();
43  static float Test3( int x );
44 };
45 
46 void Test()
47 {
48  CTestClass testClass;
49 
50  // CUtlDelegate is a class that can be used to invoke methods of classes
51  // or static functions in a highly efficient manner.
52 
53  // There are a couple ways to hook up a delegate. One is in a constructor
54  // Note that the template parameter of CUtlFastDelegate looks like the
55  // function type: first, you have the return type, then ( parameter list )
56  CUtlDelegate< int ( char *, float ) > delegate1( &Test1 );
57 
58  // Another way is to use the UtlMakeDelegate method, allowing you to
59  // define the delegate later. Note that UtlMakeDelegate does *not* do a heap allocation
60  CUtlDelegate< void () > delegate2;
61  delegate2 = UtlMakeDelegate( &testClass, &CTestClass::Test2 );
62 
63  // A third method is to use the Bind() method of CUtlFastDelegate
64  // Note that you do not pass in the class pointer for static functions
65  CUtlDelegate< float ( int ) > delegate3;
66  delegate3.Bind( &CTestClass::Test3 );
67 
68  // Use the () operator to invoke the function calls.
69  int x = delegate1( "hello", 1.0f );
70  delegate2();
71  float y = delegate3( 5 );
72 
73  // Use the Clear() method to unbind a delegate.
74  delegate1.Clear();
75 
76  // You can use operator! or IsEmpty() to see if a delegate is bound
77  if ( !delegate1.IsEmpty() )
78  {
79  delegate1( "hello2" );
80  }
81 
82  // Delegates maintain an internal non-templatized representation of the
83  // functions they are bound to called CUtlAbstractDelegate. These are
84  // useful when keeping a list of untyped delegates or when passing
85  // delegates across interface boundaries.
86  const CUtlAbstractDelegate &abstractDelegate3 = delegate3.GetAbstractDelegate();
87  CUtlDelegate< float ( int ) > delegate4;
88  delegate4.SetAbstractDelegate( abstractDelegate3 );
89  delegate4( 10 );
90 }
