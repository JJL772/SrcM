208 //-----------------------------------------------------------------------------
209 //-----------------------------------------------------------------------------
210 template<class Data, typename C, typename K>
211 inline bool CUtlHash<Data, C, K>::DoFind( Data const &src, unsigned int *pBucket, int *pIndex ) const
212 {
213  // generate the data "key"
214  unsigned int key = m_KeyFunc( src );
215 
216  // hash the "key" - get the correct hash table "bucket"
217  unsigned int ndxBucket;
218  if( m_bPowerOfTwo )
219  {
220  *pBucket = ndxBucket = ( key & m_ModMask );
221  }
222  else
223  {
224  int bucketCount = m_Buckets.Count();
225  *pBucket = ndxBucket = key % bucketCount;
226  }
227 
228  int ndxKeyData;
229  const CUtlVector<Data> &bucket = m_Buckets[ndxBucket];
230  int keyDataCount = bucket.Count();
231  for( ndxKeyData = 0; ndxKeyData < keyDataCount; ndxKeyData++ )
232  {
233  if( m_CompareFunc( bucket.Element( ndxKeyData ), src ) )
234  break;
235  }
236 
237  if( ndxKeyData == keyDataCount )
238  return false;
239 
240  *pIndex = ndxKeyData;
241  return true;
242 }

404 template<class Data, typename C, typename K>
405 inline UtlHashHandle_t CUtlHash<Data, C, K>::GetNextHandle( UtlHashHandle_t handle ) const
406 {
407  ++handle; // start at the first possible handle after the one given
408 
409  int bi = GetBucketIndex( handle );
410  int ki = GetKeyDataIndex( handle );
411 
412  int nBuckets = m_Buckets.Count();
413  for ( ; bi < nBuckets; ++bi )
414  {
415  if ( ki < m_Buckets[ bi ].Count() )
416  return BuildHandle( bi, ki );
417 
418  ki = 0;
419  }
420 
421  return InvalidHandle();
422 }

567 //-----------------------------------------------------------------------------
568 template<class Data, class HashFuncs> bool CUtlHashFast<Data,HashFuncs>::Init( int nBucketCount )

571  if ( !IsPowerOfTwo( nBucketCount ) )

613 //-----------------------------------------------------------------------------
614 // Purpose: Insert data into the hash table given its key (unsigned int),
615 // without a check to see if the element already exists within the tree.
616 //-----------------------------------------------------------------------------
617 template<class Data, class HashFuncs> inline UtlHashFastHandle_t CUtlHashFast<Data,HashFuncs>::FastInsert( unsigned int uiKey, const Data &data )
618 {
619  // Get a new element from the pool.
620  int iHashData = m_aDataPool.Alloc( true );
621  HashFastData_t *pHashData = &m_aDataPool[iHashData];
622  if ( !pHashData )
623  return InvalidHandle();
624 
625  // Add data to new element.
626  pHashData->m_uiKey = uiKey;
627  pHashData->m_Data = data;
628 
629  // Link element.
630  int iBucket = HashFuncs::Hash( uiKey, m_uiBucketMask );
631  m_aDataPool.LinkBefore( m_aBuckets[iBucket], iHashData );
632  m_aBuckets[iBucket] = iHashData;
633 
634  return iHashData;
635 }

637 //-----------------------------------------------------------------------------
638 // Purpose: Remove a given element from the hash.
639 //-----------------------------------------------------------------------------
640 template<class Data, class HashFuncs> inline void CUtlHashFast<Data,HashFuncs>::Remove( UtlHashFastHandle_t hHash )
641 {
642  int iBucket = HashFuncs::Hash( m_aDataPool[hHash].m_uiKey, m_uiBucketMask );
643  if ( m_aBuckets[iBucket] == hHash )
644  {
645  // It is a bucket head.
646  m_aBuckets[iBucket] = m_aDataPool.Next( hHash );
647  }
648  else
649  {
650  // Not a bucket head.
651  m_aDataPool.Unlink( hHash );
652  }
653 
654  // Remove the element.
655  m_aDataPool.Remove( hHash );
656 }
