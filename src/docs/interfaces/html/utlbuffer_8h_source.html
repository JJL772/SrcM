647 template <>
648 inline void CUtlBuffer::GetTypeBin< float >( float &dest )
649 {
650  if ( CheckGet( sizeof( float ) ) )
651  {
652  uintptr_t pData = (uintptr_t)PeekGet();
653  if ( IsX360() && ( pData & 0x03 ) )
654  {
655  // handle unaligned read
656  ((unsigned char*)&dest)[0] = ((unsigned char*)pData)[0];
657  ((unsigned char*)&dest)[1] = ((unsigned char*)pData)[1];
658  ((unsigned char*)&dest)[2] = ((unsigned char*)pData)[2];
659  ((unsigned char*)&dest)[3] = ((unsigned char*)pData)[3];
660  }
661  else
662  {
663  // aligned read
664  dest = *(float *)pData;
665  }
666  if ( m_Byteswap.IsSwappingBytes() )
667  {
668  m_Byteswap.SwapBufferToTargetEndian< float >( &dest, &dest );
669  }
670  m_Get += sizeof( float );
671  }
672  else
673  {
674  dest = 0;
