157  void WriteDelta( SoundInfo_t *delta, bf_write &buffer)
158  {
159  if ( nEntityIndex == delta->nEntityIndex )
160  {
161  buffer.WriteOneBit( 0 );
162  }
163  else
164  {
165  buffer.WriteOneBit( 1 );
166 
167  if ( nEntityIndex <= 31)
168  {
169  buffer.WriteOneBit( 1 );
170  buffer.WriteUBitLong( nEntityIndex, 5 );
171  }
172  else
173  {
174  buffer.WriteOneBit( 0 );
175  buffer.WriteUBitLong( nEntityIndex, MAX_EDICT_BITS );
176  }
177  }
178 
179  WRITE_DELTA_UINT( nSoundNum, MAX_SOUND_INDEX_BITS );
180 
181  WRITE_DELTA_UINT( nFlags, SND_FLAG_BITS_ENCODE );
182 
183  WRITE_DELTA_UINT( nChannel, 3 );
184 
185  buffer.WriteOneBit( bIsAmbient?1:0 );
186  buffer.WriteOneBit( bIsSentence?1:0 ); // NOTE: SND_STOP behavior is different depending on this flag
187 
188  if ( nFlags != SND_STOP )
189  {
190  if ( nSequenceNumber == delta->nSequenceNumber )
191  {
192  // didn't change, most often case
193  buffer.WriteOneBit( 1 );
194  }
195  else if ( nSequenceNumber == (delta->nSequenceNumber+1) )
196  {
197  // increased by one
198  buffer.WriteOneBit( 0 );
199  buffer.WriteOneBit( 1 );
200  }
201  else
202  {
203  // send full seqnr
204  buffer.WriteUBitLong( 0, 2 ); // 2 zero bits
205  buffer.WriteUBitLong( nSequenceNumber, SOUND_SEQNUMBER_BITS );
206  }
207 
208  if ( fVolume == delta->fVolume )
209  {
210  buffer.WriteOneBit( 0 );
211  }
212  else
213  {
214  buffer.WriteOneBit( 1 );
215  buffer.WriteUBitLong( (unsigned int)(fVolume*127.0f), 7 );
216  }
217 
218  WRITE_DELTA_UINT( Soundlevel, MAX_SNDLVL_BITS );
219 
220  WRITE_DELTA_UINT( nPitch, 8 );
221 
222  WRITE_DELTA_UINT( nSpecialDSP, 8 );
223 
224  if ( fDelay == delta->fDelay )
225  {
226  buffer.WriteOneBit( 0 );
227  }
228  else
229  {
230  buffer.WriteOneBit( 1 );
231 
232  // skipahead works in 10 ms increments
233  // bias results so that we only incur the precision loss on relatively large skipaheads
234  fDelay += SOUND_DELAY_OFFSET;
235 
236  // Convert to msecs
237  int iDelay = fDelay * 1000.0f;
238 
239  iDelay = clamp( iDelay, (int)(-10 * MAX_SOUND_DELAY_MSEC), (int)(MAX_SOUND_DELAY_MSEC) );
240 
241  if ( iDelay < 0 )
242  {
243  iDelay /=10;
244  }
245 
246  buffer.WriteSBitLong( iDelay , MAX_SOUND_DELAY_MSEC_ENCODE_BITS );
247  }
248 
249  // don't transmit sounds with high precision
250  WRITE_DELTA_SINT_SCALE( vOrigin.x, 8.0f, COORD_INTEGER_BITS - 2 );
251  WRITE_DELTA_SINT_SCALE( vOrigin.y, 8.0f, COORD_INTEGER_BITS - 2 );
252  WRITE_DELTA_SINT_SCALE( vOrigin.z, 8.0f, COORD_INTEGER_BITS - 2 );
253 
254  WRITE_DELTA_SINT( nSpeakerEntity, MAX_EDICT_BITS + 1 );
255  }
256  else
257  {
258  ClearStopFields();
259  }
260  };

262  void ReadDelta( SoundInfo_t *delta, bf_read &buffer, int nProtoVersion )
263  {
264  if ( !buffer.ReadOneBit() )
265  {
266  nEntityIndex = delta->nEntityIndex;
267  }
268  else
269  {
270  if ( buffer.ReadOneBit() )
271  {
272  nEntityIndex = buffer.ReadUBitLong( 5 );
273  }
274  else
275  {
276  nEntityIndex = buffer.ReadUBitLong( MAX_EDICT_BITS );
277  }
278  }
279 
280  if ( nProtoVersion > 22 )
281  {
282  READ_DELTA_UINT( nSoundNum, MAX_SOUND_INDEX_BITS );
283  }
284  else
285  {
286  READ_DELTA_UINT( nSoundNum, 13 );
287  }
288 
289  if ( nProtoVersion > 18 )
290  {
291  READ_DELTA_UINT( nFlags, SND_FLAG_BITS_ENCODE );
292  }
293  else
294  {
295  // There was 9 flag bits for version 18 and below (prior to Halloween 2011)
296  READ_DELTA_UINT( nFlags, 9 );
297  }
298 
299  READ_DELTA_UINT( nChannel, 3 );
300 
301  bIsAmbient = buffer.ReadOneBit() != 0;
302  bIsSentence = buffer.ReadOneBit() != 0; // NOTE: SND_STOP behavior is different depending on this flag
303 
304  if ( nFlags != SND_STOP )
305  {
306  if ( buffer.ReadOneBit() != 0 )
307  {
308  nSequenceNumber = delta->nSequenceNumber;
309  }
310  else if ( buffer.ReadOneBit() != 0 )
311  {
312  nSequenceNumber = delta->nSequenceNumber + 1;
313  }
314  else
315  {
316  nSequenceNumber = buffer.ReadUBitLong( SOUND_SEQNUMBER_BITS );
317  }
318 
319  if ( buffer.ReadOneBit() != 0 )
320  {
321  fVolume = (float)buffer.ReadUBitLong( 7 )/127.0f;
322  }
323  else
324  {
325  fVolume = delta->fVolume;
326  }
327 
328  if ( buffer.ReadOneBit() != 0 )
329  {
330  Soundlevel = (soundlevel_t)buffer.ReadUBitLong( MAX_SNDLVL_BITS );
331  }
332  else
333  {
334  Soundlevel = delta->Soundlevel;
335  }
336 
337  READ_DELTA_UINT( nPitch, 8 );
338 
339  if ( nProtoVersion > 21 )
340  {
341  // These bit weren't written in version 19 and below
342  READ_DELTA_UINT( nSpecialDSP, 8 );
343  }
344 
345  if ( buffer.ReadOneBit() != 0 )
346  {
347  // Up to 4096 msec delay
348  fDelay = (float)buffer.ReadSBitLong( MAX_SOUND_DELAY_MSEC_ENCODE_BITS ) / 1000.0f; ;
349 
350  if ( fDelay < 0 )
351  {
352  fDelay *= 10.0f;
353  }
354  // bias results so that we only incur the precision loss on relatively large skipaheads
355  fDelay -= SOUND_DELAY_OFFSET;
356  }
357  else
358  {
359  fDelay = delta->fDelay;
360  }
361 
362  READ_DELTA_SINT_SCALE( vOrigin.x, 8.0f, COORD_INTEGER_BITS - 2 );
363  READ_DELTA_SINT_SCALE( vOrigin.y, 8.0f, COORD_INTEGER_BITS - 2 );
364  READ_DELTA_SINT_SCALE( vOrigin.z, 8.0f, COORD_INTEGER_BITS - 2 );
365 
366  READ_DELTA_SINT( nSpeakerEntity, MAX_EDICT_BITS + 1 );
367  }
368  else
369  {
370  ClearStopFields();
371  }
372  }
