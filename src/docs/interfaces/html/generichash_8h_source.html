29 FORCEINLINE uint32 HashIntAlternate( uint32 n)
30 {
31  n = ( n + 0x7ed55d16 ) + ( n << 12 );
32  n = ( n ^ 0xc761c23c ) ^ ( n >> 19 );
33  n = ( n + 0x165667b1 ) + ( n << 5 );
34  n = ( n + 0xd3a2646c ) ^ ( n << 9 );
35  n = ( n + 0xfd7046c5 ) + ( n << 3 );
36  n = ( n ^ 0xb55a4f09 ) ^ ( n >> 16 );

38 }

40 inline unsigned HashIntConventional( const int n ) // faster but less effective
