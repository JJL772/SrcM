86  // ZigZag Transform: Encodes signed integers so that they can be
87  // effectively used with varint encoding.
88  //
89  // varint operates on unsigned integers, encoding smaller numbers into
90  // fewer bytes. If you try to use it on a signed integer, it will treat
91  // this number as a very large unsigned integer, which means that even
92  // small signed numbers like -1 will take the maximum number of bytes
93  // (10) to encode. ZigZagEncode() maps signed integers to unsigned
94  // in such a way that those with a small absolute value will have smaller
95  // encoded values, making them appropriate for encoding using varint.
96  //
97  // int32 -> uint32
98  // -------------------------
99  // 0 -> 0
100  // -1 -> 1
101  // 1 -> 2
102  // -2 -> 3
103  // ... -> ...
104  // 2147483647 -> 4294967294
105  // -2147483648 -> 4294967295
106  //
107  // >> encode >>
108  // << decode <<
109 
110  inline uint32 ZigZagEncode32(int32 n)
111  {
112  // Note: the right-shift must be arithmetic
113  return(n << 1) ^ (n >> 31);
114  }
115 
116  inline int32 ZigZagDecode32(uint32 n)
117  {
118  return(n >> 1) ^ -static_cast<int32>(n & 1);
119  }
120 
121  inline uint64 ZigZagEncode64(int64 n)
122  {
123  // Note: the right-shift must be arithmetic
124  return(n << 1) ^ (n >> 63);
125  }
126 
127  inline int64 ZigZagDecode64(uint64 n)
128  {
129  return(n >> 1) ^ -static_cast<int64>(n & 1);
130  }
131 
132  const int kMaxVarintBytes = 10;
133  const int kMaxVarint32Bytes = 5;
134 }

390 BITBUF_INLINE void bf_write::WriteUBitLong( unsigned int curData, int numbits, bool bCheckRange ) RESTRICT
391 {
392 #ifdef _DEBUG
393  // Make sure it doesn't overflow.
394  if ( bCheckRange && numbits < 32 )
395  {
396  if ( curData >= (unsigned long)(1 << numbits) )
397  {
398  CallErrorHandler( BITBUFERROR_VALUE_OUT_OF_RANGE, GetDebugName() );
399  }
400  }
401  Assert( numbits >= 0 && numbits <= 32 );
402 #endif
403 
404  if ( GetNumBitsLeft() < numbits )
405  {
406  m_iCurBit = m_nDataBits;
407  SetOverflowFlag();
408  CallErrorHandler( BITBUFERROR_BUFFER_OVERRUN, GetDebugName() );
409  return;
410  }
411 
412  int iCurBitMasked = m_iCurBit & 31;
413  int iDWord = m_iCurBit >> 5;
414  m_iCurBit += numbits;
415 
416  // Mask in a dword.
417  Assert( (iDWord*4 + sizeof(long)) <= (unsigned int)m_nDataBytes );
418  unsigned long * RESTRICT pOut = &m_pData[iDWord];
419 
420  // Rotate data into dword alignment
421  curData = (curData << iCurBitMasked) | (curData >> (32 - iCurBitMasked));
422 
423  // Calculate bitmasks for first and second word
424  unsigned int temp = 1 << (numbits-1);
425  unsigned int mask1 = (temp*2-1) << iCurBitMasked;
426  unsigned int mask2 = (temp-1) >> (31 - iCurBitMasked);
427 
428  // Only look beyond current word if necessary (avoid access violation)
429  int i = mask2 & 1;
430  unsigned long dword1 = LoadLittleDWord( pOut, 0 );
431  unsigned long dword2 = LoadLittleDWord( pOut, i );
432 
433  // Drop bits into place
434  dword1 ^= ( mask1 & ( curData ^ dword1 ) );
435  dword2 ^= ( mask2 & ( curData ^ dword2 ) );
436 
437  // Note reversed order of writes so that dword1 wins if mask2 == 0 && i == 0
438  StoreLittleDWord( pOut, i, dword2 );
439  StoreLittleDWord( pOut, 0, dword1 );
440 }

442 // writes an unsigned integer with variable bit length
443 BITBUF_INLINE void bf_write::WriteUBitVar( unsigned int data )
444 {
445  /* Reference:
446  if ( data < 0x10u )
447  WriteUBitLong( 0, 2 ), WriteUBitLong( data, 4 );
448  else if ( data < 0x100u )
449  WriteUBitLong( 1, 2 ), WriteUBitLong( data, 8 );
450  else if ( data < 0x1000u )
451  WriteUBitLong( 2, 2 ), WriteUBitLong( data, 12 );
452  else
453  WriteUBitLong( 3, 2 ), WriteUBitLong( data, 32 );
454  */
455  // a < b ? -1 : 0 translates into a CMP, SBB instruction pair
456  // with no flow control. should also be branchless on consoles.
457  int n = (data < 0x10u ? -1 : 0) + (data < 0x100u ? -1 : 0) + (data < 0x1000u ? -1 : 0);
458  WriteUBitLong( data*4 + n + 3, 6 + n*4 + 12 );
459  if ( data >= 0x1000u )
460  {
461  WriteUBitLong( data >> 16, 16 );
462  }
463 }
