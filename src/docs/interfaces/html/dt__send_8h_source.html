22 // ------------------------------------------------------------------------ //
23 // Send proxies can be used to convert a variable into a networkable type
24 // (a good example is converting an edict pointer into an integer index).
25 
26 // These allow you to translate data. For example, if you had a user-entered
27 // string number like "10" (call the variable pUserStr) and wanted to encode
28 // it as an integer, you would use a SendPropInt32 and write a proxy that said:
29 // pOut->m_Int = atoi(pUserStr);
30 
31 // pProp : the SendProp that has the proxy
32 // pStructBase : the base structure (like CBaseEntity*).
33 // pData : the address of the variable to proxy.
34 // pOut : where to output the proxied value.
35 // iElement : the element index if this data is part of an array (or 0 if not).
36 // objectID : entity index for debugging purposes.
37 
38 // Return false if you don't want the engine to register and send a delta to
39 // the clients for this property (regardless of whether it actually changed or not).
40 // ------------------------------------------------------------------------ //
41 typedef void (*SendVarProxyFn)( const SendProp *pProp, const void *pStructBase, const void *pData, DVariant *pOut, int iElement, int objectID );

536 // ------------------------------------------------------------------------------------------------------ //
537 // Use BEGIN_SEND_TABLE if you want to declare a SendTable and have it inherit all the properties from
538 // its base class. There are two requirements for this to work:
539 
540 // 1. Its base class must have a static SendTable pointer member variable called m_pClassSendTable which
541 // points to its send table. The DECLARE_SERVERCLASS and IMPLEMENT_SERVERCLASS macros do this automatically.
542 
543 // 2. Your class must typedef its base class as BaseClass. So it would look like this:
544 // class Derived : public CBaseEntity
545 // {
546 // typedef CBaseEntity BaseClass;
547 // };
548 
549 // If you don't want to interit a base class's properties, use BEGIN_SEND_TABLE_NOBASE.
550 // ------------------------------------------------------------------------------------------------------ //
551 #define BEGIN_SEND_TABLE(className, tableName) \
552  BEGIN_SEND_TABLE_NOBASE(className, tableName) \
553  SendPropDataTable("baseclass", 0, className::BaseClass::m_pClassSendTable, SendProxy_DataTableToDataTable),
554 
555 #define BEGIN_SEND_TABLE_NOBASE(className, tableName) \
556  template <typename T> int ServerClassInit(T *); \
557  namespace tableName { \
558  struct ignored; \
559  } \
560  template <> int ServerClassInit<tableName::ignored>(tableName::ignored *); \
561  namespace tableName { \
562  SendTable g_SendTable;\
563  int g_SendTableInit = ServerClassInit((tableName::ignored *)NULL); \
564  } \
565  template <> int ServerClassInit<tableName::ignored>(tableName::ignored *) \
566  { \
567  typedef className currentSendDTClass; \
568  static const char *g_pSendTableName = #tableName; \
569  SendTable &sendTable = tableName::g_SendTable; \
570  static SendProp g_SendProps[] = { \
571  SendPropInt("should_never_see_this", 0, sizeof(int)), // It adds a dummy property at the start so you can define "empty" SendTables.
572 
573 #define END_SEND_TABLE() \
574  };\
575  sendTable.Construct(g_SendProps+1, sizeof(g_SendProps) / sizeof(SendProp) - 1, g_pSendTableName);\
576  return 1; \
577  }
