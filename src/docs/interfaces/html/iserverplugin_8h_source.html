132 //-----------------------------------------------------------------------------

135 abstract_class IServerPluginHelpers
136 {
137 public:
138  // creates an onscreen menu with various option buttons
139  // The keyvalues param can contain these fields:
140  // "title" - (string) the title to show in the hud and in the title bar
141  // "msg" - (string) a longer message shown in the GameUI
142  // "color" - (color) the color to display the message in the hud (white by default)
143  // "level" - (int) the priority of this message (closer to 0 is higher), only 1 message can be outstanding at a time
144  // "time" - (int) the time in seconds this message should stay active in the GameUI (min 10 sec, max 200 sec)
145  //
146  // For DIALOG_MENU add sub keys for each option with these fields:
147  // "command" - (string) client command to run if selected
148  // "msg" - (string) button text for this option
149  //
150  virtual void CreateMessage( edict_t *pEntity, DIALOG_TYPE type, KeyValues *data, IServerPluginCallbacks *plugin ) = 0;
151  virtual void ClientCommand( edict_t *pEntity, const char *cmd ) = 0;
152 
153  // Call this to find out the value of a cvar on the client.
154  //
155  // It is an asynchronous query, and it will call IServerPluginCallbacks::OnQueryCvarValueFinished when
156  // the value comes in from the client.
157  //
158  // Store the return value if you want to match this specific query to the OnQueryCvarValueFinished call.
159  // Returns InvalidQueryCvarCookie if the entity is invalid.
160  virtual QueryCvarCookie_t StartQueryCvarValue( edict_t *pEntity, const char *pName ) = 0;
161 };
