305 //-----------------------------------------------------------------------------
306 // Is the color var white?
307 //-----------------------------------------------------------------------------
308 inline bool CBaseShader::IsWhite( int colorVar )
309 {
310  if (colorVar < 0)
311  return true;
312 
313  if (!s_ppParams[colorVar]->IsDefined())
314  return true;
315 
316  float color[3];
317  s_ppParams[colorVar]->GetVecValue( color, 3 );
318  return (color[0] >= 1.0f) && (color[1] >= 1.0f) && (color[2] >= 1.0f);
319 }
