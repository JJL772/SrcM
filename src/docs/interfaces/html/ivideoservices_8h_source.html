346 // Frame Rate Class
347 //-----------------------------------------------------------------------------
348 class VideoFrameRate_t
349 {
350  public:
351  inline VideoFrameRate_t() : m_TimeUnitsPerSecond( 0 ), m_TimeUnitsPerFrame( 1000 ) {};
352 
353  inline VideoFrameRate_t( int FPS, bool NTSC ) { SetFPS( FPS, NTSC); }
354  inline explicit VideoFrameRate_t( float FPS ) { SetFPS( FPS); };
355 
356  inline VideoFrameRate_t& operator=( const VideoFrameRate_t& rhs ) { m_TimeUnitsPerSecond = rhs.m_TimeUnitsPerSecond; m_TimeUnitsPerFrame = rhs.m_TimeUnitsPerFrame; return *this; }
357  inline VideoFrameRate_t( const VideoFrameRate_t &rhs ) { *this = rhs; };
358 
359  inline void SetRaw( int timeUnitsPerSecond, int TimeUnitsPerFrame ) { m_TimeUnitsPerSecond = timeUnitsPerSecond; m_TimeUnitsPerFrame = TimeUnitsPerFrame; }
360 
361  inline float GetFPS() const { return (float) m_TimeUnitsPerSecond / (float) m_TimeUnitsPerFrame; }
362  inline int GetIntFPS() const { return (int) ( (float) m_TimeUnitsPerSecond / (float) m_TimeUnitsPerFrame + 0.5f ); }
363  inline bool IsNTSCRate() const { return ( m_TimeUnitsPerFrame == 1001 ); }
364  inline int GetUnitsPerSecond() const { return m_TimeUnitsPerSecond; }
365  inline int GetUnitsPerFrame() const { return m_TimeUnitsPerFrame; }
366 
367  inline void SetFPS( int FPS, bool NTSC ) { m_TimeUnitsPerSecond = FPS * 1000; m_TimeUnitsPerFrame = 1000 + (uint) NTSC; }
368  inline void SetFPS( float FPS ) { m_TimeUnitsPerSecond = (uint) ( FPS * 1000.0f ); m_TimeUnitsPerFrame = 1000; }
369 
370  static inline bool IsNTSC( float FPS ) { float diff = ceil(FPS) - FPS; return ( diff > 0.02f && diff < 0.05f); }
371 
372  inline void Clear() { m_TimeUnitsPerSecond = 0; m_TimeUnitsPerFrame = 1000; }
373  inline bool IsValid() { return ( m_TimeUnitsPerSecond != 0); }
374 
375  private:
376  uint32 m_TimeUnitsPerSecond;
377  uint32 m_TimeUnitsPerFrame;
378 };
