406 //-----------------------------------------------------------------------------
407 // Purpose:
408 //-----------------------------------------------------------------------------
409 inline int CDispCollTree::Nodes_GetIndexFromComponents( int x, int y )
410 {
411  int nIndex = 0;
412 
413  // Interleave bits from the x and y values to create the index
414  int iShift;
415  for( iShift = 0; x != 0; iShift += 2, x >>= 1 )
416  {
417  nIndex |= ( x & 1 ) << iShift;
418  }
419 
420  for( iShift = 1; y != 0; iShift += 2, y >>= 1 )
421  {
422  nIndex |= ( y & 1 ) << iShift;
423  }
424 
425  return nIndex;
426 }
