311 inline VertexShaderHandle_t IShaderDevice::CreateVertexShader( CUtlBuffer &buf, const char *pShaderVersion )
312 {
313  // NOTE: Text buffers are assumed to have source-code shader files
314  // Binary buffers are assumed to have compiled shader files
315  if ( buf.IsText() )
316  {
317  Assert( pShaderVersion );
318  return CreateVertexShader( (const char *)buf.Base(), buf.TellMaxPut(), pShaderVersion );
319  }
320 
321  CUtlShaderBuffer shaderBuffer( buf );
322  return CreateVertexShader( &shaderBuffer );
323 }

325 inline VertexShaderHandle_t IShaderDevice::CreateVertexShader( const char *pProgram, size_t nBufLen, const char *pShaderVersion )
326 {
327  VertexShaderHandle_t hVertexShader = VERTEX_SHADER_HANDLE_INVALID;
328  IShaderBuffer* pShaderBuffer = CompileShader( pProgram, nBufLen, pShaderVersion );
329  if ( pShaderBuffer )
330  {
331  hVertexShader = CreateVertexShader( pShaderBuffer );
332  pShaderBuffer->Release();
333  }
334  return hVertexShader;
335 }
