461 // check if a point is in the field of a view of an object. supports up to 180 degree fov.
462 FORCEINLINE bool PointWithinViewAngle( Vector const &vecSrcPosition,
463  Vector const &vecTargetPosition,
464  Vector const &vecLookDirection, float flCosHalfFOV )
465 {
466  Vector vecDelta = vecTargetPosition - vecSrcPosition;
467  float cosDiff = DotProduct( vecLookDirection, vecDelta );
468 
469  if ( cosDiff < 0 )
470  return false;
471 
472  float flLen2 = vecDelta.LengthSqr();
473 
474  // a/sqrt(b) > c == a^2 > b * c ^2
475  return ( cosDiff * cosDiff > flLen2 * flCosHalfFOV * flCosHalfFOV );
476 
477 }
