18 // Purpose: class for keeping track of all the references that exist to an object. When the object
19 // being referenced is freed, all of the references pointing at it will become null.
20 //
21 // To Use:
22 // Add a DECLARE_REFERENCED_CLASS to the class that you want to use CutlReferences with.
23 // Replace pointers to that class with CUtlReferences.
24 // Check these references for null in appropriate places.
25 //
26 // NOTE : You can still happily use pointers instead of references where you want to - these
27 // pointers will not magically become null like references would, but if you know no one is going
28 // to delete the underlying object during a partcular section of code, it doesn't
29 // matter. Basically, CUtlReferences don't rely on every use of an object using one.

34 template<class T> class CUtlReference
35 {
36 public:
37  FORCEINLINE CUtlReference(void)
38  {
39  m_pNext = m_pPrev = NULL;
40  m_pObject = NULL;
41  }

109

111 
112  FORCEINLINE void AddRef( T *pObj )
113  {
114  m_pObject = pObj;
115  if ( pObj )
116  {
117  pObj->m_References.AddToHead( this );
118  }
119  }
120 
121  FORCEINLINE void KillRef(void)
122  {
123  if ( m_pObject )
124  {
125  m_pObject->m_References.RemoveNode( this );
126  m_pObject = NULL;
127  }
128  }

132 template<class T> class CUtlReferenceList : public CUtlIntrusiveDList< CUtlReference<T> >
133 {
134 public:
135  ~CUtlReferenceList( void )
136  {
137  CUtlReference<T> *i = CUtlIntrusiveDList<CUtlReference<T> >::m_pHead;
138  while( i )
139  {
140  CUtlReference<T> *n = i->m_pNext;
141  i->m_pNext = NULL;
142  i->m_pPrev = NULL;
143  i->m_pObject = NULL;
144  i = n;
145  }
146  CUtlIntrusiveDList<CUtlReference<T> >::m_pHead = NULL;
147  }
148 };
