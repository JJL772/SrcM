27 // strided pointers. gives you a class that acts like a pointer, but the ++ and += operators do the
28 // right thing
29 template<class T> class CStridedPtr
30 {
31 protected:
32  T *m_pData;
33  size_t m_nStride;
34 
35 public:
36  FORCEINLINE CStridedPtr<T>( void *pData, size_t nByteStride )
37  {
38  m_pData = reinterpret_cast<T *>( pData );
39  m_nStride = nByteStride / sizeof( T );
40  }
41 
42  FORCEINLINE CStridedPtr<T>( void ) {}
43  T *operator->(void) const
44  {
45  return m_pData;
46  }
47 
48  T & operator*(void) const
49  {
50  return *m_pData;
51  }
52 
53  FORCEINLINE operator T *(void)
54  {
55  return m_pData;
56  }
57 
58  FORCEINLINE CStridedPtr<T> & operator++(void)
59  {
60  m_pData += m_nStride;
61  return *this;
62  }
63 
64  FORCEINLINE void operator+=( size_t nNumElements )
65  {
66  m_pData += nNumElements * m_nStride;
67  }
68 
69 };

179  // set the data type for an attribute. If you set the data type, but tell it not to allocate,
180  // the data type will be set but writes will assert, and reads will give you back zeros.
181  FORCEINLINE void SetAttributeType( int nAttrIdx, EAttributeDataType nDataType, bool bAllocateMemory = true )
182  {
183  Assert( !m_pDataMemory ); // can't change after memory allocated
184  Assert( nAttrIdx < MAX_SOA_FIELDS );
185  m_nDataType[nAttrIdx] = nDataType;
186  if ( ( m_nDataType[nAttrIdx] != ATTRDATATYPE_NONE ) && bAllocateMemory )
187  m_nFieldPresentMask |= ( 1 << nAttrIdx );
188  else
189  m_nFieldPresentMask &= ~( 1 << nAttrIdx );
190  }
