145 //-----------------------------------------------------------------------------
146 // Set the size by which the memory grows - round up to the next power of 2
147 //-----------------------------------------------------------------------------
148 template< class T, class I >
149 void CUtlBlockMemory<T,I>::Init( int nGrowSize /* = 0 */, int nInitSize /* = 0 */ )
150 {
151  Purge();
152 
153  if ( nGrowSize == 0)
154  {
155  // default grow size is smallest size s.t. c++ allocation overhead is ~6% of block size
156  nGrowSize = ( 127 + sizeof( T ) ) / sizeof( T );
157  }
158  nGrowSize = SmallestPowerOfTwoGreaterOrEqual( nGrowSize );
159  m_nIndexMask = nGrowSize - 1;
160 
161  m_nIndexShift = 0;
162  while ( nGrowSize > 1 )
163  {
164  nGrowSize >>= 1;
165  ++m_nIndexShift;
166  }
167  Assert( m_nIndexMask + 1 == ( 1 << m_nIndexShift ) );
168 
169  Grow( nInitSize );
170 }

230 {
231  if ( num <= 0 )

240 template< class T, class I >
241 void CUtlBlockMemory<T,I>::ChangeSize( int nBlocks )
242 {
243  UTLBLOCKMEMORY_TRACK_FREE(); // this must stay before the recalculation of m_nBlocks, since it implicitly uses the old value
244 
245  int nBlocksOld = m_nBlocks;
246  m_nBlocks = nBlocks;
247 
248  UTLBLOCKMEMORY_TRACK_ALLOC(); // this must stay after the recalculation of m_nBlocks, since it implicitly uses the new value
249 
250  if ( m_pMemory )
251  {
252  // free old blocks if shrinking
253  // Only possible if m_pMemory is non-NULL (and avoids PVS-Studio warning)
254  for ( int i = m_nBlocks; i < nBlocksOld; ++i )
255  {
256  UTLBLOCKMEMORY_TRACK_FREE();
257  free( (void*)m_pMemory[ i ] );
258  }
259 
260  MEM_ALLOC_CREDIT_CLASS();
261  m_pMemory = (T**)realloc( m_pMemory, m_nBlocks * sizeof(T*) );
262  Assert( m_pMemory );
263  }
264  else
265  {
266  MEM_ALLOC_CREDIT_CLASS();
267  m_pMemory = (T**)malloc( m_nBlocks * sizeof(T*) );
268  Assert( m_pMemory );
269  }
270 
271  if ( !m_pMemory )
272  {
273  Error( "CUtlBlockMemory overflow!\n" );
274  }
275 
276  // allocate new blocks if growing
277  int nBlockSize = NumElementsInBlock();
278  for ( int i = nBlocksOld; i < m_nBlocks; ++i )
279  {
280  MEM_ALLOC_CREDIT_CLASS();
281  m_pMemory[ i ] = (T*)malloc( nBlockSize * sizeof( T ) );
282  Assert( m_pMemory[ i ] );
283  }
284 }

302 {
303  if ( !m_pMemory )
304  return;

306  for ( int i = 0; i < m_nBlocks; ++i )
307  {
308  UTLBLOCKMEMORY_TRACK_FREE();
309  free( (void*)m_pMemory[ i ] );
310  }
311  m_nBlocks = 0;
312

314  free( (void*)m_pMemory );
315  m_pMemory = 0;
316 }

318 template< class T, class I >
319 void CUtlBlockMemory<T,I>::Purge( int numElements )
320 {
321  Assert( numElements >= 0 );
322 
323  int nAllocated = NumAllocated();
324  if ( numElements > nAllocated )
325  {
326  // Ensure this isn't a grow request in disguise.
327  Assert( numElements <= nAllocated );
328  return;
329  }
330 
331  if ( numElements <= 0 )
332  {
333  Purge();
334  return;
335  }
336 
337  int nBlockSize = NumElementsInBlock();
338  int nBlocksOld = m_nBlocks;
339  int nBlocks = ( numElements + nBlockSize - 1 ) / nBlockSize;
340 
341  // If the number of blocks is the same as the allocated number of blocks, we are done.
342  if ( nBlocks == m_nBlocks )
343  return;
344 
345  ChangeSize( nBlocks );
346 }
