201 template<int M, int N>
202 bool NMatrixMN::InverseGeneral( NMatrixMN &mInverse ) const
203 {
204  int iRow, i, j, iTemp, iTest;
205  float mul, fTest, fLargest;
206  float mat[N][2*N];
207  int rowMap[N], iLargest;
208  float *pOut, *pRow, *pScaleRow;
209 
210 
211  // Can only invert square matrices.
212  if( M != N )
213  {
214  assert( !"Tried to invert a non-square matrix" );
215  return false;
216  }
217 
218 
219  // How it's done.
220  // AX = I
221  // A = this
222  // X = the matrix we're looking for
223  // I = identity
224 
225  // Setup AI
226  for(i=0; i < N; i++)
227  {
228  const float *pIn = m[i];
229  pOut = mat[i];
230 
231  for(j=0; j < N; j++)
232  {
233  pOut[j] = pIn[j];
234  }
235 
236  for(j=N; j < 2*N; j++)
237  pOut[j] = 0;
238 
239  pOut[i+N] = 1.0f;
240 
241  rowMap[i] = i;
242  }
243 
244  // Use row operations to get to reduced row-echelon form using these rules:
245  // 1. Multiply or divide a row by a nonzero number.
246  // 2. Add a multiple of one row to another.
247  // 3. Interchange two rows.
248 
249  for(iRow=0; iRow < N; iRow++)
250  {
251  // Find the row with the largest element in this column.
252  fLargest = 0.001f;
253  iLargest = -1;
254  for(iTest=iRow; iTest < N; iTest++)
255  {
256  fTest = (float)fabs(mat[rowMap[iTest]][iRow]);
257  if(fTest > fLargest)
258  {
259  iLargest = iTest;
260  fLargest = fTest;
261  }
262  }
263 
264  // They're all too small.. sorry.
265  if(iLargest == -1)
266  {
267  return false;
268  }
269 
270  // Swap the rows.
271  iTemp = rowMap[iLargest];
272  rowMap[iLargest] = rowMap[iRow];
273  rowMap[iRow] = iTemp;
274 
275  pRow = mat[rowMap[iRow]];
276 
277  // Divide this row by the element.
278  mul = 1.0f / pRow[iRow];
279  for(j=0; j < 2*N; j++)
280  pRow[j] *= mul;
281 
282  pRow[iRow] = 1.0f; // Preserve accuracy...
283 
284  // Eliminate this element from the other rows using operation 2.
285  for(i=0; i < N; i++)
286  {
287  if(i == iRow)
288  continue;
289 
290  pScaleRow = mat[rowMap[i]];
291 
292  // Multiply this row by -(iRow*the element).
293  mul = -pScaleRow[iRow];
294  for(j=0; j < 2*N; j++)
295  {
296  pScaleRow[j] += pRow[j] * mul;
297  }
298 
299  pScaleRow[iRow] = 0.0f; // Preserve accuracy...
300  }
301  }
302 
303  // The inverse is on the right side of AX now (the identity is on the left).
304  for(i=0; i < N; i++)
305  {
306  const float *pIn = mat[rowMap[i]] + N;
307  pOut = mInverse.m[i];
308 
309  for(j=0; j < N; j++)
310  {
311  pOut[j] = pIn[j];
312  }
313  }
314 
315  return true;
316 }
