52 FORCEINLINE float CSparseMatrix::Element( int nRow, int nCol ) const
53 {
54  Assert( nCol < m_nNumCols );
55  int nCount = m_rowDescriptors[nRow].m_nNonZeroCount;
56  if ( nCount )
57  {
58  NonZeroValueDescriptor_t const *pValue = &(m_entries[m_rowDescriptors[nRow].m_nDataIndex]);
59  do
60  {
61  int nIdx = pValue->m_nColumnNumber;
62  if ( nIdx == nCol )
63  {
64  return pValue->m_flValue;
65  }
66  if ( nIdx > nCol )
67  {
68  break;
69  }
70  pValue++;
71  } while( --nCount );
72  }
73  return 0;
74 }
