154 template< bool CASE_SENSITIVE > inline bool _V_strEndsWithInner( const char *pStr, const char *pSuffix )
155 {
156  int nSuffixLen = V_strlen( pSuffix );
157  int nStringLen = V_strlen( pStr );
158  if ( nSuffixLen == 0 )
159  return true; // All strings end with the empty string (matches Java & .NET behaviour)
160  if ( nStringLen < nSuffixLen )
161  return false;
162  pStr += nStringLen - nSuffixLen;
163  if ( CASE_SENSITIVE )
164  return !V_strcmp( pStr, pSuffix );
165  else
166  return !V_stricmp( pStr, pSuffix );
167 }

874  void PopulateUTF8()
875  {
876  if ( !m_bCreatedUTF16 )
877  return; // no work to do
878  if ( m_pwch == NULL )
879  return; // don't have a UTF-16 string to convert
880  if ( m_pch != NULL )

884  // also need to leave room for the terminating NUL.
885  uint32 cbMax = 4 * static_cast<uint32>( V_wcslen( m_pwch ) ) + 1;

894  }
895  else

898  delete [] pchTemp;
899  }

902  // ensure we have done any conversion work required to farm out a
903  // UTF-16 encoded string.
904  //

906  // (one code point per UTF-8 byte). This is sometimes pessimistic,
907  // so we perform a second allocation that's just the size we need.
908  void PopulateUTF16()
909  {
910  if ( m_bCreatedUTF16 )
911  return; // no work to do
912  if ( m_pch == NULL )
913  return; // no UTF-8 string to convert
914  if ( m_pwch != NULL )
915  return; // already been converted to UTF-16; no work to do
