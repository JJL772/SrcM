163  break;
164 #endif
165  case DPT_String :
166  if ( m_pString )
167  return m_pString;
168  else
169  return "NULL";
170  break;
171  case DPT_Array :
172  Q_snprintf( text, sizeof(text), "Array" );
173  break;

176  break;
