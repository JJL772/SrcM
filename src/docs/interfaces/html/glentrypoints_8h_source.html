98 template < class FunctionType, typename Result >
99 class CDynamicFunctionOpenGLBase
100 {
101 public:
102  // Construct with a NULL function pointer. You must manually call
103  // Lookup() before you can call a dynamic function through this interface.
104  CDynamicFunctionOpenGLBase() : m_pFn(NULL) {}
105 
106  // Construct and do a lookup right away. You will need to make sure that
107  // the lookup actually succeeded, as the gl library might have failed to load
108  // or (fn) might not exist in it.
109  CDynamicFunctionOpenGLBase(const char *fn, FunctionType fallback=NULL) : m_pFn(NULL)
110  {
111  Lookup(fn, fallback);
112  }
113 
114  // Construct and do a lookup right away. See comments in Lookup() about what (okay) does.
115  CDynamicFunctionOpenGLBase(const char *fn, bool &okay, FunctionType fallback=NULL) : m_pFn(NULL)
116  {
117  Lookup(fn, okay, fallback);
118  }
119 
120  // Load library if necessary, look up symbol. Returns true and sets
121  // m_pFn on successful lookup, returns false otherwise. If the
122  // function pointer is already looked up, this return true immediately.
123  // Use Reset() first if you want to look up the symbol again.
124  // This function will return false immediately unless (okay) is true.
125  // This allows you to chain lookups like this:
126  // bool okay = true;
127  // x.Lookup(lib, "x", okay);
128  // y.Lookup(lib, "y", okay);
129  // z.Lookup(lib, "z", okay);
130  // if (okay) { printf("All functions were loaded successfully!\n"); }
131  // If you supply a fallback, it'll be used if the lookup fails (and if
132  // non-NULL, means this will always return (okay)).
133  bool Lookup(const char *fn, bool &okay, FunctionType fallback=NULL)
134  {
135  if (!okay)
136  return false;
137  else if (this->m_pFn == NULL)
138  {
139  this->m_pFn = (FunctionType) VoidFnPtrLookup_GlMgr(fn, okay, false, (void *) fallback);
140  this->SetFuncName( fn );
141  }
142  okay = m_pFn != NULL;
143  return okay;
144  }
145 
146  // Load library if necessary, look up symbol. Returns true and sets
147  // m_pFn on successful lookup, returns false otherwise. If the
148  // function pointer is already looked up, this return true immediately.
149  // Use Reset() first if you want to look up the symbol again.
150  // This function will return false immediately unless (okay) is true.
151  // If you supply a fallback, it'll be used if the lookup fails (and if
152  // non-NULL, means this will always return true).
153  bool Lookup(const char *fn, FunctionType fallback=NULL)
154  {
155  bool okay = true;
156  return Lookup(fn, okay, fallback);
157  }
158 
159  // Invalidates the current lookup. Makes the function pointer NULL. You
160  // will need to call Lookup() before you can call a dynamic function
161  // through this interface again.
162  void Reset() { m_pFn = NULL; }
163 
164  // Force this to be a specific function pointer.
165  void Force(FunctionType ptr) { m_pFn = ptr; }
166 
167  // Retrieve the actual function pointer.
168  FunctionType Pointer() const { return m_pFn; }
169

173  #else
174  #define GL_FUNC_NAME ""
