231  if ( lhs->m_Hash != rhs->m_Hash )
232  return false;
233  if ( !CASEINSENSITIVE )
234  {
235  return ( !Q_strcmp( lhs->String(), rhs->String() ) ? true : false );
236  }

239  }

242 /*
243  NOTE: So the only crappy thing about using a CUtlTSHash here is that the KEYTYPE is a CUtlSymbolTableLargeBaseTreeEntry_t ptr which has both the
244  hash and the string since with strings there is a good chance of hash collision after you have a fair number of strings so we have to implement
245  a Compare method (above) which falls back to strcmp/stricmp if the hashes are equal. This means that all of the data is in the KEYTYPE of the hash and the
246  payload doesn't matter. So I made the payload also be a pointer to a CUtlSymbolTableLargeBaseTreeEntry_t since that makes using the API more convenient
247 
248  TODO: If we have a CUtlTSHash that was all about the existence of the KEYTYPE and didn't require a payload (or template on 'void') then we could eliminate
249  50% of the pointer overhead used for this data structure.
250 */

415 //-----------------------------------------------------------------------------
416 // Finds and/or creates a symbol based on the string
417 //-----------------------------------------------------------------------------
418 template < class TreeType, bool CASEINSENSITIVE, size_t POOL_SIZE >
419 inline CUtlSymbolLarge CUtlSymbolTableLargeBase<TreeType, CASEINSENSITIVE, POOL_SIZE>::AddString( const char* pString )
420 {
421  VPROF("CUtlSymbolLarge::AddString");
422  if (!pString)
423  return UTL_INVAL_SYMBOL_LARGE;
424 
425  CUtlSymbolLarge id = Find( pString );
426  if ( id != UTL_INVAL_SYMBOL_LARGE )
427  return id;
428 
429  int lenString = Q_strlen(pString) + 1; // length of just the string
430  int lenDecorated = lenString + sizeof(LargeSymbolTableHashDecoration_t); // and with its hash decoration
431  // make sure that all strings are aligned on 2-byte boundaries so the hashes will read correctly
432  // This assert seems to be invalid because LargeSymbolTableHashDecoration_t is always
433  // a uint32, by design.
434  //COMPILE_TIME_ASSERT(sizeof(LargeSymbolTableHashDecoration_t) == sizeof(intp));
435  lenDecorated = ALIGN_VALUE(lenDecorated, sizeof( intp ) );
436 
437  // Find a pool with space for this string, or allocate a new one.
438  int iPool = FindPoolWithSpace( lenDecorated );
439  if ( iPool == -1 )
440  {
441  // Add a new pool.
442  int newPoolSize = MAX( lenDecorated + sizeof( StringPool_t ), POOL_SIZE );
443  StringPool_t *pPool = (StringPool_t*)malloc( newPoolSize );
444 
445  pPool->m_TotalLen = newPoolSize - sizeof( StringPool_t );
446  pPool->m_SpaceUsed = 0;
447  iPool = m_StringPools.AddToTail( pPool );
448  }
449 
450  // Compute a hash
451  LargeSymbolTableHashDecoration_t hash = CUtlSymbolLarge_Hash( CASEINSENSITIVE, pString, lenString );
452 
453  // Copy the string in.
454  StringPool_t *pPool = m_StringPools[iPool];
455  // Assert( pPool->m_SpaceUsed < 0xFFFF ); // Pool could be bigger than 2k
456  // This should never happen, because if we had a string > 64k, it
457  // would have been given its entire own pool.
458 
459  CUtlSymbolTableLargeBaseTreeEntry_t *entry = ( CUtlSymbolTableLargeBaseTreeEntry_t * )&pPool->m_Data[ pPool->m_SpaceUsed ];
460 
461  pPool->m_SpaceUsed += lenDecorated;
462 
463  entry->m_Hash = hash;
464  char *pText = (char *)&entry->m_String [ 0 ];
465  Q_memcpy( pText, pString, lenString );
466 
467  // insert the string into the database
468  MEM_ALLOC_CREDIT();
469  int idx = m_Lookup.Insert( entry );
470  return m_Lookup.Element( idx )->ToSymbol();
471 }
