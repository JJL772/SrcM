184  void Append( const char *pchValue )
185  {
186  // This function is close to the metal to cut down on the CPU cost
187  // of the previous incantation of Append which was implemented as
188  // AppendFormat( "%s", pchValue ). This implementation, though not
189  // as easy to read, instead does a strcpy from the existing end
190  // point of the CFmtStrN. This brings something like a 10-20x speedup
191  // in my rudimentary tests. It isn't using V_strncpy because that
192  // function doesn't return the number of characters copied, which
193  // we need to adjust m_nLength. Doing the V_strncpy with a V_strlen
194  // afterwards took twice as long as this implementations in tests,
195  // so V_strncpy's implementation was used to write this method.
196  char *pDest = m_szBuf + m_nLength;
197  const int maxLen = SIZE_BUF - m_nLength;
198  char *pLast = pDest + maxLen - 1;
199  while ( (pDest < pLast) && (*pchValue != 0) )
200  {
201  *pDest = *pchValue;
202  ++pDest; ++pchValue;
203  }
204  *pDest = 0;
205  m_nLength = pDest - m_szBuf;
206  }
207 
208  //optimized version of append for just adding a single character
209  void Append( char ch )
210  {
211  if( m_nLength < SIZE_BUF - 1 )
212  {
213  m_szBuf[ m_nLength ] = ch;
214  m_nLength++;
215  m_szBuf[ m_nLength ] = '\0';
216  }
217  }
