164 struct vehicle_steeringparams_t
165 {
166  DECLARE_SIMPLE_DATADESC();
167 
168  float degreesSlow; // angle in degrees of steering at slow speed
169  float degreesFast; // angle in degrees of steering at fast speed
170  float degreesBoost; // angle in degrees of steering at fast speed
171  float steeringRateSlow; // this is the speed the wheels are steered when the vehicle is slow
172  float steeringRateFast; // this is the speed the wheels are steered when the vehicle is "fast"
173  float steeringRestRateSlow; // this is the speed at which the wheels move toward their resting state (straight ahead) at slow speed
174  float steeringRestRateFast; // this is the speed at which the wheels move toward their resting state (straight ahead) at fast speed
175  float speedSlow; // this is the max speed of "slow"
176  float speedFast; // this is the min speed of "fast"
177  float turnThrottleReduceSlow; // this is the amount of throttle reduction to apply at the maximum steering angle
178  float turnThrottleReduceFast; // this is the amount of throttle reduction to apply at the maximum steering angle
179  float brakeSteeringRateFactor; // this scales the steering rate when the brake/handbrake is down
180  float throttleSteeringRestRateFactor; // this scales the steering rest rate when the throttle is down
181  float powerSlideAccel; // scale of speed to acceleration
182  float boostSteeringRestRateFactor; // this scales the steering rest rate when boosting
183  float boostSteeringRateFactor; // this scales the steering rest rate when boosting
184  float steeringExponent; // this makes the steering response non-linear. The steering function is linear, then raised to this power
185 
186  bool isSkidAllowed; // true/false skid flag
