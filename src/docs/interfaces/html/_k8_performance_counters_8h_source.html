117  void SetCaptureMode(PrivilegeCapture priv)
118  {
119  PerfEvtSel & select = eventSelect[eventSelectNum];
120  StopCounter();
121 
122  switch (priv)
123  {
124  case OS_Only:
125  select.USR = 0;
126  select.OS = 1;
127  break;
128 
129  case USR_Only:
130  select.USR = 1;
131  select.OS = 0;
132  break;
133 
134  case OS_and_USR:
135  select.USR = 1;
136  select.OS = 1;
137  break;
138  }
139 
140 
141  select.UnitMask = m_eventMask;
142  select.EventMask = event_id;
143 
144 
145  int selectPort = MSR_K8_EVNTSEL0 + eventSelectNum;
146  pme->WriteMSR(selectPort, select.flat);
147  }
148 
149 
150  void SetFiltering(CompareState compareEnable,
151  CompareMethod compareMethod,
152  uint8 threshold,
153  EdgeState edgeEnable)
154  {
155 
156  PerfEvtSel & select = eventSelect[eventSelectNum];
157 
158  StopCounter();
159 
160  if (compareEnable == CompareDisable)
161  select.Threshold = 0;
162  else
163  select.Threshold = threshold;
164 
165  select.Complement = compareMethod;
166 
167  select.Edge = edgeEnable;
168 
169  int selectPort = MSR_K8_EVNTSEL0 + eventSelectNum;
170  pme->WriteMSR(selectPort, select.flat);
171 
172 
173  }
174 
175 
176  void StartCounter()
177  {
178  PerfEvtSel & select = eventSelect[eventSelectNum];
179 
180  select.Enable = 1;

182 
183  pme->WriteMSR(selectPort, select.flat);
184 
185  }

187  {
188  PerfEvtSel & select = eventSelect[eventSelectNum];
189  select.Enable = 0;

191 
192  pme->WriteMSR(selectPort, select.flat);
193  }

197  void ClearCounter()

206  void WriteCounter(int64 value)
207  {
208 
209  PerfEvtSel & select = eventSelect[eventSelectNum];

212  pme->WriteMSR(counterPort, value); // clear

215  int64 ReadCounter()
216  {
217 
218 #if PME_DEBUG
219  PerfEvtSel & select = eventSelect[eventSelectNum];
220 
221  if (select.USR == 0 && select.OS == 0)
222  return -1; // no area to collect, use SetCaptureMode
223 
224  if (select.EventMask == 0)
225  return -2; // no event mask set
226 
227  if (eventSelectNum < 0 || eventSelectNum > 3)
228  return -3; // counter not legal
229 
230  // check revision
231 
232 #endif
233 
234  // ReadMSR should work here too, but RDPMC should be faster
235  //ReadMSR(counterPort, int64);
236 
237  // we need to copy this into a temp for some reason
238 #ifdef COMPILER_MSVC64
239  return __readpmc((unsigned long) eventSelectNum);
240 #else
241  int temp = eventSelectNum;
242  _asm
243  {
244  mov ecx, temp
245  RDPMC
246  }
247 #endif
248 
249  }
