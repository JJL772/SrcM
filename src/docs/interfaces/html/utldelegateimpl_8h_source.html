160 // horrible_cast< >
161 // This is truly evil. It completely subverts C++'s type system, allowing you
162 // to cast from any class to any other class. Technically, using a union
163 // to perform the cast is undefined behaviour (even in C). But we can see if
164 // it is OK by checking that the union is the same size as each of its members.
165 // horrible_cast<> should only be used for compiler-specific workarounds.
166 // Usage is identical to reinterpret_cast<>.
167 
168 // This union is declared outside the horrible_cast because BCC 5.5.1
169 // can't inline a function with a nested class, and gives a warning.
170 template <class OutputClass, class InputClass>
171 union horrible_union
172 {
173  OutputClass out;
174  InputClass in;
175 };
176 
177 template <class OutputClass, class InputClass>
178 inline OutputClass horrible_cast(const InputClass input)
179 {
180  horrible_union<OutputClass, InputClass> u;
181  // Cause a compile-time error if in, out and u are not the same size.
182  // If the compile fails here, it means the compiler has peculiar
183  // unions which would prevent the cast from working.
184  typedef int ERROR_CantUseHorrible_cast[sizeof(InputClass)==sizeof(u)
185  && sizeof(InputClass)==sizeof(OutputClass) ? 1 : -1];
186  u.in = input;
187  return u.out;
188 }

376 // work for all cases, and generated a compile-time error instead.

382 // is internally defined as:
383 struct MicrosoftVirtualMFP
384 {
385  void (GenericClass::*codeptr)(); // points to the actual member function

549 // CUtlAbstractDelegate -- an opaque structure which can hold an arbitary delegate.
550 // It knows nothing about the calling convention or number of arguments used by
551 // the function pointed to.
552 // It supplies comparison operators so that it can be stored in STL collections.
553 // It cannot be set to anything other than null, nor invoked directly:
554 // it must be converted to a specific delegate.
555 
556 // Implementation:
557 // There are two possible implementations: the Safe method and the Evil method.
558 // CUtlAbstractDelegate - Safe version
559 //
560 // This implementation is standard-compliant, but a bit tricky.
561 // A static function pointer is stored inside the class.
562 // Here are the valid values:
563 // +-- Static pointer --+--pThis --+-- pMemFunc-+-- Meaning------+
564 // | 0 | 0 | 0 | Empty |
565 // | !=0 |(dontcare)| Invoker | Static function|
566 // | 0 | !=0 | !=0* | Method call |
567 // +--------------------+----------+------------+----------------+
568 // * For Metrowerks, this can be 0. (first virtual function in a
569 // single_inheritance class).
570 // When stored stored inside a specific delegate, the 'dontcare' entries are replaced
571 // with a reference to the delegate itself. This complicates the = and == operators
572 // for the delegate class.
573 
574 // CUtlAbstractDelegate - Evil version
575 //
576 // For compilers where data pointers are at least as big as code pointers, it is
577 // possible to store the function pointer in the this pointer, using another
578 // horrible_cast. In this case the CUtlAbstractDelegate implementation is simple:
579 // +--pThis --+-- pMemFunc-+-- Meaning---------------------+
580 // | 0 | 0 | Empty |
581 // | !=0 | !=0* | Static function or method call|
582 // +----------+------------+-------------------------------+
583 // * For Metrowerks, this can be 0. (first virtual function in a
584 // single_inheritance class).
585 // Note that the Sun C++ and MSVC documentation explicitly state that they
