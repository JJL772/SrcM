16 // Use this #define to allow SSE versions of Quaternion math
17 // to exist on PC.
18 // On PC, certain horizontal vector operations are not supported.
19 // This causes the SSE implementation of quaternion math to mix the
20 // vector and scalar floating point units, which is extremely
21 // performance negative if you don't compile to native SSE2 (which
22 // we don't as of Sept 1, 2007). So, it's best not to allow these
23 // functions to exist at all. It's not good enough to simply replace
24 // the contents of the functions with scalar math, because each call
25 // to LoadAligned and StoreAligned will result in an unnecssary copy
26 // of the quaternion, and several moves to and from the XMM registers.
27 //
28 // Basically, the problem you run into is that for efficient SIMD code,
29 // you need to load the quaternions and vectors into SIMD registers and
30 // keep them there as long as possible while doing only SIMD math,
31 // whereas for efficient scalar code, each time you copy onto or ever
32 // use a fltx4, it hoses your pipeline. So the difference has to be
33 // in the management of temporary variables in the calling function,
34 // not inside the math functions.
35 //
36 // If you compile assuming the presence of SSE2, the MSVC will abandon
37 // the traditional x87 FPU operations altogether and make everything use
38 // the SSE2 registers, which lessens this problem a little.
39 
40 // permitted only on 360, as we've done careful tuning on its Altivec math:
41 #ifdef _X360
42 #define ALLOW_SIMD_QUATERNION_MATH 1 // not on PC!
43 #endif

95 // Make sure quaternions are within 180 degrees of one another, if not, reverse q
96 //---------------------------------------------------------------------
97 FORCEINLINE fltx4 QuaternionAlignSIMD( const fltx4 &p, const fltx4 &q )
98 {
99  // decide if one of the quaternions is backwards
100  fltx4 a = SubSIMD( p, q );
101  fltx4 b = AddSIMD( p, q );
102  a = Dot4SIMD( a, a );
103  b = Dot4SIMD( b, b );
104  fltx4 cmp = CmpGtSIMD( a, b );
105  fltx4 result = MaskedAssign( cmp, NegSIMD(q), q );
106  return result;
107 }

194 FORCEINLINE fltx4 QuaternionMultSIMD( const fltx4 &p, const fltx4 &q )
195 {

222 //---------------------------------------------------------------------
223 // Quaternion scale
224 //---------------------------------------------------------------------
225 #ifndef _X360
226 
227 // SSE and STDC
228 FORCEINLINE fltx4 QuaternionScaleSIMD( const fltx4 &p, float t )
229 {
230  float r;
231  fltx4 q;
232 
233  // FIXME: nick, this isn't overly sensitive to accuracy, and it may be faster to
234  // use the cos part (w) of the quaternion (sin(omega)*N,cos(omega)) to figure the new scale.
235  float sinom = sqrt( SubFloat( p, 0 ) * SubFloat( p, 0 ) + SubFloat( p, 1 ) * SubFloat( p, 1 ) + SubFloat( p, 2 ) * SubFloat( p, 2 ) );
236  sinom = min( sinom, 1.f );
237 
238  float sinsom = sin( asin( sinom ) * t );
239 
240  t = sinsom / (sinom + FLT_EPSILON);
241  SubFloat( q, 0 ) = t * SubFloat( p, 0 );
242  SubFloat( q, 1 ) = t * SubFloat( p, 1 );
243  SubFloat( q, 2 ) = t * SubFloat( p, 2 );
244 
245  // rescale rotation
246  r = 1.0f - sinsom * sinsom;
247 
248  // Assert( r >= 0 );
249  if (r < 0.0f)
250  r = 0.0f;
251  r = sqrt( r );
252 
253  // keep sign of rotation
254  SubFloat( q, 3 ) = fsel( SubFloat( p, 3 ), r, -r );
255  return q;
256 }

261 FORCEINLINE fltx4 QuaternionScaleSIMD( const fltx4 &p, float t )
262 {
263  fltx4 sinom = Dot3SIMD( p, p );
264  sinom = SqrtSIMD( sinom );
265  sinom = MinSIMD( sinom, Four_Ones );
266  fltx4 sinsom = ArcSinSIMD( sinom );
267  fltx4 t4 = ReplicateX4( t );
268  sinsom = MulSIMD( sinsom, t4 );
269  sinsom = SinSIMD( sinsom );
270  sinom = AddSIMD( sinom, Four_Epsilons );
271  sinom = ReciprocalSIMD( sinom );
272  t4 = MulSIMD( sinsom, sinom );
273  fltx4 result = MulSIMD( p, t4 );
274 
275  // rescale rotation
276  sinsom = MulSIMD( sinsom, sinsom );
277  fltx4 r = SubSIMD( Four_Ones, sinsom );
278  r = MaxSIMD( r, Four_Zeros );
279  r = SqrtSIMD( r );
280 
281  // keep sign of rotation
282  fltx4 cmp = CmpGeSIMD( p, Four_Zeros );
283  r = MaskedAssign( cmp, r, NegSIMD( r ) );
284 
285  result = __vrlimi(result, r, 1, 0);
286  return result;
287 }

297 // SSE and STDC
298 FORCEINLINE fltx4 QuaternionSlerpNoAlignSIMD( const fltx4 &p, const fltx4 &q, float t )
299 {
300  float omega, cosom, sinom, sclp, sclq;
301 
302  fltx4 result;
303 
304  // 0.0 returns p, 1.0 return q.
305  cosom = SubFloat( p, 0 ) * SubFloat( q, 0 ) + SubFloat( p, 1 ) * SubFloat( q, 1 ) +
306  SubFloat( p, 2 ) * SubFloat( q, 2 ) + SubFloat( p, 3 ) * SubFloat( q, 3 );
307 
308  if ( (1.0f + cosom ) > 0.000001f )
309  {
310  if ( (1.0f - cosom ) > 0.000001f )
311  {
312  omega = acos( cosom );
313  sinom = sin( omega );
314  sclp = sin( (1.0f - t)*omega) / sinom;
315  sclq = sin( t*omega ) / sinom;
316  }
317  else
318  {
319  // TODO: add short circuit for cosom == 1.0f?
320  sclp = 1.0f - t;
321  sclq = t;
322  }
323  SubFloat( result, 0 ) = sclp * SubFloat( p, 0 ) + sclq * SubFloat( q, 0 );
324  SubFloat( result, 1 ) = sclp * SubFloat( p, 1 ) + sclq * SubFloat( q, 1 );
325  SubFloat( result, 2 ) = sclp * SubFloat( p, 2 ) + sclq * SubFloat( q, 2 );
326  SubFloat( result, 3 ) = sclp * SubFloat( p, 3 ) + sclq * SubFloat( q, 3 );
327  }
328  else
329  {
330  SubFloat( result, 0 ) = -SubFloat( q, 1 );
331  SubFloat( result, 1 ) = SubFloat( q, 0 );
332  SubFloat( result, 2 ) = -SubFloat( q, 3 );
333  SubFloat( result, 3 ) = SubFloat( q, 2 );
334  sclp = sin( (1.0f - t) * (0.5f * M_PI));
335  sclq = sin( t * (0.5f * M_PI));
336  SubFloat( result, 0 ) = sclp * SubFloat( p, 0 ) + sclq * SubFloat( result, 0 );
337  SubFloat( result, 1 ) = sclp * SubFloat( p, 1 ) + sclq * SubFloat( result, 1 );
338  SubFloat( result, 2 ) = sclp * SubFloat( p, 2 ) + sclq * SubFloat( result, 2 );
339  }
340 
341  return result;
342 }
