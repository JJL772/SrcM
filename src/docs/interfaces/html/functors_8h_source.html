3 // Purpose: Implements a generic infrastucture for functors combining
4 // a number of techniques to provide transparent parameter type
5 // deduction and packaging. Supports both member and non-member functions.
6 //
7 // See also: http://en.wikipedia.org/wiki/Function_object
8 //
9 // Note that getting into the guts of this file is not for the
10 // feint of heart. The core concept here is that the compiler can
11 // figure out all the parameter types.
12 //
13 // E.g.:
14 //
15 // struct CMyClass
16 // {
17 // void foo( int i) {}
18 // };
19 //
20 // int bar(int i) { return i; }
21 //
22 // CMyClass myInstance;
23 //
24 // CFunctor *pFunctor = CreateFunctor( &myInstance, CMyClass::foo, 8675 );
25 // CFunctor *pFunctor2 = CreateFunctor( &bar, 309 );
26 //
27 // void CallEm()
28 // {
29 // (*pFunctor)();
30 // (*pFunctor2)();
31 // }
32 //
33 //=============================================================================
