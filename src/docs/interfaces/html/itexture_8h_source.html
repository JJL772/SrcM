24 // This will get called on procedural textures to re-fill the textures
25 // with the appropriate bit pattern. Calling Download() will also
26 // cause this interface to be called. It will also be called upon
27 // mode switch, or on other occasions where the bits are discarded.
28 //-----------------------------------------------------------------------------
29 abstract_class ITextureRegenerator
30 {
31 public:
32  // This will be called when the texture bits need to be regenerated.
33  // Use the VTFTexture interface, which has been set up with the
34  // appropriate texture size + format
35  // The rect specifies which part of the texture needs to be updated
36  // You can choose to update all of the bits if you prefer
37  virtual void RegenerateTextureBits( ITexture *pTexture, IVTFTexture *pVTFTexture, Rect_t *pRect ) = 0;
38 
39  // This will be called when the regenerator needs to be deleted
40  // which will happen when the texture is destroyed
41  virtual void Release() = 0;
