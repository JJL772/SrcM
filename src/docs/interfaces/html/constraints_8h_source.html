81 // Builds a transform which maps points in the input object's local space
82 // to the output object's local space
83 inline void BuildObjectRelativeXform( IPhysicsObject *pOutputSpace, IPhysicsObject *pInputSpace, matrix3x4_t &xformInToOut )
84 {
85  matrix3x4_t outInv, tmp, input;
86  pOutputSpace->GetPositionMatrix( &tmp );
87  MatrixInvert( tmp, outInv );
88  pInputSpace->GetPositionMatrix( &input );
89  ConcatTransforms( outInv, input, xformInToOut );
90 }

279  float totalLength; // Length of rope/spring/constraint. Distance to maintain

283  {
284  pRef->WorldToLocal( &objectPosition[0], refPosition );

288  }
