379 //-----------------------------------------------------------------------------
380 // Disk format for VTF files ver. 7.2 and earlier
381 //
382 // NOTE: After the header is the low-res image data
383 // Then follows image data, which is sorted in the following manner
384 //
385 // for each mip level (starting with 1x1, and getting larger)
386 // for each animation frame
387 // for each face
388 // store the image data for the face
389 //
390 // NOTE: In memory, we store the data in the following manner:
391 // for each animation frame
392 // for each face
393 // for each mip level (starting with the largest, and getting smaller)
394 // store the image data for the face
395 //
396 // This is done because the various image manipulation function we have
397 // expect this format
398 //-----------------------------------------------------------------------------
399 // Disk format for VTF files ver. 7.3
400 //
401 // NOTE: After the header is the array of ResourceEntryInfo structures,
402 // number of elements in the array is defined by "numResources".
403 // there are entries for:
404 // eRsrcLowResImage = low-res image data
405 // eRsrcSheet = sheet data
406 // eRsrcImage = image data
407 // {
408 // for each mip level (starting with 1x1, and getting larger)
409 // for each animation frame
410 // for each face
411 // store the image data for the face
412 //
413 // NOTE: In memory, we store the data in the following manner:
414 // for each animation frame
415 // for each face
416 // for each mip level (starting with the largest, and getting smaller)
417 // store the image data for the face
418 // }
419 //
420 //-----------------------------------------------------------------------------
