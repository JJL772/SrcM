393 //-----------------------------------------------------------------------------
394 // A simple stream class for printing information to the Maya script editor
395 //-----------------------------------------------------------------------------
396 class CMayaStream
397 {
398 public:
399  enum StreamType { kInfo, kWarning, kError };
400 
401  CMayaStream( const StreamType i_streamType = kInfo )
402  : m_streamType( i_streamType )
403  {}
404 
405  template < class T_t >
406  CMayaStream &operator<<( const T_t &v );
407 
408  // This is a hack so CMayaStream << std::endl works as expected
409  CMayaStream &operator<<( std::ostream &(*StdEndl_t)( std::ostream & ) )
410  {
411  return flush();
412  }
413 
414  CMayaStream &flush() { return outputString(); }
415 
416 protected:
417  CMayaStream &outputString()
418  {
419  // Always ensure it's terminated with a newline
420  if ( *( m_string.asChar() + m_string.length() - 1 ) != '\n' )
421  {
422  m_string += "\n";
423  }
424 
425  const char *pBegin = m_string.asChar();
426  const char *const pEnd = pBegin + m_string.length();
427  const char *pCurr = pBegin;
428 
429  while ( *pCurr && pCurr < pEnd )
430  {
431  if ( *pCurr == '\n' )
432  {
433  switch ( m_streamType )
434  {
435  case kWarning:
436  MGlobal::displayWarning( MString( pBegin, pCurr - pBegin ) );
437  break;
438  case kError:
439  MGlobal::displayError( MString( pBegin, pCurr - pBegin ) );
440  break;
441  default:
442  MGlobal::displayInfo( MString( pBegin, pCurr - pBegin ) );
443  break;
444  }
445 
446  ++pCurr;
447  pBegin = pCurr;
448  }
449  else
450  {
451  ++pCurr;
452  }
453  }
454 
455  m_string.clear();
456 
457  return *this;
458  }
459 
460  CMayaStream &checkForFlush()
461  {
462  const char *pCurr = m_string.asChar();
463  const char *pEnd = pCurr + m_string.length();
464  while ( *pCurr && pCurr != pEnd )
465  {
466  if ( *pCurr == '\n' )
467  {
468  return outputString();
469  }
470 
471  ++pCurr;
472  }
473 
474  return *this;
475  }
476 
477  StreamType m_streamType;
478  MString m_string;
479 };
