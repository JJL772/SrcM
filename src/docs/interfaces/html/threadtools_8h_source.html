434 //
435 // A super-fast thread-safe integer A simple class encapsulating the notion of an
436 // atomic integer used across threads that uses the built in and faster
437 // "interlocked" functionality rather than a full-blown mutex. Useful for simple
438 // things like reference counts, etc.
439 //

442 template <typename T>
443 class CInterlockedIntT
444 {
445 public:
446  CInterlockedIntT() : m_value( 0 ) { COMPILE_TIME_ASSERT( sizeof(T) == sizeof(long) ); }
447  CInterlockedIntT( T value ) : m_value( value ) {}
448 
449  T GetRaw() const { return m_value; }
450 
451  operator T() const { return m_value; }
452 
453  bool operator!() const { return ( m_value == 0 ); }
454  bool operator==( T rhs ) const { return ( m_value == rhs ); }
455  bool operator!=( T rhs ) const { return ( m_value != rhs ); }
456 
457  T operator++() { return (T)ThreadInterlockedIncrement( (long *)&m_value ); }
458  T operator++(int) { return operator++() - 1; }
459 
460  T operator--() { return (T)ThreadInterlockedDecrement( (long *)&m_value ); }
461  T operator--(int) { return operator--() + 1; }
462 
463  bool AssignIf( T conditionValue, T newValue ) { return ThreadInterlockedAssignIf( (long *)&m_value, (long)newValue, (long)conditionValue ); }
464 
465  T operator=( T newValue ) { ThreadInterlockedExchange((long *)&m_value, newValue); return m_value; }
466 
467  void operator+=( T add ) { ThreadInterlockedExchangeAdd( (long *)&m_value, (long)add ); }
468  void operator-=( T subtract ) { operator+=( -subtract ); }
469  void operator*=( T multiplier ) {
470  T original, result;
471  do
472  {
473  original = m_value;
474  result = original * multiplier;
475  } while ( !AssignIf( original, result ) );
476  }
477  void operator/=( T divisor ) {
478  T original, result;
479  do
480  {
481  original = m_value;
482  result = original / divisor;
483  } while ( !AssignIf( original, result ) );
484  }
485 
486  T operator+( T rhs ) const { return m_value + rhs; }
487  T operator-( T rhs ) const { return m_value - rhs; }
488 
489 private:
490  volatile T m_value;
491 };
