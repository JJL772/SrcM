245 abstract_class IShaderShadow
246 {
247 public:
248  // Sets the default *shadow* state
249  virtual void SetDefaultState() = 0;
250 
251  // Methods related to depth buffering
252  virtual void DepthFunc( ShaderDepthFunc_t depthFunc ) = 0;
253  virtual void EnableDepthWrites( bool bEnable ) = 0;
254  virtual void EnableDepthTest( bool bEnable ) = 0;
255  virtual void EnablePolyOffset( PolygonOffsetMode_t nOffsetMode ) = 0;
256 
257  // These methods for controlling stencil are obsolete and stubbed to do nothing. Stencil
258  // control is via the shaderapi/material system now, not part of the shadow state.
259  // Methods related to stencil
260  virtual void EnableStencil( bool bEnable ) = 0;
261  virtual void StencilFunc( ShaderStencilFunc_t stencilFunc ) = 0;
262  virtual void StencilPassOp( ShaderStencilOp_t stencilOp ) = 0;
263  virtual void StencilFailOp( ShaderStencilOp_t stencilOp ) = 0;
264  virtual void StencilDepthFailOp( ShaderStencilOp_t stencilOp ) = 0;
265  virtual void StencilReference( int nReference ) = 0;
266  virtual void StencilMask( int nMask ) = 0;
267  virtual void StencilWriteMask( int nMask ) = 0;
268 
269  // Suppresses/activates color writing
270  virtual void EnableColorWrites( bool bEnable ) = 0;
271  virtual void EnableAlphaWrites( bool bEnable ) = 0;
272 
273  // Methods related to alpha blending
274  virtual void EnableBlending( bool bEnable ) = 0;
275  virtual void BlendFunc( ShaderBlendFactor_t srcFactor, ShaderBlendFactor_t dstFactor ) = 0;
276  // More below...
277 
278  // Alpha testing
279  virtual void EnableAlphaTest( bool bEnable ) = 0;
280  virtual void AlphaFunc( ShaderAlphaFunc_t alphaFunc, float alphaRef /* [0-1] */ ) = 0;
281 
282  // Wireframe/filled polygons
283  virtual void PolyMode( ShaderPolyModeFace_t face, ShaderPolyMode_t polyMode ) = 0;
284 
285  // Back face culling
286  virtual void EnableCulling( bool bEnable ) = 0;
287 
288  // constant color + transparency
289  virtual void EnableConstantColor( bool bEnable ) = 0;
290 
291  // Indicates the vertex format for use with a vertex shader
292  // The flags to pass in here come from the VertexFormatFlags_t enum
293  // If pTexCoordDimensions is *not* specified, we assume all coordinates
294  // are 2-dimensional
295  virtual void VertexShaderVertexFormat( unsigned int nFlags,
296  int nTexCoordCount, int* pTexCoordDimensions, int nUserDataSize ) = 0;
297 
298  // Pixel and vertex shader methods
299  virtual void SetVertexShader( const char* pFileName, int nStaticVshIndex ) = 0;
300  virtual void SetPixelShader( const char* pFileName, int nStaticPshIndex = 0 ) = 0;
301 
302  // Indicates we're going to light the model
303  virtual void EnableLighting( bool bEnable ) = 0;
304 
305  // Enables specular lighting (lighting has also got to be enabled)
306  virtual void EnableSpecular( bool bEnable ) = 0;
307 
308  // Convert from linear to gamma color space on writes to frame buffer.
309  virtual void EnableSRGBWrite( bool bEnable ) = 0;
310 
311  // Convert from gamma to linear on texture fetch.
312  virtual void EnableSRGBRead( Sampler_t sampler, bool bEnable ) = 0;
313 
314  // Activate/deactivate skinning. Indexed blending is automatically
315  // enabled if it's available for this hardware. When blending is enabled,
316  // we allocate enough room for 3 weights (max allowed)
317  virtual void EnableVertexBlend( bool bEnable ) = 0;
318 
319  // per texture unit stuff
320  virtual void OverbrightValue( TextureStage_t stage, float value ) = 0;
321  virtual void EnableTexture( Sampler_t sampler, bool bEnable ) = 0;
322  virtual void EnableTexGen( TextureStage_t stage, bool bEnable ) = 0;
323  virtual void TexGen( TextureStage_t stage, ShaderTexGenParam_t param ) = 0;
324 
325  // alternate method of specifying per-texture unit stuff, more flexible and more complicated
326  // Can be used to specify different operation per channel (alpha/color)...
327  virtual void EnableCustomPixelPipe( bool bEnable ) = 0;
328  virtual void CustomTextureStages( int stageCount ) = 0;
329  virtual void CustomTextureOperation( TextureStage_t stage, ShaderTexChannel_t channel,
330  ShaderTexOp_t op, ShaderTexArg_t arg1, ShaderTexArg_t arg2 ) = 0;
331 
332  // indicates what per-vertex data we're providing
333  virtual void DrawFlags( unsigned int drawFlags ) = 0;
334 
335  // A simpler method of dealing with alpha modulation
336  virtual void EnableAlphaPipe( bool bEnable ) = 0;
337  virtual void EnableConstantAlpha( bool bEnable ) = 0;
338  virtual void EnableVertexAlpha( bool bEnable ) = 0;
339  virtual void EnableTextureAlpha( TextureStage_t stage, bool bEnable ) = 0;
340 
341  // GR - Separate alpha blending
342  virtual void EnableBlendingSeparateAlpha( bool bEnable ) = 0;
343  virtual void BlendFuncSeparateAlpha( ShaderBlendFactor_t srcFactor, ShaderBlendFactor_t dstFactor ) = 0;
344  virtual void FogMode( ShaderFogMode_t fogMode ) = 0;
345 
346  virtual void SetDiffuseMaterialSource( ShaderMaterialSource_t materialSource ) = 0;
347 
348  // Indicates the morph format for use with a vertex shader
349  // The flags to pass in here come from the MorphFormatFlags_t enum
350  virtual void SetMorphFormat( MorphFormat_t flags ) = 0;
351 
352  virtual void DisableFogGammaCorrection( bool bDisable ) = 0; //some blending modes won't work properly with corrected fog
353 
354  // Alpha to coverage
355  virtual void EnableAlphaToCoverage( bool bEnable ) = 0;
356 
357  // Shadow map filtering
358  virtual void SetShadowDepthFiltering( Sampler_t stage ) = 0;
359 
360  // More alpha blending state
361  virtual void BlendOp( ShaderBlendOp_t blendOp ) = 0;
362  virtual void BlendOpSeparateAlpha( ShaderBlendOp_t blendOp ) = 0;
363 };
