258 T* CUtlCachedFileData<T>::Get( char const *filename )
259 {
260  int idx = GetIndex( filename );
261 
262  ElementType_t& e = m_Elements[ idx ];
263 
264  if ( e.fileinfo == -1 &&
265  m_fileCheckType == UTL_CACHED_FILE_USE_FILESIZE )
266  {
267  e.fileinfo = 0;
268  }
269  long cachefileinfo = e.fileinfo;
270  // Set the disk fileinfo the first time we encounter the filename
271  if ( e.diskfileinfo == UTL_CACHED_FILE_DATA_UNDEFINED_DISKINFO )
272  {
273  if ( m_bNeverCheckDisk )
274  {
275  e.diskfileinfo = cachefileinfo;
276  }
277  else
278  {
279  if ( m_fileCheckType == UTL_CACHED_FILE_USE_FILESIZE )
280  {
281  e.diskfileinfo = g_pFullFileSystem->Size( filename, "GAME" );
282  // Missing files get a disk file size of 0
283  if ( e.diskfileinfo == -1 )
284  {
285  e.diskfileinfo = 0;
286  }
287  }
288  else
289  {
290  e.diskfileinfo = g_pFullFileSystem->GetFileTime( filename, "GAME" );
291  }
292  }
293  }
294 
295  Assert( e.dataIndex != m_Data.InvalidIndex() );
296 
297  T *data = m_Data[ e.dataIndex ];
298 
299  Assert( data );
300 
301  // Compare fileinfo to disk fileinfo and rebuild cache if out of date or not correct...
302  if ( cachefileinfo != e.diskfileinfo )
303  {
304  if ( !m_bReadOnly )
305  {
306  RebuildCache( filename, data );
307  }
308  e.fileinfo = e.diskfileinfo;
309  }
310 
311  return data;

345 template <class T>
346 bool CUtlCachedFileData<T>::IsUpToDate()
347 {
348  // Don't call Init/Shutdown if using this method!!!
349  Assert( !m_bInitialized );
350 
351  if ( m_sRepositoryFileName.IsEmpty() )
352  {
353  Error( "CUtlCachedFileData: Can't IsUpToDate, no repository file specified." );
354  return false;
355  }
356 
357  // Always compute meta checksum
358  m_uCurrentMetaChecksum = m_pfnMetaChecksum ? (*m_pfnMetaChecksum)() : 0;
359 
360  FileHandle_t fh;
361 
362  fh = g_pFullFileSystem->Open( m_sRepositoryFileName, "rb", "MOD" );
363  if ( fh == FILESYSTEM_INVALID_HANDLE )
364  {
365  return false;
366  }
367 
368  // Version data is in first 12 bytes of file
369  byte header[ 12 ];
370  g_pFullFileSystem->Read( header, sizeof( header ), fh );
371  g_pFullFileSystem->Close( fh );
372 
373  int cacheversion = *( int *)&header[ 0 ];
374 
375  if ( UTL_CACHE_SYSTEM_VERSION != cacheversion )
376  {
377  DevMsg( "Discarding repository '%s' due to cache system version change\n", m_sRepositoryFileName.String() );
378  Assert( !m_bReadOnly );
379  if ( !m_bReadOnly )
380  {
381  g_pFullFileSystem->RemoveFile( m_sRepositoryFileName, "MOD" );
382  }
383  return false;
384  }
385 
386  // Now parse data from the buffer
387  int version = *( int *)&header[ 4 ];
388  if ( version != m_nVersion )
389  {
390  DevMsg( "Discarding repository '%s' due to version change\n", m_sRepositoryFileName.String() );
391  Assert( !m_bReadOnly );
392  if ( !m_bReadOnly )
393  {
394  g_pFullFileSystem->RemoveFile( m_sRepositoryFileName, "MOD" );
395  }
396  return false;
397  }
398 
399  // This is a checksum based on any meta data files which the cache depends on (supplied by a passed in
400  // meta data function
401  unsigned int cache_meta_checksum = (unsigned int)*( int *)&header[ 8 ];
402 
403  if ( cache_meta_checksum != m_uCurrentMetaChecksum )
404  {
405  DevMsg( "Discarding repository '%s' due to meta checksum change\n", m_sRepositoryFileName.String() );
406  Assert( !m_bReadOnly );
407  if ( !m_bReadOnly )
408  {
409  g_pFullFileSystem->RemoveFile( m_sRepositoryFileName, "MOD" );
410  }
411  return false;
412  }
413 
414  // Looks valid
415  return true;
416 }

419 void CUtlCachedFileData<T>::InitSmallBuffer( FileHandle_t& fh, int fileSize, bool& deleteFile )
420 {
421  deleteFile = false;
422 
423  CUtlBuffer loadBuf;
424  g_pFullFileSystem->ReadToBuffer( fh, loadBuf );
425  g_pFullFileSystem->Close( fh );
426 
427  int cacheversion = 0;
428  loadBuf.Get( &cacheversion, sizeof( cacheversion ) );
429 
430  if ( UTL_CACHE_SYSTEM_VERSION == cacheversion )
431  {
432  // Now parse data from the buffer
433  int version = loadBuf.GetInt();
434 
435  if ( version == m_nVersion )
436  {
437  // This is a checksum based on any meta data files which the cache depends on (supplied by a passed in
438  // meta data function
439  unsigned int cache_meta_checksum = loadBuf.GetInt();
440 
441  if ( cache_meta_checksum == m_uCurrentMetaChecksum )
442  {
443  int count = loadBuf.GetInt();
444 
445  Assert( count < 2000000 );
446 
447  CUtlBuffer buf( 0, 0, 0 );
448 
449  for ( int i = 0 ; i < count; ++i )
450  {
451  int bufsize = loadBuf.GetInt();
452  Assert( bufsize < 1000000 );
453 
454  buf.Clear();
455  buf.EnsureCapacity( bufsize );
456 
457  loadBuf.Get( buf.Base(), bufsize );
458 
459  buf.SeekGet( CUtlBuffer::SEEK_HEAD, 0 );
460  buf.SeekPut( CUtlBuffer::SEEK_HEAD, bufsize );
461 
462  // Read the element name
463  char elementFileName[ 512 ];
464  buf.GetString( elementFileName );
465 
466  // Now read the element
467  int slot = GetIndex( elementFileName );
468 
469  Assert( slot != m_Elements.InvalidIndex() );
470 
471  ElementType_t& element = m_Elements[ slot ];
472 
473  element.fileinfo = buf.GetInt();
474  if ( ( element.fileinfo == -1 ) &&
475  ( m_fileCheckType == UTL_CACHED_FILE_USE_FILESIZE ) )
476  {
477  element.fileinfo = 0;
478  }
479 
480  Assert( element.dataIndex != m_Data.InvalidIndex() );
481 
482  T *data = m_Data[ element.dataIndex ];
483 
484  Assert( data );
485 
486  ((IBaseCacheInfo *)data)->Restore( buf );
487  }
488  }
489  else
490  {
491  Msg( "Discarding repository '%s' due to meta checksum change\n", m_sRepositoryFileName.String() );
492  deleteFile = true;
493  }
494  }
495  else
496  {
497  Msg( "Discarding repository '%s' due to version change\n", m_sRepositoryFileName.String() );
498  deleteFile = true;
499  }
500  }
501  else
502  {
503  DevMsg( "Discarding repository '%s' due to cache system version change\n", m_sRepositoryFileName.String() );
504  deleteFile = true;
505  }

665 template <class T>
666 void CUtlCachedFileData<T>::Save()
667 {
668  char path[ 512 ];
669  Q_strncpy( path, m_sRepositoryFileName, sizeof( path ) );
670  Q_StripFilename( path );
671 
672  g_pFullFileSystem->CreateDirHierarchy( path, "MOD" );
673 
674  if ( g_pFullFileSystem->FileExists( m_sRepositoryFileName, "MOD" ) &&
675  !g_pFullFileSystem->IsFileWritable( m_sRepositoryFileName, "MOD" ) )
676  {
677  g_pFullFileSystem->SetFileWritable( m_sRepositoryFileName, true, "MOD" );
678  }
679 
680  // Now write to file
681  FileHandle_t fh;
682  fh = g_pFullFileSystem->Open( m_sRepositoryFileName, "wb" );
683  if ( FILESYSTEM_INVALID_HANDLE == fh )
684  {
685  ExecuteNTimes( 25, Warning( "Unable to persist cache '%s', check file permissions\n", m_sRepositoryFileName.String() ) );
686  }
687  else
688  {
689  SetDirty( false );
690 
691  int v = UTL_CACHE_SYSTEM_VERSION;
692  g_pFullFileSystem->Write( &v, sizeof( v ), fh );
693  v = m_nVersion;
694  g_pFullFileSystem->Write( &v, sizeof( v ), fh );
695  v = (int)m_uCurrentMetaChecksum;
696  g_pFullFileSystem->Write( &v, sizeof( v ), fh );
697 
698  // Element count
699  int c = Count();
700 
701  g_pFullFileSystem->Write( &c, sizeof( c ), fh );
702 
703  // Save repository back out to disk...
704  CUtlBuffer buf( 0, 0, 0 );
705 
706  for ( int i = m_Elements.FirstInorder(); i != m_Elements.InvalidIndex(); i = m_Elements.NextInorder( i ) )
707  {
708  buf.SeekPut( CUtlBuffer::SEEK_HEAD, 0 );
709 
710  ElementType_t& element = m_Elements[ i ];
711 
712  char fn[ 512 ];
713  g_pFullFileSystem->String( element.handle, fn, sizeof( fn ) );
714 
715  buf.PutString( fn );
716  buf.PutInt( element.fileinfo );
717 
718  Assert( element.dataIndex != m_Data.InvalidIndex() );
719 
720  T *data = m_Data[ element.dataIndex ];
721 
722  Assert( data );
723 
724  ((IBaseCacheInfo *)data)->Save( buf );
725 
726  int bufsize = buf.TellPut();
727  g_pFullFileSystem->Write( &bufsize, sizeof( bufsize ), fh );
728  g_pFullFileSystem->Write( buf.Base(), bufsize, fh );
729  }
730 
731  g_pFullFileSystem->Close( fh );
732  }
733 
734  if ( m_bSaveManifest )
735  {
736  SaveManifest();
737  }
738 }

955  {
956  bCheck = false;

963 
964  for ( i = listStart; i < listEnd; i++ )
965  {
966  ElementType_t& element = m_Elements[ list[i].index ];
967

973  }
974  else
