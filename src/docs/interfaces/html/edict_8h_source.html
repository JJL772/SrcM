286 inline void CBaseEdict::StateChanged( unsigned short offset )
287 {
288  if ( m_fStateFlags & FL_FULL_EDICT_CHANGED )
289  return;
290 
291  m_fStateFlags |= FL_EDICT_CHANGED;
292 
293  IChangeInfoAccessor *accessor = GetChangeAccessor();
294 
295  if ( accessor->GetChangeInfoSerialNumber() == g_pSharedChangeInfo->m_iSerialNumber )
296  {
297  // Ok, I still own this one.
298  CEdictChangeInfo *p = &g_pSharedChangeInfo->m_ChangeInfos[accessor->GetChangeInfo()];
299 
300  // Now add this offset to our list of changed variables.
301  for ( unsigned short i=0; i < p->m_nChangeOffsets; i++ )
302  if ( p->m_ChangeOffsets[i] == offset )
303  return;
304 
305  if ( p->m_nChangeOffsets == MAX_CHANGE_OFFSETS )
306  {
307  // Invalidate our change info.
308  accessor->SetChangeInfoSerialNumber( 0 );
309  m_fStateFlags |= FL_FULL_EDICT_CHANGED; // So we don't get in here again.
310  }
311  else
312  {
313  p->m_ChangeOffsets[p->m_nChangeOffsets++] = offset;
314  }
315  }
316  else
317  {
318  if ( g_pSharedChangeInfo->m_nChangeInfos == MAX_EDICT_CHANGE_INFOS )
319  {
320  // Shucks.. have to mark the edict as fully changed because we don't have room to remember this change.
321  accessor->SetChangeInfoSerialNumber( 0 );
322  m_fStateFlags |= FL_FULL_EDICT_CHANGED;
323  }
324  else
325  {
326  // Get a new CEdictChangeInfo and fill it out.
327  accessor->SetChangeInfo( g_pSharedChangeInfo->m_nChangeInfos );
328  g_pSharedChangeInfo->m_nChangeInfos++;
329 
330  accessor->SetChangeInfoSerialNumber( g_pSharedChangeInfo->m_iSerialNumber );
331 
332  CEdictChangeInfo *p = &g_pSharedChangeInfo->m_ChangeInfos[accessor->GetChangeInfo()];
333  p->m_ChangeOffsets[0] = offset;
334  p->m_nChangeOffsets = 1;
335  }
336  }
337 }
