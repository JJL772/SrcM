47 struct TOGL_CLASS IUnknown
48 {

52  IUnknown()
53  {
54  m_refcount[0] = 1;

56  m_mark = (IUNKNOWN_ALLOC_SPEW_MARK_ALL != 0); // either all are marked, or only the ones that have SetMark(true) called on them
57 
58  #if IUNKNOWN_ALLOC_SPEW
59  if (m_mark)
60  {
61  GLMPRINTF(("-A- IUnew (%08x) refc -> (%d,%d) ",this,m_refcount[0],m_refcount[1]));
62  }
63  #endif
64  };
65 
66  virtual ~IUnknown()
67  {
68  #if IUNKNOWN_ALLOC_SPEW
69  if (m_mark)
70  {
71  GLMPRINTF(("-A- IUdel (%08x) ",this ));
72  }
73  #endif
74  };

76  void AddRef( int which=0, char *comment = NULL )
77  {
78  Assert( which >= 0 );
79  Assert( which < 2 );
80  m_refcount[which]++;
81 
82  #if IUNKNOWN_ALLOC_SPEW
83  if (m_mark)
84  {
85  GLMPRINTF(("-A- IUAddRef (%08x,%d) refc -> (%d,%d) [%s]",this,which,m_refcount[0],m_refcount[1],comment?comment:"...")) ;
86  if (!comment)
87  {
88  GLMPRINTF(("")) ; // place to hang a breakpoint
89  }
90  }
91  #endif
92  };
93 
94  ULONG __stdcall Release( int which=0, char *comment = NULL )
95  {
96  Assert( which >= 0 );
97  Assert( which < 2 );
98 
99  //int oldrefcs[2] = { m_refcount[0], m_refcount[1] };
100  bool deleting = false;
101 
102  m_refcount[which]--;
103  if ( (!m_refcount[0]) && (!m_refcount[1]) )
104  {
105  deleting = true;
106  }
107 
108  #if IUNKNOWN_ALLOC_SPEW
109  if (m_mark)
110  {
111  GLMPRINTF(("-A- IURelease (%08x,%d) refc -> (%d,%d) [%s] %s",this,which,m_refcount[0],m_refcount[1],comment?comment:"...",deleting?"->DELETING":""));
112  if (!comment)
113  {
114  GLMPRINTF(("")) ; // place to hang a breakpoint
115  }
116  }
117  #endif
118 
119  if (deleting)
120  {
121  if (m_mark)
122  {
123  GLMPRINTF(("")) ; // place to hang a breakpoint
124  }
125  delete this;
126  return 0;
127  }
128  else
129  {
130  return m_refcount[0];
131  }
132  };
133 
134  void SetMark( bool markValue, char *comment=NULL )
135  {
136  #if IUNKNOWN_ALLOC_SPEW
137  if (!m_mark && markValue) // leading edge detect
138  {
139  // print the same thing that the constructor would have printed if it had been marked from the beginning
140  // i.e. it's anticipated that callers asking for marking will do so right at create time
141  GLMPRINTF(("-A- IUSetMark (%08x) refc -> (%d,%d) (%s) ",this,m_refcount[0],m_refcount[1],comment?comment:"..."));
142  }
143  #endif
144 
145  m_mark = markValue;
146  }

355 struct RenderTargetState_t
356 {
357  void clear() { V_memset( this, 0, sizeof( *this ) ); }
358 
359  CGLMTex *m_pRenderTargets[4];
360  CGLMTex *m_pDepthStencil;
361 
362  inline bool RefersTo( CGLMTex * pSurf ) const
363  {
364  for ( uint i = 0; i < 4; i++ )
365  if ( m_pRenderTargets[i] == pSurf )
366  return true;
367 
368  if ( m_pDepthStencil == pSurf )
369  return true;
370 
371  return false;
372  }
373 
374  static inline bool LessFunc( const RenderTargetState_t &lhs, const RenderTargetState_t &rhs )
375  {
376  COMPILE_TIME_ASSERT( sizeof( lhs.m_pRenderTargets[0] ) == sizeof( uint32 ) );
377  uint64 lhs0 = reinterpret_cast<const uint64 *>(lhs.m_pRenderTargets)[0];
378  uint64 rhs0 = reinterpret_cast<const uint64 *>(rhs.m_pRenderTargets)[0];
379  if ( lhs0 < rhs0 )
380  return true;
381  else if ( lhs0 == rhs0 )
382  {
383  uint64 lhs1 = reinterpret_cast<const uint64 *>(lhs.m_pRenderTargets)[1];
384  uint64 rhs1 = reinterpret_cast<const uint64 *>(rhs.m_pRenderTargets)[1];
385  if ( lhs1 < rhs1 )
386  return true;
387  else if ( lhs1 == rhs1 )
388  {
389  return lhs.m_pDepthStencil < rhs.m_pDepthStencil;
390  }
391  }
392  return false;
393  }
394 
395  inline bool operator < ( const RenderTargetState_t &rhs ) const
396  {
397  return LessFunc( *this, rhs );
398  }
399 };
