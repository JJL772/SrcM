89 // MODEL INDEX RULES
90 // If index >= 0, then index references the precached model string table
91 // If index == -1, then the model is invalid
92 // If index < -1, then the model is DYNAMIC and has a DYNAMIC INDEX of (-2 - index)
93 // - if the dynamic index is ODD, then the model is CLIENT ONLY
94 // and has a m_LocalDynamicModels lookup index of (dynamic index)>>1
95 // - if the dynamic index is EVEN, then the model is NETWORKED
96 // and has a dynamic model string table index of (dynamic index)>>1
97 
98 inline bool IsDynamicModelIndex( int modelindex ) { return modelindex < -1; }
99 inline bool IsClientOnlyModelIndex( int modelindex ) { return modelindex < -1 && (modelindex & 1); }
