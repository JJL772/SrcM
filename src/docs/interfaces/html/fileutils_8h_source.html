68 // simple file classes. File I/O mode (text/binary, read/write) is based upon the subclass chosen.
69 // classes with the word Required on them abort with a message if the file can't be opened.
70 // destructores close the file handle, or it can be explicitly closed with the Close() method.

72 class CBaseFile
73 {
74 public:

76 
77  CBaseFile(void)

92  void Close( void )
93  {
94  if ( m_FileHandle != FILESYSTEM_INVALID_HANDLE )

96  m_FileHandle = FILESYSTEM_INVALID_HANDLE;
97  }
98 
99  void Open( char const *fname, char const *modes )
100  {
101  Close();

103  }

117 
118  void MustRead( void* pOutput, int size )
119  {
120  int ret=Read( pOutput, size );
121  if (ret != size )
122  Error("failed to read %d bytes\n", size );
123  }
124 
125  int Write( void const* pInput, int size)
126  {
127  return g_pFullFileSystem->Write( pInput, size, m_FileHandle );
128  }

131  // {Get|Put}{Int|Float} read and write ints and floats from a file in x86 order, swapping on
132  // input for big-endian systems.
133  void PutInt( int n )
134  {
135  int n1=LittleDWord( n );
136  Write(&n1, sizeof( n1 ) );
137  }
138 
139  int GetInt( void )
140  {
141  int ret;
142  MustRead( &ret, sizeof( ret ));
143  return LittleDWord( ret );
144  }
145 
146  float GetFloat( void )
147  {
148  float ret;
149  MustRead( &ret, sizeof( ret ));
150  LittleFloat( &ret, &ret );
151  return ret;
152  }
153  void PutFloat( float f )
154  {
155  LittleFloat( &f, &f );
156  Write( &f, sizeof( f ) );
157  }
158 
159  bool IsOk( void )
160  {
161  return ( m_FileHandle != FILESYSTEM_INVALID_HANDLE) &&
162  ( g_pFullFileSystem->IsOk( m_FileHandle ) );
163  }
164 
165  void Seek( int pos, FileSystemSeek_t nSeekType = FILESYSTEM_SEEK_HEAD )
166  {
167  g_pFullFileSystem->Seek( m_FileHandle, pos, nSeekType );
168  }
169 
170  unsigned int Tell()
171  {
172  return g_pFullFileSystem->Tell( m_FileHandle );
173  }
174 
175  unsigned int Size( void )
176  {
177  Assert( IsOk() );
178  return g_pFullFileSystem->Size( m_FileHandle );
179  }
180 
181  void ReadFile( CUtlBuffer &dataBuf );
182 };
