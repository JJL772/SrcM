3 // Purpose: a fast growable hashtable with stored hashes, L2-friendly behavior.
4 // Useful as a string dictionary or a low-overhead set/map for small POD types.
5 //
6 // Usage notes:
7 // - handles are NOT STABLE across element removal! use RemoveAndAdvance()
8 // if you are removing elements while iterating through the hashtable.
9 // Use CUtlStableHashtable if you need stable handles (less efficient).
10 // - Insert() first searches for an existing match and returns it if found
11 // - a value type of "empty_t" can be used to eliminate value storage and
12 // switch Element() to return const Key references instead of values
13 // - an extra user flag bit is accessible via Get/SetUserFlag()
14 // - hash function pointer / functor is exposed via GetHashRef()
15 // - comparison function pointer / functor is exposed via GetEqualRef()
16 // - if your value type cannot be copy-constructed, use key-only Insert()
17 // to default-initialize the value and then manipulate it afterwards.
18 //
19 // Implementation notes:
20 // - overall hash table load is kept between .25 and .75
21 // - items which would map to the same ideal slot are chained together
22 // - chained items are stored sequentially in adjacent free spaces
23 // - "root" entries are prioritized over chained entries; if a
24 // slot is not occupied by an item in its root position, the table
25 // is guaranteed to contain no keys which would hash to that slot.
26 // - new items go at the head of the chain (ie, in their root slot)
27 // and evict / "bump" any chained entries which occupy that slot
28 // - chain-following skips over unused holes and continues examining
29 // table entries until a chain entry with FLAG_LAST is encountered
30 //
31 // CUtlHashtable< uint32 > setOfIntegers;
32 // CUtlHashtable< const char* > setOfStringPointers;
33 // CUtlHashtable< int, CUtlVector<blah_t> > mapFromIntsToArrays;
34 //
35 // $NoKeywords: $
36 //
37 // A closed-form (open addressing) hashtable with linear sequential probing.
38 //=============================================================================//

330 // Move an existing entry to a free slot, leaving a hole behind
331 template <typename KeyT, typename ValueT, typename KeyHashT, typename KeyIsEqualT, typename AltKeyT>
332 void CUtlHashtable<KeyT, ValueT, KeyHashT, KeyIsEqualT, AltKeyT>::BumpEntry( unsigned int idx )
333 {
334  Assert( m_table[idx].IsValid() );
335  Assert( m_nUsed < m_table.Count() );
336 
337  entry_t* table = m_table.Base();
338  unsigned int slotmask = m_table.Count()-1;
339  unsigned int new_flags_and_hash = table[idx].flags_and_hash & (FLAG_LAST | MASK_HASH);
340 
341  unsigned int chainid = entry_t::IdealIndex( new_flags_and_hash, slotmask );
342 
343  // Look for empty slots scanning forward, stripping FLAG_LAST as we go.
344  // Note: this potentially strips FLAG_LAST from table[idx] if we pass it
345  int newIdx = chainid; // start at ideal slot
346  for ( ; ; newIdx = (newIdx + 1) & slotmask )
347  {
348  if ( table[newIdx].IdealIndex( slotmask ) == chainid )
349  {
350  if ( table[newIdx].flags_and_hash & FLAG_LAST )
351  {
352  table[newIdx].flags_and_hash &= ~FLAG_LAST;
353  new_flags_and_hash |= FLAG_LAST;
354  }
355  continue;
356  }
357  if ( table[newIdx].IsValid() )
358  {
359  continue;
360  }
361  break;
362  }
363 
364  // Did we pick something closer to the ideal slot, leaving behind a
365  // FLAG_LAST bit on the current slot because we didn't scan past it?
366  if ( table[idx].flags_and_hash & FLAG_LAST )
367  {
368 #ifdef _DEBUG
369  Assert( new_flags_and_hash & FLAG_LAST );
370  // Verify logic: we must have moved to an earlier slot, right?
371  uint offset = ((uint)idx - chainid + slotmask + 1) & slotmask;
372  uint newOffset = ((uint)newIdx - chainid + slotmask + 1) & slotmask;
373  Assert( newOffset < offset );
374 #endif
375  // Scan backwards from old to new location, depositing FLAG_LAST on
376  // the first match we find. (+slotmask) is the same as (-1) without
377  // having to make anyone think about two's complement shenanigans.
378  int scan = (idx + slotmask) & slotmask;
379  while ( scan != newIdx )
380  {
381  if ( table[scan].IdealIndex( slotmask ) == chainid )
382  {
383  table[scan].flags_and_hash |= FLAG_LAST;
384  new_flags_and_hash &= ~FLAG_LAST;
385  break;
386  }
387  scan = (scan + slotmask) & slotmask;
388  }
389  }
390 
391  // Move entry to the free slot we found, leaving a hole at idx
392  table[newIdx].flags_and_hash = new_flags_and_hash;
393  table[newIdx].MoveDataFrom( table[idx] );
394  table[idx].MarkInvalid();
395 }

398 // Insert a value at the root position for that value's hash chain.
399 template <typename KeyT, typename ValueT, typename KeyHashT, typename KeyIsEqualT, typename AltKeyT>
400 int CUtlHashtable<KeyT, ValueT, KeyHashT, KeyIsEqualT, AltKeyT>::DoInsertUnconstructed( unsigned int h, bool allowGrow )
401 {
402  if ( allowGrow && !m_bSizeLocked )
403  {
404  // Keep the load factor between .25 and .75
405  int newSize = m_nUsed + 1;
406  if ( ( newSize*4 < m_table.Count() && m_table.Count() > m_nMinSize*2 ) || newSize*4 > m_table.Count()*3 )
407  {
408  DoRealloc( newSize * 4 / 3 );
409  }
410  }
411  Assert( m_nUsed < m_table.Count() );
412  ++m_nUsed;
413 
414  entry_t* table = m_table.Base();
415  unsigned int slotmask = m_table.Count()-1;
416  unsigned int new_flags_and_hash = FLAG_LAST | (h & MASK_HASH);
417  unsigned int idx = entry_t::IdealIndex( h, slotmask );
418  if ( table[idx].IdealIndex( slotmask ) == idx )
419  {
420  // There is already an entry in this chain.
421  new_flags_and_hash &= ~FLAG_LAST;
422  BumpEntry(idx);
423  }
424  else if ( table[idx].IsValid() )
425  {
426  // Somebody else is living in our ideal index but does not belong
427  // to our entry chain; move it out of the way, start a new chain.
428  BumpEntry(idx);
429  }
430  table[idx].flags_and_hash = new_flags_and_hash;
431  return idx;
432 }

435 // Key lookup. Can also return previous-in-chain if result is a chained slot.
436 template <typename KeyT, typename ValueT, typename KeyHashT, typename KeyIsEqualT, typename AltKeyT>
437 template <typename KeyParamT>
438 UtlHashHandle_t CUtlHashtable<KeyT, ValueT, KeyHashT, KeyIsEqualT, AltKeyT>::DoLookup( KeyParamT x, unsigned int h, handle_t *pPreviousInChain ) const
439 {
440  if ( m_nUsed == 0 )
441  {
442  // Empty table.
443  return (handle_t) -1;
444  }
445 
446  const entry_t* table = m_table.Base();
447  unsigned int slotmask = m_table.Count()-1;
448  Assert( m_table.Count() > 0 && (slotmask & m_table.Count()) == 0 );
449  unsigned int chainid = entry_t::IdealIndex( h, slotmask );
450 
451  unsigned int idx = chainid;
452  if ( table[idx].IdealIndex( slotmask ) != chainid )
453  {
454  // Nothing in root position? No match.
455  return (handle_t) -1;
456  }
457 
458  // Linear scan until found or end of chain
459  handle_t lastIdx = (handle_t) -1;
460  while (1)
461  {
462  // Only examine this slot if it is valid and belongs to our hash chain
463  if ( table[idx].IdealIndex( slotmask ) == chainid )
464  {
465  // Test the full-width hash to avoid unnecessary calls to m_eq()
466  if ( ((table[idx].flags_and_hash ^ h) & MASK_HASH) == 0 && m_eq( table[idx]->m_key, x ) )
467  {
468  // Found match!
469  if (pPreviousInChain)
470  *pPreviousInChain = lastIdx;
471 
472  return (handle_t) idx;
473  }
474 
475  if ( table[idx].flags_and_hash & FLAG_LAST )
476  {
477  // End of chain. No match.
478  return (handle_t) -1;

533 // Remove single element by key + hash. Returns the location of the new empty hole.
534 template <typename KeyT, typename ValueT, typename KeyHashT, typename KeyIsEqualT, typename AltKeyT>
535 template <typename KeyParamT>
536 int CUtlHashtable<KeyT, ValueT, KeyHashT, KeyIsEqualT, AltKeyT>::DoRemove( KeyParamT x, unsigned int h )
537 {
538  unsigned int slotmask = m_table.Count()-1;
539  handle_t previous = (handle_t) -1;
540  int idx = (int) DoLookup<KeyParamT>( x, h, &previous );
541  if (idx == -1)
542  {
543  return -1;
544  }
545 
546  enum { FAKEFLAG_ROOT = 1 };
547  int nLastAndRootFlags = m_table[idx].flags_and_hash & FLAG_LAST;
548  nLastAndRootFlags |= ( (uint)idx == m_table[idx].IdealIndex( slotmask ) );
549 
550  // Remove from table
551  m_table[idx].MarkInvalid();
552  Destruct( m_table[idx].Raw() );
553  --m_nUsed;
554 
555  if ( nLastAndRootFlags == FLAG_LAST ) // last only, not root
556  {
557  // This was the end of the chain - mark previous as last.
558  // (This isn't the root, so there must be a previous.)
559  Assert( previous != (handle_t) -1 );
560  m_table[previous].flags_and_hash |= FLAG_LAST;
561  }
562 
563  if ( nLastAndRootFlags == FAKEFLAG_ROOT ) // root only, not last
564  {
565  // If we are removing the root and there is more to the chain,
566  // scan to find the next chain entry and move it to the root.
567  unsigned int chainid = entry_t::IdealIndex( h, slotmask );
568  unsigned int nextIdx = idx;
569  while (1)
570  {
571  nextIdx = (nextIdx + 1) & slotmask;
572  if ( m_table[nextIdx].IdealIndex( slotmask ) == chainid )
573  {
574  break;
575  }
576  }
577  Assert( !(m_table[nextIdx].flags_and_hash & FLAG_FREE) );
578 
579  // Leave a hole where the next entry in the chain was.
580  m_table[idx].flags_and_hash = m_table[nextIdx].flags_and_hash;
581  m_table[idx].MoveDataFrom( m_table[nextIdx] );
582  m_table[nextIdx].MarkInvalid();
583  return nextIdx;
584  }
585 
586  // The hole is still where the element used to be.
587  return idx;
588 }
