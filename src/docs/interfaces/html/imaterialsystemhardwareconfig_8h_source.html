46 // For now, vertex compression is simply "on or off" (for the sake of simplicity
47 // and MeshBuilder perf.), but later we may support multiple flavours.
48 enum VertexCompressionType_t
49 {
50  // This indicates an uninitialized VertexCompressionType_t value
51  VERTEX_COMPRESSION_INVALID = 0xFFFFFFFF,
52 
53  // 'VERTEX_COMPRESSION_NONE' means that no elements of a vertex are compressed
54  VERTEX_COMPRESSION_NONE = 0,
55 
56  // Currently (more stuff may be added as needed), 'VERTEX_COMPRESSION_ON' means:
57  // - if a vertex contains VERTEX_ELEMENT_NORMAL, this is compressed
58  // (see CVertexBuilder::CompressedNormal3f)
59  // - if a vertex contains VERTEX_ELEMENT_USERDATA4 (and a normal - together defining a tangent
60  // frame, with the binormal reconstructed in the vertex shader), this is compressed
61  // (see CVertexBuilder::CompressedUserData)
62  // - if a vertex contains VERTEX_ELEMENT_BONEWEIGHTSx, this is compressed
63  // (see CVertexBuilder::CompressedBoneWeight3fv)
64  VERTEX_COMPRESSION_ON = 1
65 };
