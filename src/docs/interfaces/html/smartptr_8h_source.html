49 //
50 // CPlainAutoPtr
51 // is a smart wrapper for a pointer on the stack that performs "delete" upon destruction.
52 //
53 // No reference counting is performed, copying is prohibited "s_p2.Attach( s_p1.Detach() )" should be used
54 // for readability and ease of maintenance.
55 //
56 // Auto pointer supports an "arrow" operator for invoking methods on the pointee and a "dereference" operator
57 // for getting a pointee reference.
58 //
59 // No automatic casting to bool/ptrtype is performed to avoid bugs and problems (read on "safe bool idiom"
60 // if casting to bool or pointer happens to be useful).
61 //
62 // Test for validity with "IsValid", get the pointer with "Get".
63 //
64 template < typename T >
65 class CPlainAutoPtr
66 {
67 public:
68  explicit CPlainAutoPtr( T *p = NULL ) : m_p( p ) {}
69  ~CPlainAutoPtr( void ) { Delete(); }
70 
71 public:
72  void Delete( void ) { delete Detach(); }
73 
74 private: // Disallow copying, use Detach() instead to avoid ambiguity
75  CPlainAutoPtr( CPlainAutoPtr const &x );
76  CPlainAutoPtr & operator = ( CPlainAutoPtr const &x );
77 
78 public:
79  void Attach( T *p ) { m_p = p; }
80  T * Detach( void ) { T * p( m_p ); m_p = NULL; return p; }
81 
82 public:
83  bool IsValid( void ) const { return m_p != NULL; }
84  T * Get( void ) const { return m_p; }
85  T * operator -> ( void ) const { return Get(); }
86  T & operator * ( void ) const { return *Get(); }
87 
88 private:
89  T * m_p;
90 };

92 //
93 // CArrayAutoPtr
94 // is a smart wrapper for an array pointer on the stack that performs "delete []" upon destruction.
95 //
96 // No reference counting is performed, copying is prohibited "s_p2.Attach( s_p1.Detach() )" should be used
97 // for readability and ease of maintenance.
98 //
99 // Auto pointer supports an "indexing" operator for accessing array elements.
100 //
101 // No automatic casting to bool/ptrtype is performed to avoid bugs and problems (read on "safe bool idiom"
102 // if casting to bool or pointer happens to be useful).
103 //
104 // Test for validity with "IsValid", get the array pointer with "Get".
105 //
106 template < typename T >
107 class CArrayAutoPtr : public CPlainAutoPtr < T > // Warning: no polymorphic destructor (delete on base class will be a mistake)
108 {
109 public:
110  explicit CArrayAutoPtr( T *p = NULL ) { this->Attach( p ); }
111  ~CArrayAutoPtr( void ) { this->Delete(); }
112 
113 public:
114  void Delete( void ) { delete [] CPlainAutoPtr < T >::Detach(); }
115 
116 public:
117  T & operator [] ( int k ) const { return CPlainAutoPtr < T >::Get()[ k ]; }
118 };

121 // Smart pointers can be used to automatically free an object when nobody points
122 // at it anymore. Things contained in smart pointers must implement AddRef and Release
123 // functions. If those functions are private, then the class must make
124 // CRefCountAccessor a friend.

136  const T* operator->() const;

138  bool operator!() const;
139  bool operator==( const T *pOther ) const;
140  bool IsValid() const; // Tells if the pointer is valid.
141  T* GetObject() const; // Get temporary object pointer, don't store it for later reuse!

143

246 // CAutoPushPop
247 // allows you to set value of a variable upon construction and destruction.
248 // Constructors:
249 // CAutoPushPop x( myvar )
250 // saves the value and restores upon destruction.
251 // CAutoPushPop x( myvar, newvalue )
252 // saves the value, assigns new value upon construction, restores saved value upon destruction.
253 // CAutoPushPop x( myvar, newvalue, restorevalue )
254 // assigns new value upon construction, assignes restorevalue upon destruction.
255 //

257 class CAutoPushPop
258 {
259 public:
260  explicit CAutoPushPop( T& var ) : m_rVar( var ), m_valPop( var ) {}
261  CAutoPushPop( T& var, T const &valPush ) : m_rVar( var ), m_valPop( var ) { m_rVar = valPush; }
262  CAutoPushPop( T& var, T const &valPush, T const &valPop ) : m_rVar( var ), m_valPop( var ) { m_rVar = valPush; }
263 
264  ~CAutoPushPop() { m_rVar = m_valPop; }
265 
266 private: // forbid copying
267  CAutoPushPop( CAutoPushPop const &x );
268  CAutoPushPop & operator = ( CAutoPushPop const &x );
269 
270 public:

272 
273 private:

275  T m_valPop;
276 };
