64 // CGLMProgram can contain two flavors of the same program, one in assembler, one in GLSL.
65 // these flavors are pretty different in terms of the API's that are used to activate them -
66 // for example, assembler programs can just get bound to the context, whereas GLSL programs
67 // have to be linked. To some extent we try to hide that detail inside GLM.
68 
69 // for now, make CGLMProgram a container, it does not set policy or hold a preference as to which
70 // flavor you want to use. GLMContext has to handle that.

232  FORCEINLINE void UpdateScreenUniform( uint nWidthHeight )
233  {
234  if ( m_nScreenWidthHeight == nWidthHeight )
235  return;
236 
237  m_nScreenWidthHeight = nWidthHeight;
238 
239  float fWidth = (float)( nWidthHeight & 0xFFFF ), fHeight = (float)( nWidthHeight >> 16 );
240  // Apply half pixel offset to output vertices to account for the pixel center difference between D3D9 and OpenGL.
241  // We output vertices in clip space, which ranges from [-1,1], so 1.0/width in clip space transforms into .5/width in screenspace, see: "Viewports and Clipping (Direct3D 9)" in the DXSDK
242  float v[4] = { 1.0f / fWidth, 1.0f / fHeight, fWidth, fHeight };
243  if ( m_locVertexScreenParams >= 0 )
244  gGL->glUniform4fv( m_locVertexScreenParams, 1, v );
245  }

259  // actually we only need it for samplers and params.

383 FORCEINLINE void CGLMShaderPairCache::HashRowProbe( CGLMPairCacheEntry *row, CGLMProgram *vp, CGLMProgram *fp, uint extraKeyBits, int& hitway, int& emptyway, int& oldestway )
384 {
385  hitway = -1;
386  emptyway = -1;
387  oldestway = -1;
388 
389  // scan this row to see if the desired pair is present
390  CGLMPairCacheEntry *cursor = row;
391  long long oldestmark = 0xFFFFFFFFFFFFFFFFLL;
392 
393  for( uint way = 0; way < m_ways; ++way )
394  {
395  if ( cursor->m_lastMark != 0 ) // occupied slot
396  {
397  // check if this is the oldest one on the row - only occupied slots are checked
398  if ( cursor->m_lastMark < oldestmark )
399  {
400  oldestway = way;
401  oldestmark = cursor->m_lastMark;
402  }
403 
404  if ( ( cursor->m_vertexProg == vp ) && ( cursor->m_fragmentProg == fp ) && ( cursor->m_extraKeyBits == extraKeyBits ) ) // match?
405  {
406  // found it
407  hitway = way;
408  break;
409  }
410  }
411  else
412  {
413  // empty way, log it if first one seen
414  if (emptyway<0)
415  {
416  emptyway = way;
417  }
418  }
419  cursor++;
420  }
421 }

423 FORCEINLINE CGLMShaderPair *CGLMShaderPairCache::SelectShaderPair( CGLMProgram *vp, CGLMProgram *fp, uint extraKeyBits )
424 {
425  // select row where pair would be found if it exists
426  uint rowIndex = HashRowIndex( vp, fp, extraKeyBits );
427 
428  CGLMPairCacheEntry *pCursor = HashRowPtr( rowIndex );
429 
430  if ( ( pCursor->m_fragmentProg != fp ) || ( pCursor->m_vertexProg != vp ) || ( pCursor->m_extraKeyBits != extraKeyBits ) )
431  {
432  CGLMPairCacheEntry *pLastCursor = pCursor + m_ways;
433 
434  ++pCursor;
435 
436  while ( pCursor != pLastCursor )
437  {
438  if ( ( pCursor->m_fragmentProg == fp ) && ( pCursor->m_vertexProg == vp ) && ( pCursor->m_extraKeyBits == extraKeyBits ) ) // match?
439  break;
440  ++pCursor;
441  };
442 
443  if ( pCursor == pLastCursor )
444  return SelectShaderPairInternal( vp, fp, extraKeyBits, rowIndex );
445  }
446 
447  // found it. mark it and return
448  pCursor->m_lastMark = m_mark++;
449 
450 #if GL_SHADER_PAIR_CACHE_STATS
451  // count the hit
452  m_hits[ rowIndex ] ++;
453 #endif
454 
455  return pCursor->m_pair;
456 }
