92 class CTelemetryLock
93 {
94 public:
95  CTelemetryLock(void *plocation, const char *description)
96  {
97  m_plocation = (const char *)plocation;
98  m_description = description;
99  TelemetrySetLockName( TELEMETRY_LEVEL1, m_plocation, m_description );
100  tmTryLock( TELEMETRY_LEVEL1, m_plocation, "%s", m_description );
101  }
102  ~CTelemetryLock()
103  {
104  Unlocked();
105  }
106  void Locked()
107  {
108  tmEndTryLock( TELEMETRY_LEVEL1, m_plocation, TMLR_SUCCESS );
109  tmSetLockState( TELEMETRY_LEVEL1, m_plocation, TMLS_LOCKED, "%s Locked", m_description );
110  }
111  void Unlocked()
112  {
113  if( m_plocation )
114  {
115  tmSetLockState( TELEMETRY_LEVEL1, m_plocation, TMLS_RELEASED, "%s Released", m_description );
116  m_plocation = NULL;
117  }
118  }
119 
120 public:
121  const char *m_plocation;
122  const char *m_description;
123 };

129  CTelemetrySpikeDetector( const char *msg, float threshold = 5 ) :
130  m_message( msg ), m_threshold( threshold ), time0( tmFastTime() ) {}
131  ~CTelemetrySpikeDetector()
132  {
133  float time = ( tmFastTime() - time0 ) * g_Telemetry.flRDTSCToMilliSeconds;
134  if( time >= m_threshold )
135  {
136  tmMessage( TELEMETRY_LEVEL0, TMMF_ICON_NOTE | TMMF_SEVERITY_WARNING, "(source/spike)%s %.2fms %t", m_message, time, tmSendCallStack( TELEMETRY_LEVEL0, 0 ) );
137  }
138  }

143  const char *m_message;
