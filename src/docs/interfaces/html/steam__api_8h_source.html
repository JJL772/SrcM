137 //----------------------------------------------------------------------------------------------------------------------------------------------------------//
138 // steam callback and call-result helpers
139 //
140 // The following macros and classes are used to register your application for
141 // callbacks and call-results, which are delivered in a predictable manner.
142 //
143 // STEAM_CALLBACK macros are meant for use inside of a C++ class definition.
144 // They map a Steam notification callback directly to a class member function
145 // which is automatically prototyped as "void func( callback_type *pParam )".
146 //
147 // CCallResult is used with specific Steam APIs that return "result handles".
148 // The handle can be passed to a CCallResult object's Set function, along with
149 // an object pointer and member-function pointer. The member function will
150 // be executed once the results of the Steam API call are available.
151 //
152 // CCallback and CCallbackManual classes can be used instead of STEAM_CALLBACK
153 // macros if you require finer control over registration and unregistration.
154 //
155 // Callbacks and call-results are queued automatically and are only
156 // delivered/executed when your application calls SteamAPI_RunCallbacks().
157 //----------------------------------------------------------------------------------------------------------------------------------------------------------//
158

164 // registers the callback on object creation and unregisters on destruction.
165 // The optional fourth 'var' param exists only for backwards-compatibility
166 // and can be ignored.
167 #define STEAM_CALLBACK( thisclass, func, .../*callback_type, [deprecated] var*/ ) \

225 //-----------------------------------------------------------------------------
226 // Purpose: maps a steam async call result to a class member function
227 // template params: T = local class, P = parameter struct
228 //-----------------------------------------------------------------------------
229 template< class T, class P >
230 class CCallResult : private CCallbackBase
231 {
232 public:
233  typedef void (T::*func_t)( P*, bool );
234 
235  CCallResult()
236  {
237  m_hAPICall = k_uAPICallInvalid;
238  m_pObj = NULL;
239  m_Func = NULL;
240  m_iCallback = P::k_iCallback;
241  }
242 
243  void Set( SteamAPICall_t hAPICall, T *p, func_t func )
244  {
245  if ( m_hAPICall )
246  SteamAPI_UnregisterCallResult( this, m_hAPICall );
247 
248  m_hAPICall = hAPICall;
249  m_pObj = p;
250  m_Func = func;
251 
252  if ( hAPICall )
253  SteamAPI_RegisterCallResult( this, hAPICall );
254  }
255 
256  bool IsActive() const
257  {
258  return ( m_hAPICall != k_uAPICallInvalid );
259  }
260 
261  void Cancel()
262  {
263  if ( m_hAPICall != k_uAPICallInvalid )
264  {
265  SteamAPI_UnregisterCallResult( this, m_hAPICall );
266  m_hAPICall = k_uAPICallInvalid;
267  }
268 
269  }
270 
271  ~CCallResult()
272  {
273  Cancel();
274  }
275 
276  void SetGameserverFlag() { m_nCallbackFlags |= k_ECallbackFlagsGameServer; }
277 private:
278  virtual void Run( void *pvParam )
279  {
280  m_hAPICall = k_uAPICallInvalid; // caller unregisters for us
281  (m_pObj->*m_Func)( (P *)pvParam, false );
282  }
283  virtual void Run( void *pvParam, bool bIOFailure, SteamAPICall_t hSteamAPICall )
284  {
285  if ( hSteamAPICall == m_hAPICall )
286  {
287  m_hAPICall = k_uAPICallInvalid; // caller unregisters for us
288  (m_pObj->*m_Func)( (P *)pvParam, bIOFailure );
289  }
290  }
291  virtual int GetCallbackSizeBytes()
292  {
293  return sizeof( P );
294  }
295 
296  SteamAPICall_t m_hAPICall;
297  T *m_pObj;
298  func_t m_Func;
299 };

303 //-----------------------------------------------------------------------------
304 // Purpose: maps a steam callback to a class member function
305 // template params: T = local class, P = parameter struct,
306 // bGameserver = listen for gameserver callbacks instead of client callbacks
307 //-----------------------------------------------------------------------------
308 template< class T, class P, bool bGameserver = false >
309 class CCallback : public CCallbackImpl< sizeof( P ) >
310 {
311 public:
312  typedef void (T::*func_t)(P*);
313 
314  // NOTE: If you can't provide the correct parameters at construction time, you should
315  // use the CCallbackManual callback object (STEAM_CALLBACK_MANUAL macro) instead.
316  CCallback( T *pObj, func_t func ) : m_pObj( NULL ), m_Func( NULL )
317  {
318  if ( bGameserver )
319  {
320  this->SetGameserverFlag();
321  }
322  Register( pObj, func );
323  }
324 
325  // manual registration of the callback
326  void Register( T *pObj, func_t func )
327  {
328  if ( !pObj || !func )
329  return;
330 
331  if ( this->m_nCallbackFlags & CCallbackBase::k_ECallbackFlagsRegistered )
332  Unregister();
333 
334  m_pObj = pObj;
335  m_Func = func;
336  // SteamAPI_RegisterCallback sets k_ECallbackFlagsRegistered
337  SteamAPI_RegisterCallback( this, P::k_iCallback );
338  }
339 
340  void Unregister()
341  {
342  // SteamAPI_UnregisterCallback removes k_ECallbackFlagsRegistered
343  SteamAPI_UnregisterCallback( this );
344  }
345 
346 protected:
347  virtual void Run( void *pvParam )
348  {
349  (m_pObj->*m_Func)( (P *)pvParam );
350  }
351 
352  T *m_pObj;
353  func_t m_Func;
354 };
