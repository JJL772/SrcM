300 inline ButtonCode_t GetBaseButtonCode( ButtonCode_t code )
301 {
302  if ( IsJoystickButtonCode( code ) )
303  {
304  int offset = ( code - JOYSTICK_FIRST_BUTTON ) % JOYSTICK_MAX_BUTTON_COUNT;
305  return (ButtonCode_t)( JOYSTICK_FIRST_BUTTON + offset );
306  }
307 
308  if ( IsJoystickPOVCode( code ) )
309  {
310  int offset = ( code - JOYSTICK_FIRST_POV_BUTTON ) % JOYSTICK_POV_BUTTON_COUNT;
311  return (ButtonCode_t)( JOYSTICK_FIRST_POV_BUTTON + offset );
312  }
313 
314  if ( IsJoystickAxisCode( code ) )
315  {
316  int offset = ( code - JOYSTICK_FIRST_AXIS_BUTTON ) % JOYSTICK_AXIS_BUTTON_COUNT;
317  return (ButtonCode_t)( JOYSTICK_FIRST_AXIS_BUTTON + offset );
318  }
319 
320  return code;
321 }

323 inline int GetJoystickForCode( ButtonCode_t code )
324 {
325  if ( !IsJoystickCode( code ) )
326  return 0;
327 
328  if ( IsJoystickButtonCode( code ) )
329  {
330  int offset = ( code - JOYSTICK_FIRST_BUTTON ) / JOYSTICK_MAX_BUTTON_COUNT;
331  return offset;
332  }
333  if ( IsJoystickPOVCode( code ) )
334  {
335  int offset = ( code - JOYSTICK_FIRST_POV_BUTTON ) / JOYSTICK_POV_BUTTON_COUNT;
336  return offset;

340  int offset = ( code - JOYSTICK_FIRST_AXIS_BUTTON ) / JOYSTICK_AXIS_BUTTON_COUNT;
341  return offset;

347 inline ButtonCode_t ButtonCodeToJoystickButtonCode( ButtonCode_t code, int nDesiredJoystick )
348 {
349  if ( !IsJoystickCode( code ) || nDesiredJoystick == 0 )
350  return code;
351 
352  nDesiredJoystick = ::clamp<int>( nDesiredJoystick, 0, MAX_JOYSTICKS - 1 );
353 
354  code = GetBaseButtonCode( code );
355 
356  // Now upsample it
357  if ( IsJoystickButtonCode( code ) )
358  {
359  int nOffset = code - JOYSTICK_FIRST_BUTTON;
360  return JOYSTICK_BUTTON( nDesiredJoystick, nOffset );
361  }
362 
363  if ( IsJoystickPOVCode( code ) )
364  {
365  int nOffset = code - JOYSTICK_FIRST_POV_BUTTON;
366  return JOYSTICK_POV_BUTTON( nDesiredJoystick, nOffset );
367  }
368 
369  if ( IsJoystickAxisCode( code ) )
370  {
371  int nOffset = code - JOYSTICK_FIRST_AXIS_BUTTON;
372  return JOYSTICK_AXIS_BUTTON( nDesiredJoystick, nOffset );
373  }
374 
375  return code;
376 }
