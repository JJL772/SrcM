21 //
22 // These templates are used for intrusive linked list classes. Intrusive linked list templates
23 // force the structs and classes contained within them to have their own m_pNext, (optionally),
24 // m_pPrev, and other fields contained within. All memory management is up to the caller and their
25 // classes. No data will ever be copied. Nodes can only exist on one list at a time, because of
26 // only having on m_Next field, and manipulating the list while walking it requires that care on
27 // the part of the caller. All accessing and searching functions work by passing and returning
28 // pointers.
29 //
30 //
31 //
32 // naming and field conventions:
33 // functions referring to a DList are for doubly linked lists. nodes must have m_pHead and
34 // m_pPrev pointer fields.
35 // Functions using Priority require an m_Priority field, which must be comparable.
36 //
37 // Some functions are mean for use with lists which maintain both a head and tail pointer
38 // in order to support fast adding to the end.

73 // move a node in a doubly linked list backwards one step.
74  template <class T> inline void MoveDNodeBackwards( T *which, T * &head)
75  {
76  if (which->m_pPrev)
77  {
78  T *p=which->m_pPrev;
79  T *pp=p->m_pPrev;
80  T *n=which->m_pNext;
81  Assert(p->m_pNext == which);
82  if (n)
83  {
84  Assert(n->m_pPrev==which);
85  n->m_pPrev=p;
86  }
87  if (pp)
88  {
89  Assert(pp->m_pNext==p);
90  pp->m_pNext=which;
91  }
92  else
93  {
94  head=which; // this node is the new root!
95  }
96  which->m_pNext=p;
97  which->m_pPrev=pp;
98  p->m_pNext=n;
99  p->m_pPrev=which;
100  }
101  ValidateDList(head);
102  }

280  // Add a node to a d-list, keeping the highest priority nodes first. This is a simple
281  // linear search to insert, NOT a O(logn) heap.
282  template<class T> inline void AddToDPriority(T * &head, T *which)
283  {
284  T* prevnode=0;
285  for(T *curnode=head;curnode;curnode=curnode->m_pNext)
286  {
287  if (which->m_Priority>=curnode->m_Priority)
288  break;
289  prevnode=curnode;
290  }
291  // now, we have either run out of list, or we have found an
292  // element to add this one before
293  if (! prevnode)
294  {
295  AddToDHead(head,which);
296  }
297  else
298  {
299  which->m_pNext=prevnode->m_pNext;
300  prevnode->m_pNext=which;
301  which->m_pPrev=prevnode;
302  if (which->m_pNext)
303  which->m_pNext->m_pPrev=which;
304  }
305  }

608  // NOT fast.
609  template<class T> void SortList(T * &head, int (*comparefn)(T * a, T * b))
610  {
611  int didswap=1;
612  while(didswap)
613  {
614  didswap=0;
615  T *prev=0;
616  for(T *i=head;i && i->m_pNext; i=i->m_pNext)
617  {
619  int rslt=(*comparefn)(i,i->m_pNext);
620  if (rslt==-1)
621  {
623  didswap=1;
624  T *newfirst=i->m_pNext;
625  if (prev)
626  {
627  prev->m_pNext=newfirst;
628  i->m_pNext=newfirst->m_pNext;
629  newfirst->m_pNext=i;
630  }
631  else
632  {
633  head=i->m_pNext;
634  i->m_pNext=newfirst->m_pNext;
635  newfirst->m_pNext=i;
636  }
637  i=newfirst;
638  }
639  prev=i;
640  }
641  }
