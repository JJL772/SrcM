17 template <typename T>
18 inline void SafeAssign(T** ppInoutDst, T* pInoutSrc )
19 {
20  Assert( ppInoutDst );
21 
22  // Do addref before release
23  if ( pInoutSrc )
24  ( pInoutSrc )->AddRef();
25 
26  // Do addref before release
27  if ( *ppInoutDst )
28  ( *ppInoutDst )->Release();
29 
30  // Do the assignment
31  ( *ppInoutDst ) = pInoutSrc;
32 }

69 template <class REFCOUNTED_ITEM_PTR>
70 inline int SafeRelease( REFCOUNTED_ITEM_PTR &pRef )
71 {
72  // Use funny syntax so that this works on "auto pointers"
73  REFCOUNTED_ITEM_PTR *ppRef = &pRef;
74  if ( *ppRef )
75  {
76  int result = (*ppRef)->Release();
77  *ppRef = NULL;
78  return result;
79  }
80  return 0;
81 }
