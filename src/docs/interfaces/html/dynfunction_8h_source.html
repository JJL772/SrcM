9 // This makes it easy to dynamically load a shared library and lookup a
10 // function in that library.
11 //
12 // Usage:
13 // CDynamicFunction<void (*)(const char *)> MyPuts(libname, "puts");
14 // if (MyPuts)
15 // MyPuts("Hello world!");
16 //
17 // Please note that this interface does not distinguish between functions and
18 // data. If you look up a global variable in your shared library, or simply
19 // mess up the function signature, you'll get a valid pointer and a crash
20 // if you call it as a function.

31 template < class FunctionType >
32 class CDynamicFunction
33 {
34 public:
35  // Construct with a NULL function pointer. You must manually call
36  // Lookup() before you can call a dynamic function through this interface.
37  CDynamicFunction() : m_pFn(NULL) {}
38 
39  // Construct and do a lookup right away. You will need to make sure that
40  // the lookup actually succeeded, as (libname) might have failed to load
41  // or (fn) might not exist in it.
42  CDynamicFunction(const char *libname, const char *fn, FunctionType fallback=NULL) : m_pFn(NULL)
43  {
44  Lookup(libname, fn, fallback);
45  }
46 
47  // Construct and do a lookup right away. See comments in Lookup() about what (okay) does.
48  CDynamicFunction(const char *libname, const char *fn, bool &okay, FunctionType fallback=NULL) : m_pFn(NULL)
49  {
50  Lookup(libname, fn, okay, fallback);
51  }
52 
53  // Load library if necessary, look up symbol. Returns true and sets
54  // m_pFn on successful lookup, returns false otherwise. If the
55  // function pointer is already looked up, this return true immediately.
56  // Use Reset() first if you want to look up the symbol again.
57  // This function will return false immediately unless (okay) is true.
58  // This allows you to chain lookups like this:
59  // bool okay = true;
60  // x.Lookup(lib, "x", okay);
61  // y.Lookup(lib, "y", okay);
62  // z.Lookup(lib, "z", okay);
63  // if (okay) { printf("All functions were loaded successfully!\n"); }
64  // If you supply a fallback, it'll be used if the lookup fails (and if
65  // non-NULL, means this will always return (okay)).
66  bool Lookup(const char *libname, const char *fn, bool &okay, FunctionType fallback=NULL)
67  {
68  if (!okay)
69  return false;
70  else if (m_pFn == NULL)
71  m_pFn = (FunctionType) VoidFnPtrLookup_Tier0(libname, fn, (void *) fallback);
72  okay = m_pFn != NULL;
73  return okay;
74  }
75 
76  // Load library if necessary, look up symbol. Returns true and sets
77  // m_pFn on successful lookup, returns false otherwise. If the
78  // function pointer is already looked up, this return true immediately.
79  // Use Reset() first if you want to look up the symbol again.
80  // This function will return false immediately unless (okay) is true.
81  // If you supply a fallback, it'll be used if the lookup fails (and if
82  // non-NULL, means this will always return true).
83  bool Lookup(const char *libname, const char *fn, FunctionType fallback=NULL)
84  {
85  bool okay = true;
86  return Lookup(libname, fn, okay, fallback);
87  }
88 
89  // Invalidates the current lookup. Makes the function pointer NULL. You
90  // will need to call Lookup() before you can call a dynamic function
91  // through this interface again.
92  void Reset() { m_pFn = NULL; }
93 
94  // Force this to be a specific function pointer.
95  void Force(FunctionType ptr) { m_pFn = ptr; }
96 
97  // Retrieve the actual function pointer.
98  FunctionType Pointer() const { return m_pFn; }
99  operator FunctionType() const { return m_pFn; }
100 
101  // Can be used to verify that we have an actual function looked up and
102  // ready to call: if (!MyDynFunc) { printf("Function not found!\n"); }
103  operator bool () const { return m_pFn != NULL; }
104  bool operator !() const { return m_pFn == NULL; }
105 
106 protected:
107  FunctionType m_pFn;
108 };
