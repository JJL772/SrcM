464 inline unsigned short CSaveRestoreSegment::FindCreateSymbol( const char *pszToken )
465 {
466  unsigned short hash = (unsigned short)(HashString( pszToken ) % (unsigned)tokenCount );
467 
468 #if _DEBUG
469  static int tokensparsed = 0;
470  tokensparsed++;
471  if ( !tokenCount || !pTokens )
472  {
473  AssertMsg( 0, ("No token table array in TokenHash()!") );
474  }
475 #endif
476 
477  for ( int i=0; i<tokenCount; i++ )
478  {
479 #if _DEBUG

486 #endif

489  if ( index >= tokenCount )
490  index -= tokenCount;
491

493  {

514 #ifndef _WIN32
515 inline unsigned CSaveRestoreSegment::_rotr ( unsigned val, int shift)
516 {
517  unsigned lobit; /* non-zero means lo bit set */
518  unsigned num = val; /* number to rotate */
519 
520  shift &= 0x1f; /* modulo 32 -- this will also make
521  negative shifts work */
522 
523  while (shift--)
524  {
525  lobit = num & 1; /* get high bit */
526  num >>= 1; /* shift right one bit */
527  if (lobit)
528  num |= 0x80000000; /* set hi bit if lo bit was set */
529  }
530 
531  return num;
532 }
533 #endif
