20 //
21 // A union of all the possible password hash types.
22 // When adding to this, if the maximum size of the
23 // structure has changed then the Account table and
24 // the AccountSecurityHistory table need to be manually
25 // revised using something like the following SQL:
26 //
27 // ALTER TABLE Account ALTER COLUMN PasswordHash binary(32)
28 // ALTER TABLE AccountSecurityHistory ALTER COLUMN PasswordHashOld binary(32)
29 // ALTER TABLE AccountSecurityHistory ALTER COLUMN PasswordHashNew binary(32)
30 //
31 // Replace 32 with the new size of PasswordHash_t.
32 //
33 // If the width of those columns does not match sizeof(PasswordHash_t), many
34 // database operations will fail.
35 //
36 // Note that while this query was correct at the time this code was checked in,
37 // it may not be sufficient at the time you actually change the size of the
38 // type, so look around.
39 //
40 typedef union
41 {
42  SHADigest_t sha;
43  BigPasswordHash_t bigpassword;
44  PBKDF2Hash_t pbkdf2;
45 } PasswordHash_t;

48 //
49 // Enum of all available password hash algorithms. These should
50 // be consecutive and ordinals should never be reused.
51 //
52 // k_EHashSHA1: A salted SHA-1 hash, width 20 bytes.
53 // k_EHashBigPassword: For testing purposes, a salted SHA-1 hash extended to 32 bytes, with 6 bytes of 0x1 on either side.
54 // k_EHashPBKDF2_1000: A PKCS#5 v2.0 Password-Based Key Derivation Function hash (PBKDF2-HMAC-SHA256) with 1,000 iterations.
55 // The digest width is 32 bytes.
56 // k_EHashPBKDF2_5000: A PKCS#5 v2.0 Password-Based Key Derivation Function hash (PBKDF2-HMAC-SHA256) with 5,000 iterations.
57 // k_EHashPBKDF2_10000: A PKCS#5 v2.0 Password-Based Key Derivation Function hash (PBKDF2-HMAC-SHA256) with 10,000 iterations.
58 // k_EHashSHA1WrappedWithPBKDF2_10000: A SHA-1 hash which is then further hashed with PBKDF2 at 10,000 rounds. Used for
59 // strengthening old hashes in the database that haven't been logged in in a long time.
60 //
61 // Make sure to update k_EHashMax when adding new hash types. Also add the length into the k_HashLengths array below.
62 enum EPasswordHashAlg
63 {
64  k_EHashSHA1 = 0,
65  k_EHashBigPassword = 1,
66  k_EHashPBKDF2_1000 = 2,
67  k_EHashPBKDF2_5000 = 3,
68  k_EHashPBKDF2_10000 = 4,
69  k_EHashSHA1WrappedWithPBKDF2_10000 = 5,
70  k_EHashMax = 5,
71 };
