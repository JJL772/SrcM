220 // Purges the list and calls delete on each element in it.
221 template< typename K, typename T, typename I >
222 inline void CUtlMap<K, T, I>::PurgeAndDeleteElements()
223 {
224  for ( I i = 0; i < MaxElement(); ++i )
225  {
226  if ( !IsValidIndex( i ) )
227  continue;
228 
229  delete Element( i );
230  }
231 
232  Purge();
233 }

237 // This is horrible and slow and meant to be used only when you're dealing with really
238 // non-time/memory-critical code and desperately want to copy a whole map element-by-element

242 {
243  Assert( out_pmapOut );
244

246  FOR_EACH_MAP_FAST( pmapIn, i )

248  out_pmapOut->Insert( pmapIn.Key( i ), pmapIn.Element( i ) );
