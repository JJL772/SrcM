164  CTSListBase()
165  {
166  if ( ((size_t)&m_Head) % TSLIST_HEAD_ALIGNMENT != 0 )
167  {
168  Error( "CTSListBase: Misaligned list\n" );
169  DebuggerBreak();
170  }
171 
172 #ifdef USE_NATIVE_SLIST
173  InitializeSListHead( &m_Head );
174 #elif defined(PLATFORM_64BITS)
175  m_Head.value64x128 = int128_zero();
176 #else
177  m_Head.value64x128 = (int64)0;
178 #endif
179  }
180 
181  ~CTSListBase()
182  {
183  Detach();
184  }
185 
186  TSLNodeBase_t *Push( TSLNodeBase_t *pNode )
187  {
188 #ifdef _DEBUG
189  if ( (size_t)pNode % TSLIST_NODE_ALIGNMENT != 0 )
190  {
191  Error( "CTSListBase: Misaligned node\n" );
192  DebuggerBreak();
193  }
194 #endif
195 
196 #ifdef USE_NATIVE_SLIST
197 #ifdef _X360
198  // integrated write-release barrier
199  return (TSLNodeBase_t *)InterlockedPushEntrySListRelease( &m_Head, pNode );
200 #else
201  return (TSLNodeBase_t *)InterlockedPushEntrySList( &m_Head, pNode );
202 #endif
203 #else
204  TSLHead_t oldHead;
205  TSLHead_t newHead;
206 
207  #if defined( PLATFORM_PS3 ) || defined( PLATFORM_X360 )
208  __lwsync(); // write-release barrier
209  #endif
210 
211 #ifdef PLATFORM_64BITS
212  newHead.value.Padding = 0;
213 #endif
214  for (;;)
215  {
216  oldHead.value64x128 = m_Head.value64x128;
217  pNode->Next = oldHead.value.Next;
218  newHead.value.Next = pNode;
219 
220  newHead.value32.DepthAndSequence = oldHead.value32.DepthAndSequence + 0x10001;
221 
222 
223  if ( ThreadInterlockedAssignIf64x128( &m_Head.value64x128, newHead.value64x128, oldHead.value64x128 ) )
224  {
225  break;
226  }
227  ThreadPause();
228  };
229 
230  return (TSLNodeBase_t *)oldHead.value.Next;
231 #endif
232  }
233 
234  TSLNodeBase_t *Pop()
235  {
236 #ifdef USE_NATIVE_SLIST
237 #ifdef _X360
238  // integrated read-acquire barrier
239  TSLNodeBase_t *pNode = (TSLNodeBase_t *)InterlockedPopEntrySListAcquire( &m_Head );
240 #else
241  TSLNodeBase_t *pNode = (TSLNodeBase_t *)InterlockedPopEntrySList( &m_Head );
242 #endif
243  return pNode;
244 #else
245  TSLHead_t oldHead;
246  TSLHead_t newHead;
247 
248 #ifdef PLATFORM_64BITS
249  newHead.value.Padding = 0;
250 #endif
251  for (;;)
252  {
253  oldHead.value64x128 = m_Head.value64x128;
254  if ( !oldHead.value.Next )
255  return NULL;
256 
257  newHead.value.Next = oldHead.value.Next->Next;
258  newHead.value32.DepthAndSequence = oldHead.value32.DepthAndSequence - 1;
259 
260 
261  if ( ThreadInterlockedAssignIf64x128( &m_Head.value64x128, newHead.value64x128, oldHead.value64x128 ) )
262  {
263  #if defined( PLATFORM_PS3 ) || defined( PLATFORM_X360 )
264  __lwsync(); // read-acquire barrier
265  #endif
266  break;
267  }
268  ThreadPause();
269  };
270 
271  return (TSLNodeBase_t *)oldHead.value.Next;
272 #endif
273  }
274 
275  TSLNodeBase_t *Detach()
276  {
277 #ifdef USE_NATIVE_SLIST
278  TSLNodeBase_t *pBase = (TSLNodeBase_t *)InterlockedFlushSList( &m_Head );
279 #if defined( _X360 ) || defined( _PS3 )
280  __lwsync(); // read-acquire barrier
281 #endif
282  return pBase;
283 #else
284  TSLHead_t oldHead;
285  TSLHead_t newHead;
286 
287 #ifdef PLATFORM_64BITS
288  newHead.value.Padding = 0;
289 #endif
290  do
291  {
292  ThreadPause();
293 
294  oldHead.value64x128 = m_Head.value64x128;
295  if ( !oldHead.value.Next )
296  return NULL;
297 
298  newHead.value.Next = NULL;
299  // <sergiy> the reason for AND'ing it instead of poking a short into memory
300  // is probably to avoid store forward issues, but I'm not sure because
301  // I didn't construct this code. In any case, leaving it as is on big-endian
302  newHead.value32.DepthAndSequence = oldHead.value32.DepthAndSequence & 0xffff0000;
303 
304  } while( !ThreadInterlockedAssignIf64x128( &m_Head.value64x128, newHead.value64x128, oldHead.value64x128 ) );
305 
306  return (TSLNodeBase_t *)oldHead.value.Next;
307 #endif
308  }
309 
310  TSLHead_t *AccessUnprotected()
311  {
312  return &m_Head;
313  }
314 
315  int Count() const
316  {
317 #ifdef USE_NATIVE_SLIST
318  return QueryDepthSList( const_cast<TSLHead_t*>( &m_Head ) );
319 #else
320  return m_Head.value.Depth;
321 #endif
322  }

324 private:
325  TSLHead_t m_Head;
326 } TSLIST_HEAD_ALIGN_POST;

348 // have a per-item, per-alloc new/delete overhead
349 // similar to CTSSimpleList except that it allocates it's own pool objects
350 // and frees them on destruct. Also it does not overlay the TSNodeBase_t memory

352 template< class T >
353 class TSLIST_HEAD_ALIGN CTSPool : public CTSListBase
354 {
355  // packs the node and the item (T) into a single struct and pools those
356  struct TSLIST_NODE_ALIGN simpleTSPoolStruct_t : public TSLNodeBase_t
357  {
358  T elem;
359  } TSLIST_NODE_ALIGN_POST;
360 
361 public:
362 
363  ~CTSPool()
364  {
365  Purge();
366  }
367 
368  void Purge()
369  {
370  simpleTSPoolStruct_t *pNode = NULL;
371  while ( 1 )
372  {
373  pNode = (simpleTSPoolStruct_t *)CTSListBase::Pop();
374  if ( !pNode )
375  break;
376  delete pNode;
377  }
378  }
379 
380  void PutObject( T *pInfo )
381  {
382  char *pElem = (char *)pInfo;
383  pElem -= offsetof(simpleTSPoolStruct_t,elem);
384  simpleTSPoolStruct_t *pNode = (simpleTSPoolStruct_t *)pElem;
385 
386  CTSListBase::Push( pNode );
387  }
388 
389  T *GetObject()
390  {
391  simpleTSPoolStruct_t *pNode = (simpleTSPoolStruct_t *)CTSListBase::Pop();
392  if ( !pNode )
393  {
394  pNode = new simpleTSPoolStruct_t;
395  }
396  return &pNode->elem;
397  }
398 
399  // omg windows sdk - why do you #define GetObject()?
400  FORCEINLINE T *Get()
401  {

403  }

593 //
594 // A special consideration: the element type should be simple. This code
595 // actually dereferences freed nodes as part of pop, but later detects
596 // that. If the item in the queue is a complex type, only bad things can
597 // come of that. Also, therefore, if you're using Push/Pop instead of
598 // push item, be aware that the node memory cannot be freed until
599 // all threads that might have been popping have completed the pop.
600 // The PushItem()/PopItem() for handles this by keeping a persistent
601 // free list. Dont mix Push/PushItem. Note also nodes will be freed at the end,
602 // and are expected to have been allocated with operator new.

706  CTSQueue()
707  {
708  COMPILE_TIME_ASSERT( sizeof(Node_t) >= sizeof(TSLNodeBase_t) );
709  if ( ((size_t)&m_Head) % TSLIST_HEAD_ALIGNMENT != 0 )
710  {
711  Error( "CTSQueue: Misaligned queue\n" );
712  DebuggerBreak();
713  }
714  if ( ((size_t)&m_Tail) % TSLIST_HEAD_ALIGNMENT != 0 )
715  {
716  Error( "CTSQueue: Misaligned queue\n" );
717  DebuggerBreak();
718  }
719  m_Count = 0;
720  m_Head.value.sequence = m_Tail.value.sequence = 0;
721  m_Head.value.pNode = m_Tail.value.pNode = new Node_t; // list always contains a dummy node
722  m_Head.value.pNode->pNext = End();

871  Node_t *Pop()
872  {
873  #define TSQUEUE_BAD_NODE_LINK ( (Node_t *)INT_TO_POINTER( 0xdeadbeef ) )
874  NodeLink_t * volatile pHead = &m_Head;
875  NodeLink_t * volatile pTail = &m_Tail;
876  Node_t * volatile * pHeadNode = &m_Head.value.pNode;
877  volatile intp * volatile pHeadSequence = &m_Head.value.sequence;
878  Node_t * volatile * pTailNode = &pTail->value.pNode;
879 
880  NodeLink_t head;
881  NodeLink_t newHead;
882  Node_t *pNext;
883  intp tailSequence;
884  T elem;
885 
886  for (;;)
887  {
888  head.value.sequence = *pHeadSequence; // must grab sequence first, which allows condition below to ensure pNext is valid
889  ThreadMemoryBarrier(); // need a barrier to prevent reordering of these assignments
890  head.value.pNode = *pHeadNode;
891  tailSequence = pTail->value.sequence;
892  pNext = head.value.pNode->pNext;
893 
894  // Checking pNext only to force optimizer to not reorder the assignment
895  // to pNext and the compare of the sequence
896  if ( !pNext || head.value.sequence != *pHeadSequence )
897  continue;
898 
899  if ( bTestOptimizer )
900  {
901  if ( pNext == TSQUEUE_BAD_NODE_LINK )
902  {
903  Msg( "Bad node link detected\n" );
904  continue;
905  }
906  }
907 
908  if ( head.value.pNode == *pTailNode )
909  {
910  if ( pNext == End() )
911  return NULL;
912 
913  // Another thread is trying to push, help it along
914  NodeLink_t &oldTail = head; // just reuse local memory for head to build old tail
915  oldTail.value.sequence = tailSequence; // reuse head pNode
916  FinishPush( pNext, oldTail );
917  continue;
918  }
919 
920  if ( pNext != End() )
921  {
922  elem = pNext->elem; // NOTE: next could be a freed node here, by design
923  newHead.value.pNode = pNext;
924  newHead.value.sequence = head.value.sequence + 1;
925  if ( InterlockedCompareExchangeNodeLink( pHead, newHead, head ) )
926  {
927  ThreadMemoryBarrier();
928  if ( bTestOptimizer )
929  {
930  head.value.pNode->pNext = TSQUEUE_BAD_NODE_LINK;
931  }
932  break;
933  }
934  }
935  }
936 
937  m_Count--;
938  head.value.pNode->elem = elem;
939  return head.value.pNode;
940  }
