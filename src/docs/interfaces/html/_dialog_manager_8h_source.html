79 template <class TDialog, class I>
80 inline TDialog *DialogManager<TDialog, I>::FindDialog(I dialogID, bool bCreate)
81 {
82  for (int i = 0; i < m_Dialogs.MaxElementIndex(); i++)
83  {
84  if (ValidateIndex(i) && m_Dialogs[i].id == dialogID)
85  {
86  return m_Dialogs[i].dlg;
87  }
88  }
89 
90  if (bCreate)
91  {
92  int newIndex = m_Dialogs.AddToTail();
93  if (m_CreateFunc)
94  {
95  m_Dialogs[newIndex].dlg = m_CreateFunc(dialogID);
96  }
97  else
98  {
99  m_Dialogs[newIndex].dlg = new TDialog(NULL, dialogID);
100  }
101  Assert(m_pVGUIParentPanel);
102  m_Dialogs[newIndex].dlg->SetParent( m_pVGUIParentPanel );
103 
104  m_Dialogs[newIndex].id = dialogID;
105  return m_Dialogs[newIndex].dlg;
106  }
107 
108  // dlg not found, not created
109  return NULL;
110 }
