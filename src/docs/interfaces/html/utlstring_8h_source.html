431 template < typename T >
432 int CUtlConstStringBase<T>::Compare( const T *rhs ) const
433 {
434  // Empty or null RHS?
435  if ( !rhs || !rhs[0] )
436  return m_pString ? 1 : 0;
437 
438  // Empty *this, non-empty RHS?
439  if ( !m_pString )
440  return -1;
441 
442  // Neither empty
443  return StringFuncs<T>::Compare( m_pString, rhs );
444 }
