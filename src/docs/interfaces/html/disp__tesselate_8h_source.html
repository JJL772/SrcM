45 // Tesselates a single node, doesn't deal with hierarchy

47 template< class TesselateHelper >
48 inline void TesselateDisplacementNode(
49  TesselateHelper *pHelper,
50  CVertIndex const &nodeIndex,
51  int iLevel,
52  int *pActiveChildren )
53 {
54  int iPower = pHelper->m_pPowerInfo->m_Power - iLevel;
55  int vertInc = 1 << (iPower - 1);
56 
57  CTesselateWinding *pWinding = &g_TWinding;
58 
59  // Starting at the bottom-left, wind clockwise picking up vertices and
60  // generating triangles.
61  int iCurTriVert = 0;
62  for( int iVert=0; iVert < pWinding->m_nVerts; iVert++ )
63  {
64  CVertIndex sideVert = BuildOffsetVertIndex( nodeIndex, pWinding->m_Verts[iVert].m_Index, vertInc );
65 
66  int iVertNode = pWinding->m_Verts[iVert].m_iNode;
67  bool bNode = (iVertNode != -1) && pActiveChildren[iVertNode];
68  if( bNode )
69  {
70  if( iCurTriVert == 2 )
71  InternalEndTriangle( pHelper, nodeIndex, iCurTriVert );
72 
73  iCurTriVert = 0;
74  }
75  else
76  {
77  int iVertBit = InternalVertIndex( pHelper->m_pPowerInfo, sideVert );
78  if( pHelper->m_pActiveVerts[iVertBit>>5] & (1 << (iVertBit & 31)) )
79  {
80  // Ok, add a vert here.
81  pHelper->m_TempIndices[iCurTriVert] = (unsigned short)InternalVertIndex( pHelper->m_pPowerInfo, sideVert );
82  iCurTriVert++;
83  if( iCurTriVert == 2 )
84  InternalEndTriangle( pHelper, nodeIndex, iCurTriVert );
85  }
86  }
87  }
88 }

92 // Tesselates in a *breadth first* fashion
93 //-----------------------------------------------------------------------------
94 template< class T >
95 inline void TesselateDisplacement_R(
96  T *pHelper,
97  const CVertIndex &nodeIndex,
98  int iNodeBitIndex,
99  int iLevel
100  )
101 {
102  // Here's the node info for our current node
103  Assert( iNodeBitIndex < pHelper->m_pPowerInfo->m_NodeCount );
104  DispNodeInfo_t& nodeInfo = pHelper->GetNodeInfo( iNodeBitIndex );
105 
106  // Store off the current number of indices
107  int oldIndexCount = pHelper->m_nIndices;
108 
109  // Go through each quadrant. If there is an active child node, recurse down.
110  int bActiveChildren[4];
111  if( iLevel >= pHelper->m_pPowerInfo->m_Power - 1 )
112  {
113  // This node has no children.
114  bActiveChildren[0] = bActiveChildren[1] = bActiveChildren[2] = bActiveChildren[3] = false;
115  }
116  else
117  {
118  int iNodeIndex = InternalVertIndex( pHelper->m_pPowerInfo, nodeIndex );
119 
120  int iChildNodeBit = iNodeBitIndex + 1;
121  for( int iChild=0; iChild < 4; iChild++ )
122  {
123  CVertIndex const &childNode = pHelper->m_pPowerInfo->m_pChildVerts[iNodeIndex].m_Verts[iChild];
124 
125  // Make sure we really can tesselate here (a smaller neighbor displacement could
126  // have inactivated certain edge verts.
127  int iVertBit = InternalVertIndex( pHelper->m_pPowerInfo, childNode );
128  bActiveChildren[iChild] = ( pHelper->m_pActiveVerts[iVertBit>>5] & (1 << (iVertBit & 31)) );
129 
130  if( bActiveChildren[iChild] )
131  {
132  TesselateDisplacement_R( pHelper, childNode, iChildNodeBit, iLevel+1 );
133  }
134  else
135  {
136  // Make sure the triangle counts are cleared on this one because it may visit this
137  // node in GenerateDecalFragments_R if nodeInfo's CHILDREN_HAVE_TRIANGLES flag is set.
138  DispNodeInfo_t &childInfo = pHelper->GetNodeInfo( iChildNodeBit );
139  childInfo.m_Count = 0;
140  childInfo.m_Flags = 0;
141  }
142 
143  iChildNodeBit += pHelper->m_pPowerInfo->m_NodeIndexIncrements[iLevel];
144  }
145  }
146 
147  // Set the child field
148  if ( pHelper->m_nIndices != oldIndexCount )
149  {
150  nodeInfo.m_Flags = DispNodeInfo_t::CHILDREN_HAVE_TRIANGLES;
151  oldIndexCount = pHelper->m_nIndices;
152  }
153  else
154  {
155  nodeInfo.m_Flags = 0;
156  }
157 
158  // Now tesselate the node itself...
159  TesselateDisplacementNode( pHelper, nodeIndex, iLevel, bActiveChildren );
160 
161  // Now that we've tesselated, figure out how many indices we've added at this node
162  nodeInfo.m_Count = pHelper->m_nIndices - oldIndexCount;
163  nodeInfo.m_FirstTesselationIndex = oldIndexCount;
164  Assert( nodeInfo.m_Count % 3 == 0 );
165 }
