69 FORCEINLINE void *CMemoryStack::Alloc( unsigned bytes, bool bClear ) RESTRICT
70 {
71  Assert( m_pBase );
72 
73  int alignment = m_alignment;
74  if ( bytes )
75  {
76  bytes = AlignValue( bytes, alignment );
77  }
78  else
79  {
80  bytes = alignment;
81  }
82 
83 
84  void *pResult = m_pNextAlloc;
85  byte *pNextAlloc = m_pNextAlloc + bytes;
86 
87  if ( pNextAlloc > m_pCommitLimit )
88  {
89  if ( !CommitTo( pNextAlloc ) )
90  {
91  return NULL;
92  }
93  }
94 
95  if ( bClear )
96  {
97  memset( pResult, 0, bytes );
98  }
99 
100  m_pNextAlloc = pNextAlloc;
101 
102  return pResult;
103 }
