252 // One compressed normals+tangents into a SHORT2 each (8 bytes total)

260 inline int GetVertexElementSize( VertexElement_t element, VertexCompressionType_t compressionType )
261 {
262  Detect_VertexElement_t_Changes( element );
263 
264  if ( compressionType == VERTEX_COMPRESSION_ON )
265  {
266  // Compressed-vertex element sizes
267  switch ( element )
268  {
269 #if ( COMPRESSED_NORMALS_TYPE == COMPRESSED_NORMALS_SEPARATETANGENTS_SHORT2 )
270  case VERTEX_ELEMENT_NORMAL:
271  return ( 2 * sizeof( short ) );
272  case VERTEX_ELEMENT_USERDATA4:
273  return ( 2 * sizeof( short ) );
274 #else //( COMPRESSED_NORMALS_TYPE == COMPRESSED_NORMALS_COMBINEDTANGENTS_UBYTE4 )
275  // Normals and tangents (userdata4) are combined into a single UBYTE4 vertex element
276  case VERTEX_ELEMENT_NORMAL:
277  return ( 4 * sizeof( unsigned char ) );
278  case VERTEX_ELEMENT_USERDATA4:
279  return ( 0 );
280 #endif
281  // Compressed bone weights use a SHORT2 vertex element:
282  case VERTEX_ELEMENT_BONEWEIGHTS1:
283  case VERTEX_ELEMENT_BONEWEIGHTS2:
284  return ( 2 * sizeof( short ) );
285  default:
286  break;
287  }
288  }
289 
290  // Uncompressed-vertex element sizes
291  switch ( element )
292  {
293  case VERTEX_ELEMENT_POSITION: return ( 3 * sizeof( float ) );
294  case VERTEX_ELEMENT_NORMAL: return ( 3 * sizeof( float ) );
295  case VERTEX_ELEMENT_COLOR: return ( 4 * sizeof( unsigned char ) );
296  case VERTEX_ELEMENT_SPECULAR: return ( 4 * sizeof( unsigned char ) );
297  case VERTEX_ELEMENT_TANGENT_S: return ( 3 * sizeof( float ) );
298  case VERTEX_ELEMENT_TANGENT_T: return ( 3 * sizeof( float ) );
299  case VERTEX_ELEMENT_WRINKLE: return ( 1 * sizeof( float ) ); // Packed into Position.W
300  case VERTEX_ELEMENT_BONEINDEX: return ( 4 * sizeof( unsigned char ) );
301  case VERTEX_ELEMENT_BONEWEIGHTS1: return ( 1 * sizeof( float ) );
302  case VERTEX_ELEMENT_BONEWEIGHTS2: return ( 2 * sizeof( float ) );
303  case VERTEX_ELEMENT_BONEWEIGHTS3: return ( 3 * sizeof( float ) );
304  case VERTEX_ELEMENT_BONEWEIGHTS4: return ( 4 * sizeof( float ) );
305  case VERTEX_ELEMENT_USERDATA1: return ( 1 * sizeof( float ) );
306  case VERTEX_ELEMENT_USERDATA2: return ( 2 * sizeof( float ) );
307  case VERTEX_ELEMENT_USERDATA3: return ( 3 * sizeof( float ) );
308  case VERTEX_ELEMENT_USERDATA4: return ( 4 * sizeof( float ) );
309  case VERTEX_ELEMENT_TEXCOORD1D_0: return ( 1 * sizeof( float ) );
310  case VERTEX_ELEMENT_TEXCOORD1D_1: return ( 1 * sizeof( float ) );
311  case VERTEX_ELEMENT_TEXCOORD1D_2: return ( 1 * sizeof( float ) );
312  case VERTEX_ELEMENT_TEXCOORD1D_3: return ( 1 * sizeof( float ) );
