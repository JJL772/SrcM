127 inline SideType VPlane::GetPointSide(const Vector &vPoint, vec_t sideEpsilon) const
128 {
129  vec_t fDist;
130 
131  fDist = DistTo(vPoint);
132  if(fDist >= sideEpsilon)
133  return SIDE_FRONT;

162  { vMax.x, vMax.y, vMin.z },
163  };
164 
165  firstSide = GetPointSideExact(vPoints[0]);
166  for(i=1; i < 8; i++)
167  {
168  side = GetPointSideExact(vPoints[i]);
169 
170  // Does the box cross the plane?
171  if(side != firstSide)
172  return SIDE_ON;
173  }
174 
175  // Ok, they're all on the same side, return that.
176  return firstSide;
177 }
