252 inline unsigned short CDataManagerBase::FromHandle( memhandle_t handle )
253 {
254  unsigned int fullWord = (unsigned int)handle;
255  unsigned short serial = fullWord>>16;
256  unsigned short index = fullWord & 0xFFFF;
257  index--;
258  if ( m_memoryLists.IsValidIndex(index) && m_memoryLists[index].serial == serial )
259  return index;
260  return m_memoryLists.InvalidIndex();
261 }

263 inline int CDataManagerBase::LockCount( memhandle_t handle )
264 {
265  Lock();
266  int result = 0;
267  unsigned short memoryIndex = FromHandle(handle);
268  if ( memoryIndex != m_memoryLists.InvalidIndex() )
269  {
270  result = m_memoryLists[memoryIndex].lockCount;
271  }
272  Unlock();
273  return result;
