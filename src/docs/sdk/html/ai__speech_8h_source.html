29 class CAI_TimedSemaphore
30 {
31 public:
32  CAI_TimedSemaphore()
33  : m_ReleaseTime( 0 )
34  {
35  m_hCurrentTalker = NULL;
36  }
37 
38  void Acquire( float time, CBaseEntity *pTalker ) { m_ReleaseTime = gpGlobals->curtime + time; m_hCurrentTalker = pTalker; }
39  void Release() { m_ReleaseTime = 0; m_hCurrentTalker = NULL; }
40 
41  // Current owner of the semaphore is always allowed to talk
42  bool IsAvailable( CBaseEntity *pTalker ) const { return ((gpGlobals->curtime > m_ReleaseTime) || (m_hCurrentTalker == pTalker)); }
43  float GetReleaseTime() const { return m_ReleaseTime; }
44 
45  CBaseEntity *GetOwner() { return m_hCurrentTalker; }
46 
47 private:
48  float m_ReleaseTime;
49  EHANDLE m_hCurrentTalker;
50 };
