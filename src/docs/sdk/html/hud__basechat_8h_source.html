353  virtual void OnKeyCodeTyped(vgui::KeyCode code)
354  {
355  if ( code == KEY_ENTER || code == KEY_PAD_ENTER || code == KEY_ESCAPE )
356  {
357  if ( code != KEY_ESCAPE )
358  {
359  if ( m_pHudChat )
360  {
361  PostMessage( m_pHudChat, new KeyValues("ChatEntrySend") );
362  }
363  }
364 
365  // End message mode.
366  if ( m_pHudChat )
367  {
368  PostMessage( m_pHudChat, new KeyValues("ChatEntryStopMessageMode") );
369  }
370  }
371  else if ( code == KEY_TAB )
372  {
373  // Ignore tab, otherwise vgui will screw up the focus.
374  return;
375  }
376  else
377  {
378  BaseClass::OnKeyCodeTyped( code );
379  }
380  }
