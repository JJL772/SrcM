103 float SimpleConstraintSoundProfile::GetVolume(float inVel)
104 {
105  // clamped lerp on 0-1
106  if (inVel <= m_keyPoints[kMIN_THRESHOLD])
107  {
108  return 0;
109  }
110  else if (inVel >= m_keyPoints[kMIN_FULL])
111  {
112  return 1;
113  }
114  else // lerp...
115  {
116  return (inVel - m_keyPoints[kMIN_THRESHOLD])/(m_keyPoints[kMIN_FULL] - m_keyPoints[kMIN_THRESHOLD]);
117  }
118 }

197 Vector VelocitySampler::GetRelativeAngularVelocity( IPhysicsObject *pObj, IPhysicsObject *pReferenceFrame )
198 {
199  Assert(pObj);
200 
201  if ( pReferenceFrame )
202  {
203  Vector childVelocityLocal, parentVelocityLocal, childVelocityWorld, parentVelocityWorld;
204  pObj->GetImplicitVelocity( NULL, &childVelocityLocal );
205  pObj->LocalToWorldVector( &childVelocityWorld, childVelocityLocal );
206  pReferenceFrame->GetImplicitVelocity( NULL, &parentVelocityLocal );
207  pObj->LocalToWorldVector( &parentVelocityWorld, parentVelocityLocal );
208 
209  return (childVelocityWorld - parentVelocityWorld);
210  }
211  else
212  {
213  Vector childVelocityLocal, childVelocityWorld;
214  pObj->GetImplicitVelocity( NULL, &childVelocityLocal );
215  pObj->LocalToWorldVector( &childVelocityWorld, childVelocityLocal );
216 
217  return (childVelocityWorld);
218  }
