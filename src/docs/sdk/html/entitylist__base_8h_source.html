118 inline int CBaseEntityList::GetEntInfoIndex( const CEntInfo *pEntInfo ) const

126 inline CBaseHandle CBaseEntityList::GetNetworkableHandle( int iEntity ) const
127 {
128  Assert( iEntity >= 0 && iEntity < MAX_EDICTS );
129  if ( m_EntPtrArray[iEntity].m_pEntity )

136 inline IHandleEntity* CBaseEntityList::LookupEntity( const CBaseHandle &handle ) const
137 {
138  if ( handle.m_Index == INVALID_EHANDLE_INDEX )
139  return NULL;
140 
141  const CEntInfo *pInfo = &m_EntPtrArray[ handle.GetEntryIndex() ];
142  if ( pInfo->m_SerialNumber == handle.GetSerialNumber() )
143  return (IHandleEntity*)pInfo->m_pEntity;
144  else
145  return NULL;
146 }

160 inline CBaseHandle CBaseEntityList::FirstHandle() const
161 {
162  if ( !m_activeList.Head() )
163  return INVALID_EHANDLE_INDEX;
164

169 inline CBaseHandle CBaseEntityList::NextHandle( CBaseHandle hEnt ) const
170 {
171  int iSlot = hEnt.GetEntryIndex();
172  CEntInfo *pNext = m_EntPtrArray[iSlot].m_pNext;
173  if ( !pNext )
174  return INVALID_EHANDLE_INDEX;
175 
176  int index = GetEntInfoIndex( pNext );
177 
178  return CBaseHandle( index, m_EntPtrArray[index].m_SerialNumber );
