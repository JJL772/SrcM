116  {
117  CBaseEntity *pEnt = gEntList.GetBaseEntity( pHandleEntity->GetRefEHandle() );
118 
119  if ( pEnt == NULL )
120  return ITERATION_CONTINUE;
121 
122  if ( ( pEnt->ObjectCaps() & FCAP_IMPULSE_USE ) == 0 )
123  {
124  return ITERATION_CONTINUE;
125  }
126 
127  if ( FClassnameIs( pEnt, "func_door*" ) )
128  {
129  CBaseDoor *door = dynamic_cast<CBaseDoor *>(pEnt);
130  if ( !door )
131  {
132  return ITERATION_CONTINUE;
133  }
134 
135  if ( door->m_toggle_state == TS_GOING_UP || door->m_toggle_state == TS_GOING_DOWN )
136  {
137  return ITERATION_CONTINUE;
138  }
139  }
140  else if ( FClassnameIs( pEnt, "prop_door*" ) )
141  {
142  CBasePropDoor *door = dynamic_cast<CBasePropDoor *>(pEnt);
143  if ( !door )
144  {
145  return ITERATION_CONTINUE;
146  }
147 
148  if ( door->IsDoorOpening() || door->IsDoorClosing() )
149  {
150  return ITERATION_CONTINUE;
151  }
152  }
153  else
154  {
155  return ITERATION_CONTINUE;
156  }
157 
158  if ( m_nAlreadyHit < m_nMaxHits )
159  {
160  m_AlreadyHit[m_nAlreadyHit] = pEnt;
161  m_nAlreadyHit++;
162  }
163 
164  return ITERATION_CONTINUE;
165  }
