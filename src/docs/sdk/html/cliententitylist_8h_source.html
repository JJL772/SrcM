55  {
56  pEntity->m_pNext = m_pClassList;

59 
60  void Remove( T *pEntity )
61  {
62  T **pPrev = &m_pClassList;
63  T *pCur = *pPrev;
64  while ( pCur )
65  {
66  if ( pCur == pEntity )
67  {
68  *pPrev = pCur->m_pNext;
69  return;
70  }
71  pPrev = &pCur->m_pNext;
72  pCur = *pPrev;
73  }
74  }
75 
76  static T *m_pClassList;
77 };

85 // This is the IClientEntityList implemenation. It serves two functions:
86 //
87 // 1. It converts server entity indices into IClientNetworkables for the engine.
88 //
89 // 2. It provides a place to store IClientUnknowns and gives out ClientEntityHandle_t's
90 // so they can be indexed and retreived. For example, this is how static props are referenced
91 // by the spatial partition manager - it doesn't know what is being inserted, so it's
92 // given ClientEntityHandle_t's, and the handlers for spatial partition callbacks can
