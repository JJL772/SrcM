55 class CRallyPoint : public CPointEntity
56 {
57  DECLARE_CLASS( CRallyPoint, CPointEntity );
58 
59 public:
60  CRallyPoint()
61  {
62  m_hLockedBy.Set(NULL);
63  m_sExclusivity = RALLY_EXCLUSIVE_NOT_EVALUATED;
64  }
65 
66  bool Lock( CBaseEntity *pLocker )
67  {
68  if( IsLocked() )
69  {
70  // Already locked.
71  return false;
72  }
73 
74  m_hLockedBy.Set( pLocker );
75  return true;
76  }
77 
78  bool Unlock( CBaseEntity *pUnlocker )
79  {
80  if( IsLocked() )
81  {
82  if( m_hLockedBy.Get() != pUnlocker )
83  {
84  // Refuse! Only the locker may unlock.
85  return false;
86  }
87  }
88 
89  m_hLockedBy.Set( NULL );
90  return true;
91  }
92 
93  bool IsLocked( void ) { return (m_hLockedBy.Get() != NULL); }
94 
95  int DrawDebugTextOverlays();
96  bool IsExclusive();
97

100  RALLY_EXCLUSIVE_NOT_EVALUATED = -1,
101  RALLY_EXCLUSIVE_NO,
102  RALLY_EXCLUSIVE_YES,
103  };
