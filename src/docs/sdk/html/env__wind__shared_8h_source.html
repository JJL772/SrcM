69 //-----------------------------------------------------------------------------
70 // Adds an event to the queue, will pop off stale events from the queue
71 //-----------------------------------------------------------------------------
72 template <class T, class I>
73 I CTimedEventQueue<T,I>::PushEvent( float flTime, const T &data )
74 {
75  Assert( m_flQueueHeadTime <= flTime );
76  m_flQueueHeadTime = flTime;
77 
78  // First push the event...
79  I idx = m_Queue.AddToHead();
80  m_Queue[idx].m_flTime = flTime;
81  m_Queue[idx].m_Data = data;
82 
83  // Then retire stale events...
84  I i = m_Queue.Tail();
85  while (m_Queue[i].m_flTime < m_flQueueHeadTime - m_flMaxTime )
86  {
87  I prev = m_Queue.Prev(i);
88  Assert( prev != m_Queue.InvalidIndex() );
89  m_Queue.Remove(i);
90  i = prev;
91  }
92 
93  return idx;
94 }

97 //-----------------------------------------------------------------------------
98 // Grabs the last event that happened before or at the specified time
99 //-----------------------------------------------------------------------------
100 template <class T, class I>
101 I CTimedEventQueue<T,I>::GetEventIndex( float flTime ) const
102 {
103  // This checks for a request that fell off the queue
104  Assert( (flTime >= m_flQueueHeadTime - m_flMaxTime) && (flTime <= m_flQueueHeadTime) );
105 
106  // Then retire stale events...
107  I i = m_Queue.Head();
108  while( m_Queue[i].m_flTime > flTime )
109  {
110  i = m_Queue.Next(i);
111  Assert( i != m_Queue.InvalidIndex() );
112  }
113 
114  return i;
115 }
