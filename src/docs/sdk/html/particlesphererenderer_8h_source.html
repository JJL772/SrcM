90 inline void CParticleSphereRenderer::AddLightColor(
91  Vector const *pPos,
92  Vector const *pLightPos,
93  Vector const *pLightColor,
94  float flLightIntensity,
95  Vector *pOutColor )
96 {
97  if( flLightIntensity )
98  {
99  float fDist = pPos->DistToSqr( *pLightPos );
100  float fAmt;
101  if( fDist > 0.0001f )
102  fAmt = flLightIntensity / fDist;
103  else
104  fAmt = 1000.f;
105 
106  *pOutColor += *pLightColor * fAmt;
107  }
108 }

155 inline void CParticleSphereRenderer::RenderParticle(
156  ParticleDraw *pDraw,
157  const Vector &vOriginalPos,
158  const Vector &vTransformedPos,
159  float flAlpha,
160  float flParticleSize,
161  float flAngle )
162 {
163  // Make sure they called StartRender on us so we were able to set the directional light parameters.
164 #ifdef _DEBUG
165  if ( pDraw->GetMeshBuilder() )
166  {
167  Assert( m_iLastTickStartRenderCalled == gpGlobals->tickcount );
168  }
169 #endif
170 
171  Vector vColor = m_vBaseColor;
172  AddLightColor( &vOriginalPos, &m_AmbientLight.m_vPos, &m_AmbientLight.m_vColor, m_AmbientLight.m_flIntensity, &vColor );
173 
174  // If the DX8 shader isn't going to handle the directional light color, then add its contribution here.
175  if( !m_bUsingPixelShaders )
176  {
177  AddLightColor( &vOriginalPos, &m_DirectionalLight.m_vPos, &m_DirectionalLight.m_vColor, m_DirectionalLight.m_flIntensity, &vColor );
178  }
179 
180  ClampColor( vColor );
181 
182  RenderParticle_Color255SizeNormalAngle(
183  pDraw,
184  vTransformedPos,
185  vColor, // ambient color
186  flAlpha, // alpha
187  flParticleSize,
188  vec3_origin,
189  flAngle );
190 }
