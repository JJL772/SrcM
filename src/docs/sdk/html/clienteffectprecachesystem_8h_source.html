88 //-----------------------------------------------------------------------------

90  inline void ReferenceMaterial( const char *materialName, bool increment = true )
91  {
92  IMaterial *material = materials->FindMaterial( materialName, TEXTURE_GROUP_CLIENT_EFFECTS );
93  if ( !IsErrorMaterial( material ) )
94  {
95  if ( increment )
96  {
97  material->IncrementReferenceCount();
98  }
99  else
100  {
101  material->DecrementReferenceCount();
102  }
103  }
104  }
