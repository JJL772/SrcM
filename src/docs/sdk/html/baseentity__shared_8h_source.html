199 inline void CBaseEntity::RemoveEffects( int nEffects )
200 {
201 #if !defined( CLIENT_DLL )
202 #ifdef HL2_EPISODIC
203  if ( nEffects & (EF_BRIGHTLIGHT|EF_DIMLIGHT) )
204  {
205  // Hack for now, to avoid player emitting radius with his flashlight
206  if ( !IsPlayer() )
207  {
208  RemoveEntityFromDarknessCheck( this );
209  }
210  }
211 #endif // HL2_EPISODIC
212 #endif // !CLIENT_DLL
213 
214  m_fEffects &= ~nEffects;
215  if ( nEffects & EF_NODRAW )
216  {
217 #ifndef CLIENT_DLL
218  NetworkProp()->MarkPVSInformationDirty();
219  DispatchUpdateTransmitState();
220 #else
221  UpdateVisibility();
222 #endif
223  }
224 }
