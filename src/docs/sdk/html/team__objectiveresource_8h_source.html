129  void ResetHillData( int team )
130  {
131  if ( team < TEAM_TRAIN_MAX_TEAMS )
132  {
133  m_nNumNodeHillData.Set( team, 0 );
134 
135  int nNumEntriesPerTeam = TEAM_TRAIN_MAX_HILLS * TEAM_TRAIN_FLOATS_PER_HILL;
136  int iStartingIndex = team * nNumEntriesPerTeam;
137  for ( int i = 0 ; i < nNumEntriesPerTeam ; i++ )
138  {
139  m_flNodeHillData.Set( iStartingIndex + i, 0 );
140  }
141 
142  iStartingIndex = team * TEAM_TRAIN_MAX_HILLS;
143  for ( int i = 0; i < TEAM_TRAIN_MAX_HILLS; i++ )
144  {
145  m_bHillIsDownhill.Set( iStartingIndex + i, 0 );
146  }
147  }
148  }
149 
150  void SetHillData( int team, float flStart, float flEnd, bool bDownhill )
151  {
152  if ( team < TEAM_TRAIN_MAX_TEAMS )
153  {
154  int index = ( m_nNumNodeHillData[team] * TEAM_TRAIN_FLOATS_PER_HILL ) + ( team * TEAM_TRAIN_MAX_HILLS * TEAM_TRAIN_FLOATS_PER_HILL );
155  if ( index < TEAM_TRAIN_HILLS_ARRAY_SIZE - 1 ) // - 1 because we want to add 2 entries
156  {
157  m_flNodeHillData.Set( index, flStart );
158  m_flNodeHillData.Set( index + 1, flEnd );
159 
160  if ( m_nNumNodeHillData[team] < TEAM_TRAIN_MAX_HILLS )
161  {
162  m_bHillIsDownhill.Set( m_nNumNodeHillData[team] + ( team * TEAM_TRAIN_MAX_HILLS ), bDownhill );
163  }
164 
165  m_nNumNodeHillData.Set( team, m_nNumNodeHillData[team] + 1);
166  }
167  }
