22 class CAI_ProxTester
23 {
24 public:
25  CAI_ProxTester()
26  : m_distSq( 0 ),
27  m_fInside( false )
28  {
29  }
30 
31  void Init( float dist )
32  {
33  m_fInside = ( dist > 0 );
34  m_distSq = dist * dist;
35  }
36 
37  bool Check( CBaseEntity *pEntity1, CBaseEntity *pEntity2 )
38  {
39  if ( m_distSq != 0 )
40  {
41  float distSq = ( pEntity1->GetAbsOrigin() - pEntity2->GetAbsOrigin() ).LengthSqr();
42  bool fInside = ( distSq < m_distSq );
43 
44  return ( m_fInside == fInside );
45  }
46  return true;
47  }
48 
49  DECLARE_SIMPLE_DATADESC();
50 
51 private:
52 
53  float m_distSq;
54  bool m_fInside;
55 };
