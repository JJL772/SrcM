31 class CEntityClassList : public CBaseEntityClassList
32 {
33 public:
34  virtual void LevelShutdownPostEntity() { m_pClassList = NULL; }
35 
36  void Insert( T *pEntity )
37  {
38  pEntity->m_pNext = m_pClassList;
39  m_pClassList = pEntity;
40  }
41 
42  void Remove( T *pEntity )
43  {
44  T **pPrev = &m_pClassList;
45  T *pCur = *pPrev;
46  while ( pCur )
47  {
48  if ( pCur == pEntity )
49  {
50  *pPrev = pCur->m_pNext;
51  return;
52  }
53  pPrev = &pCur->m_pNext;
54  pCur = *pPrev;
55  }
56  }
57 
58  static T *m_pClassList;
59 };
