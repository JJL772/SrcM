285  template <typename T>
286  int ReadSimple( T *pValue, int nElems, int nBytesAvailable ) // must be inline in class to keep MSVS happy
287  {
288  int desired = nElems * sizeof(T);
289  int actual;
290 
291  if ( nBytesAvailable == 0 )
292  actual = desired;
293  else
294  {
295  Assert( nBytesAvailable % sizeof(T) == 0 );
296  actual = MIN( desired, nBytesAvailable );
297  }
298 
299  BufferReadBytes( (char *)pValue, actual );
300 
301  if ( actual < nBytesAvailable )
302  BufferSkipBytes( nBytesAvailable - actual );
303 
304  return ( actual / sizeof(T) );
305  }
