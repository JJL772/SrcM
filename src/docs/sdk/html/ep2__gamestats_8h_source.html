118  void AppendToBuffer( CUtlBuffer &SaveBuffer )
119  {
120  // Always write out as current version
121  m_Header.m_iVersion = EP2_GAMESTATS_CURRENT_VERSION;
122 
123  // Write out the lumps.
124  CBaseGameStats::AppendLump( EP2_MAX_LUMP_COUNT, SaveBuffer, EP2STATS_LUMP_HEADER, 1, sizeof( Ep2LevelStats_t::LevelHeader_t ), static_cast<void*>( &m_Header ) );
125  CBaseGameStats::AppendLump( EP2_MAX_LUMP_COUNT, SaveBuffer, EP2STATS_LUMP_TAG, 1, sizeof( Ep2LevelStats_t::Tag_t ), static_cast< void * >( &m_Tag ) );
126  CBaseGameStats::AppendLump( EP2_MAX_LUMP_COUNT, SaveBuffer, EP2STATS_LUMP_DEATH, m_aPlayerDeaths.Count(), sizeof( Ep2LevelStats_t::PlayerDeathsLump_t ), static_cast<void*>( m_aPlayerDeaths.Base() ) );
127  {
128  CUtlBuffer buf;
129  buf.Put( (const void *)m_IntCounters, sizeof( m_IntCounters ) );
130  buf.Put( (const void *)m_FloatCounters, sizeof( m_FloatCounters ) );
131  buf.PutInt( m_dictEntityDeaths.Count() );
132  for ( int i = m_dictEntityDeaths.First(); i != m_dictEntityDeaths.InvalidIndex(); i = m_dictEntityDeaths.Next( i ) )
133  {
134  buf.PutString( m_dictEntityDeaths.GetElementName( i ) );
135  buf.Put( (const void *)&m_dictEntityDeaths[ i ], sizeof( Ep2LevelStats_t::EntityDeathsLump_t ) );
136  }
137  CBaseGameStats::AppendLump( EP2_MAX_LUMP_COUNT, SaveBuffer, EP2STATS_LUMP_NPC, 1, buf.TellPut(), buf.Base() );
138  }
139  {
140  CUtlBuffer buf;
141  buf.PutInt( m_dictWeapons.Count() );
142  for ( int i = m_dictWeapons.First(); i != m_dictWeapons.InvalidIndex(); i = m_dictWeapons.Next( i ) )
143  {
144  buf.PutString( m_dictWeapons.GetElementName( i ) );
145  buf.Put( (const void *)&m_dictWeapons[ i ], sizeof( Ep2LevelStats_t::WeaponLump_t ) );
146  }
147  CBaseGameStats::AppendLump( EP2_MAX_LUMP_COUNT, SaveBuffer, EP2STATS_LUMP_WEAPON, 1, buf.TellPut(), buf.Base() );
148  }
149  {
150  CUtlBuffer buf;
151  buf.PutString( m_SaveGameInfo.m_sCurrentSaveFile.String() );
152  buf.PutInt( m_SaveGameInfo.m_nCurrentSaveFileTime );
153  buf.PutInt( m_SaveGameInfo.m_Records.Count() );
154  for ( int i = 0 ; i < m_SaveGameInfo.m_Records.Count(); ++i )
155  {
156  buf.Put( (const void *)&m_SaveGameInfo.m_Records[ i ], sizeof( Ep2LevelStats_t::SaveGameInfoRecord2_t ) );
157  }
158  CBaseGameStats::AppendLump( EP2_MAX_LUMP_COUNT, SaveBuffer, EP2STATS_LUMP_SAVEGAMEINFO, 1, buf.TellPut(), buf.Base() );
159  }
160  {
161  CUtlBuffer buf;
162  buf.PutShort( Ep2LevelStats_t::GenericStatsLump_t::LumpVersion );
163  buf.PutInt( m_dictGeneric.Count() );
164  for ( int i = m_dictGeneric.First(); i != m_dictGeneric.InvalidIndex(); i = m_dictGeneric.Next( i ) )
165  {
166  buf.PutString( m_dictGeneric.GetElementName( i ) );
167  buf.Put( (const void *)&m_dictGeneric[ i ], sizeof( Ep2LevelStats_t::GenericStatsLump_t ) );
168  }
169  CBaseGameStats::AppendLump( EP2_MAX_LUMP_COUNT, SaveBuffer, EP2STATS_LUMP_GENERIC, 1, buf.TellPut(), buf.Base() );
170  }

173  static void LoadData( CUtlDict<Ep2LevelStats_t, unsigned short>& items, CUtlBuffer &LoadBuffer )
174  {
175  // Read the next lump.
176  unsigned short iLump = 0;
177  unsigned short iLumpCount = 0;
178 
179  Ep2LevelStats_t *pItem = NULL;
180 
181  while( CBaseGameStats::GetLumpHeader( EP2_MAX_LUMP_COUNT, LoadBuffer, iLump, iLumpCount, true ) )
182  {
183  switch ( iLump )
184  {
185  case EP2STATS_LUMP_HEADER:
186  {
187  Ep2LevelStats_t::LevelHeader_t header;
188  CBaseGameStats::LoadLump( LoadBuffer, iLumpCount, sizeof( Ep2LevelStats_t::LevelHeader_t ), &header );
189  pItem = &items[ items.Insert( header.m_szMapName ) ];
190  pItem->m_Header = header;
191  pItem->m_Tag.Clear();
192  Assert( pItem );
193  }
194  break;
195  case EP2STATS_LUMP_TAG:
196  {
197  Assert( pItem );
198  CBaseGameStats::LoadLump( LoadBuffer, iLumpCount, sizeof( Ep2LevelStats_t::Tag_t ), &pItem->m_Tag );
199  }
200  break;
201  case EP2STATS_LUMP_DEATH:
202  {
203  Assert( pItem );
204  pItem->m_aPlayerDeaths.SetCount( iLumpCount );
205  CBaseGameStats::LoadLump( LoadBuffer, iLumpCount, sizeof( Ep2LevelStats_t::PlayerDeathsLump_t ), static_cast<void*>( pItem->m_aPlayerDeaths.Base() ) );
206  }
207  break;
208  case EP2STATS_LUMP_NPC:
209  {
210  Assert( pItem );
211  LoadBuffer.Get( ( void * )pItem->m_IntCounters, sizeof( pItem->m_IntCounters ) );
212  LoadBuffer.Get( ( void * )pItem->m_FloatCounters, sizeof( pItem->m_FloatCounters ) );
213  int c = LoadBuffer.GetInt();
214  for ( int i = 0 ; i < c; ++i )
215  {
216  Ep2LevelStats_t::EntityDeathsLump_t data;
217  char npcName[ 512 ];
218  LoadBuffer.GetString( npcName );
219  LoadBuffer.Get( &data, sizeof( data ) );
220  pItem->m_dictEntityDeaths.Insert( npcName, data );
221  }
222  }
223  break;
224  case EP2STATS_LUMP_WEAPON:
225  {
226  Assert( pItem );
227  int c = LoadBuffer.GetInt();
228  for ( int i = 0 ; i < c; ++i )
229  {
230  Ep2LevelStats_t::WeaponLump_t data;
231  char weaponName[ 512 ];
232  LoadBuffer.GetString( weaponName );
233  LoadBuffer.Get( &data, sizeof( data ) );
234  pItem->m_dictWeapons.Insert( weaponName, data );
235  }
236  }
237  break;
238  case EP2STATS_LUMP_SAVEGAMEINFO:
239  {
240  Assert( pItem );
241  Ep2LevelStats_t::SaveGameInfo_t *info = &pItem->m_SaveGameInfo;
242  char sz[ 512 ];
243  LoadBuffer.GetString( sz );
244  info->m_sCurrentSaveFile = sz;
245  info->m_nCurrentSaveFileTime = LoadBuffer.GetInt();
246  int c = LoadBuffer.GetInt();
247  for ( int i = 0 ; i < c; ++i )
248  {
249  Ep2LevelStats_t::SaveGameInfoRecord2_t rec;
250  if ( pItem->m_Header.m_iVersion >= EP2_GAMESTATS_FILE_VERSION_02 )
251  {
252  LoadBuffer.Get( &rec, sizeof( rec ) );
253  }
254  else
255  {
256  size_t s = sizeof( Ep2LevelStats_t::SaveGameInfoRecord_t );
257  LoadBuffer.Get( &rec, s );
258  }
259  info->m_Records.AddToTail( rec );
260  }
261  info->m_pCurrentRecord = NULL;
262  if ( info->m_Records.Count() > 0 )
263  {
264  info->m_pCurrentRecord = &info->m_Records[ info->m_Records.Count() - 1 ];
265  }
266  }
267  break;
268  case EP2STATS_LUMP_GENERIC:
269  {
270  Assert( pItem );
271  int version = LoadBuffer.GetShort();
272  if ( version == Ep2LevelStats_t::GenericStatsLump_t::LumpVersion )
273  {
274  int c = LoadBuffer.GetInt();
275  Assert( c < 2 * 1024 * 1024 );
276  for ( int i = 0 ; i < c; ++i )
277  {
278  Ep2LevelStats_t::GenericStatsLump_t data;
279  char pchStatName[ 512 ];
280  LoadBuffer.GetString( pchStatName );
281  LoadBuffer.Get( &data, sizeof( data ) );
282  pItem->m_dictGeneric.Insert( pchStatName, data );
283  }
284  }
285  else
286  {
287  Error( "Unsupported GenericStatsLump_t::LumpVersion" );
288  }
289  }
290  break;
291  }
292  }
293  }
