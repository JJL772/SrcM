41 inline void UpdateRefractTexture( int x, int y, int w, int h, bool bForceUpdate = false )
42 {
43  Assert( !DrawingShadowDepthView() );
44 
45  if ( !IsRetail() && !r_updaterefracttexture.GetBool() )
46  return;
47 
48  CMatRenderContextPtr pRenderContext( materials );
49  ITexture *pTexture = GetPowerOfTwoFrameBufferTexture();
50  if ( IsPC() || bForceUpdate || g_bAllowMultipleRefractUpdatesPerScenePerFrame || (gpGlobals->framecount != g_viewscene_refractUpdateFrame) )
51  {
52  // forced or only once per frame
53  Rect_t rect;
54  rect.x = x;
55  rect.y = y;
56  rect.width = w;
57  rect.height = h;
58  pRenderContext->CopyRenderTargetToTextureEx( pTexture, 0, &rect, NULL );
59 
60  g_viewscene_refractUpdateFrame = gpGlobals->framecount;
61  }
62  pRenderContext->SetFrameBufferCopyTexture( pTexture );
63 }

76 inline void UpdateScreenEffectTexture( int textureIndex, int x, int y, int w, int h, bool bDestFullScreen = false, Rect_t *pActualRect = NULL )
77 {
78  Rect_t srcRect;
79  srcRect.x = x;
80  srcRect.y = y;
81  srcRect.width = w;
82  srcRect.height = h;
83 
84  CMatRenderContextPtr pRenderContext( materials );
85  ITexture *pTexture = GetFullFrameFrameBufferTexture( textureIndex );
86  int nSrcWidth, nSrcHeight;
87  pRenderContext->GetRenderTargetDimensions( nSrcWidth, nSrcHeight );
88  int nDestWidth = pTexture->GetActualWidth();
89  int nDestHeight = pTexture->GetActualHeight();
90 
91  Rect_t destRect = srcRect;
92  if( !bDestFullScreen && ( nSrcWidth > nDestWidth || nSrcHeight > nDestHeight ) )
93  {
94  // the source and target sizes aren't necessarily the same (specifically in dx7 where
95  // nonpow2 rendertargets aren't supported), so lets figure it out here.
96  float scaleX = ( float )nDestWidth / ( float )nSrcWidth;
97  float scaleY = ( float )nDestHeight / ( float )nSrcHeight;
98  destRect.x = srcRect.x * scaleX;
99  destRect.y = srcRect.y * scaleY;
100  destRect.width = srcRect.width * scaleX;
101  destRect.height = srcRect.height * scaleY;
102  destRect.x = clamp( destRect.x, 0, nDestWidth );
103  destRect.y = clamp( destRect.y, 0, nDestHeight );
104  destRect.width = clamp( destRect.width, 0, nDestWidth - destRect.x );
105  destRect.height = clamp( destRect.height, 0, nDestHeight - destRect.y );
106  }
107 
108  pRenderContext->CopyRenderTargetToTextureEx( pTexture, 0, &srcRect, bDestFullScreen ? NULL : &destRect );
109  pRenderContext->SetFrameBufferCopyTexture( pTexture, textureIndex );
110 
111  if ( pActualRect )
112  {
113  pActualRect->x = destRect.x;
114  pActualRect->y = destRect.y;
115  pActualRect->width = destRect.width;
116  pActualRect->height = destRect.height;
117  }
118 }
