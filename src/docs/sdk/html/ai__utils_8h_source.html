48 class CAI_MoveMonitor
49 {
50 public:
51  CAI_MoveMonitor()
52  : m_vMark( 0, 0, 0 ),
53  m_flMarkTolerance( NO_MARK )
54  {
55  }
56 
57  void SetMark( CBaseEntity *pEntity, float tolerance )
58  {
59  if ( pEntity )
60  {
61  m_vMark = pEntity->GetAbsOrigin();
62  m_flMarkTolerance = tolerance;
63  }
64  }
65 
66  void ClearMark()
67  {
68  m_flMarkTolerance = NO_MARK;
69  }
70 
71  bool IsMarkSet()
72  {
73  return ( m_flMarkTolerance != NO_MARK );
74  }
75 
76  bool TargetMoved( CBaseEntity *pEntity )
77  {
78  if ( IsMarkSet() && pEntity != NULL )
79  {
80  float distance = ( m_vMark - pEntity->GetAbsOrigin() ).Length();
81  if ( distance > m_flMarkTolerance )
82  return true;
83  }
84  return false;
85  }
86 
87  bool TargetMoved2D( CBaseEntity *pEntity )
88  {
89  if ( IsMarkSet() && pEntity != NULL )
90  {
91  float distance = ( m_vMark.AsVector2D() - pEntity->GetAbsOrigin().AsVector2D() ).Length();
92  if ( distance > m_flMarkTolerance )
93  return true;
94  }
95  return false;
96  }
97 
98  Vector GetMarkPos() { return m_vMark; }
99 
100 private:
101  enum
102  {
103  NO_MARK = -1
104  };
105 
106  Vector m_vMark;
107  float m_flMarkTolerance;
108 
109  DECLARE_SIMPLE_DATADESC();
110 };
