41 class ShortestPathCost
42 {
43 public:
44  float operator() ( CNavArea *area, CNavArea *fromArea, const CNavLadder *ladder, const CFuncElevator *elevator, float length )
45  {
46  if ( fromArea == NULL )
47  {
48  // first area in path, no cost
49  return 0.0f;
50  }
51  else
52  {
53  // compute distance traveled along path so far
54  float dist;
55 
56  if ( ladder )
57  {
58  dist = ladder->m_length;
59  }
60  else if ( length > 0.0 )
61  {
62  dist = length;
63  }
64  else
65  {
66  dist = ( area->GetCenter() - fromArea->GetCenter() ).Length();
67  }
68 
69  float cost = dist + fromArea->GetCostSoFar();
70 
71  // if this is a "crouch" area, add penalty
72  if ( area->GetAttributes() & NAV_MESH_CROUCH )
73  {
74  const float crouchPenalty = 20.0f; // 10
75  cost += crouchPenalty * dist;
76  }
77 
78  // if this is a "jump" area, add penalty
79  if ( area->GetAttributes() & NAV_MESH_JUMP )
80  {
81  const float jumpPenalty = 5.0f;
82  cost += jumpPenalty * dist;
83  }
84 
85  return cost;
86  }
87  }
88 };

102 #define IGNORE_NAV_BLOCKERS true
103 template< typename CostFunctor >
104 bool NavAreaBuildPath( CNavArea *startArea, CNavArea *goalArea, const Vector *goalPos, CostFunctor &costFunc, CNavArea **closestArea = NULL, float maxPathLength = 0.0f, int teamID = TEAM_ANY, bool ignoreNavBlockers = false )
105 {
106  VPROF_BUDGET( "NavAreaBuildPath", "NextBotSpiky" );
107 
108  if ( closestArea )
109  {
110  *closestArea = startArea;
111  }
112 
113 #ifdef STAGING_ONLY
114  bool isDebug = ( g_DebugPathfindCounter-- > 0 );
115 #endif
116 
117  if (startArea == NULL)
118  return false;
119 
120  startArea->SetParent( NULL );
121 
122  if (goalArea != NULL && goalArea->IsBlocked( teamID, ignoreNavBlockers ))
123  goalArea = NULL;
124 
125  if (goalArea == NULL && goalPos == NULL)
126  return false;
127 
128  // if we are already in the goal area, build trivial path
129  if (startArea == goalArea)
130  {
131  return true;
132  }
133 
134  // determine actual goal position
135  Vector actualGoalPos = (goalPos) ? *goalPos : goalArea->GetCenter();
136 
137  // start search
138  CNavArea::ClearSearchLists();
139 
140  // compute estimate of path length
142  startArea->SetTotalCost( (startArea->GetCenter() - actualGoalPos).Length() );
143 
144  float initCost = costFunc( startArea, NULL, NULL, NULL, -1.0f );
145  if (initCost < 0.0f)
146  return false;
147  startArea->SetCostSoFar( initCost );
148  startArea->SetPathLengthSoFar( 0.0 );
149 
150  startArea->AddToOpenList();
151 
152  // keep track of the area we visit that is closest to the goal
153  float closestAreaDist = startArea->GetTotalCost();
154 
155  // do A* search
156  while( !CNavArea::IsOpenListEmpty() )
157  {
158  // get next area to check
159  CNavArea *area = CNavArea::PopOpenList();
160 
161 #ifdef STAGING_ONLY
162  if ( isDebug )
163  {
164  area->DrawFilled( 0, 255, 0, 128, 30.0f );
165  }
166 #endif
167 
168  // don't consider blocked areas
169  if ( area->IsBlocked( teamID, ignoreNavBlockers ) )
170  continue;
171 
172  // check if we have found the goal area or position
173  if (area == goalArea || (goalArea == NULL && goalPos && area->Contains( *goalPos )))
174  {
175  if (closestArea)
176  {
177  *closestArea = area;
178  }
179 
180  return true;
181  }
182 
183  // search adjacent areas
184  enum SearchType
185  {
186  SEARCH_FLOOR, SEARCH_LADDERS, SEARCH_ELEVATORS
187  };
188  SearchType searchWhere = SEARCH_FLOOR;
189  int searchIndex = 0;
190 
191  int dir = NORTH;
192  const NavConnectVector *floorList = area->GetAdjacentAreas( NORTH );
193 
194  bool ladderUp = true;
195  const NavLadderConnectVector *ladderList = NULL;
196  enum { AHEAD = 0, LEFT, RIGHT, BEHIND, NUM_TOP_DIRECTIONS };
197  int ladderTopDir = AHEAD;
198  bool bHaveMaxPathLength = ( maxPathLength > 0.0f );
199  float length = -1;
200 
201  while( true )
202  {
203  CNavArea *newArea = NULL;
204  NavTraverseType how;
205  const CNavLadder *ladder = NULL;
206  const CFuncElevator *elevator = NULL;
207 
208  //
209  // Get next adjacent area - either on floor or via ladder
210  //
211  if ( searchWhere == SEARCH_FLOOR )
212  {
213  // if exhausted adjacent connections in current direction, begin checking next direction
214  if ( searchIndex >= floorList->Count() )
215  {
216  ++dir;
217 
218  if ( dir == NUM_DIRECTIONS )
219  {
220  // checked all directions on floor - check ladders next
221  searchWhere = SEARCH_LADDERS;
222 
223  ladderList = area->GetLadders( CNavLadder::LADDER_UP );
224  searchIndex = 0;
225  ladderTopDir = AHEAD;
226  }
227  else
228  {
229  // start next direction
230  floorList = area->GetAdjacentAreas( (NavDirType)dir );
231  searchIndex = 0;
232  }
233 
234  continue;
235  }
236 
237  const NavConnect &floorConnect = floorList->Element( searchIndex );
238  newArea = floorConnect.area;
239  length = floorConnect.length;
240  how = (NavTraverseType)dir;
241  ++searchIndex;
242 
243  if ( IsX360() && searchIndex < floorList->Count() )
244  {
245  PREFETCH360( floorList->Element( searchIndex ).area, 0 );
246  }
247  }
248  else if ( searchWhere == SEARCH_LADDERS )
249  {
250  if ( searchIndex >= ladderList->Count() )
251  {
252  if ( !ladderUp )
253  {
254  // checked both ladder directions - check elevators next
255  searchWhere = SEARCH_ELEVATORS;
256  searchIndex = 0;
257  ladder = NULL;
258  }
259  else
260  {
261  // check down ladders
262  ladderUp = false;
263  ladderList = area->GetLadders( CNavLadder::LADDER_DOWN );
264  searchIndex = 0;
265  }
266  continue;
267  }
268 
269  if ( ladderUp )
270  {
271  ladder = ladderList->Element( searchIndex ).ladder;
272 
273  // do not use BEHIND connection, as its very hard to get to when going up a ladder
274  if ( ladderTopDir == AHEAD )
275  {
276  newArea = ladder->m_topForwardArea;
277  }
278  else if ( ladderTopDir == LEFT )
279  {
280  newArea = ladder->m_topLeftArea;
281  }
282  else if ( ladderTopDir == RIGHT )
283  {
284  newArea = ladder->m_topRightArea;
285  }
286  else
287  {
288  ++searchIndex;
289  ladderTopDir = AHEAD;
290  continue;
291  }
292 
293  how = GO_LADDER_UP;
294  ++ladderTopDir;
295  }
296  else
297  {
298  newArea = ladderList->Element( searchIndex ).ladder->m_bottomArea;
299  how = GO_LADDER_DOWN;
300  ladder = ladderList->Element(searchIndex).ladder;
301  ++searchIndex;
302  }
303 
304  if ( newArea == NULL )
305  continue;
306 
307  length = -1.0f;
308  }
309  else // if ( searchWhere == SEARCH_ELEVATORS )
310  {
311  const NavConnectVector &elevatorAreas = area->GetElevatorAreas();
312 
313  elevator = area->GetElevator();
314 
315  if ( elevator == NULL || searchIndex >= elevatorAreas.Count() )
316  {
317  // done searching connected areas
318  elevator = NULL;
319  break;
320  }
321 
322  newArea = elevatorAreas[ searchIndex++ ].area;
323  if ( newArea->GetCenter().z > area->GetCenter().z )
324  {
325  how = GO_ELEVATOR_UP;
326  }
327  else
328  {
329  how = GO_ELEVATOR_DOWN;
330  }
331 
332  length = -1.0f;
333  }
334 
335 
336  // don't backtrack
337  Assert( newArea );
338  if ( newArea == area->GetParent() )
339  continue;
340  if ( newArea == area ) // self neighbor?
341  continue;
342 
343  // don't consider blocked areas
344  if ( newArea->IsBlocked( teamID, ignoreNavBlockers ) )
345  continue;
346 
347  float newCostSoFar = costFunc( newArea, area, ladder, elevator, length );
348 
349  // NaNs really mess this function up causing tough to track down hangs. If
350  // we get inf back, clamp it down to a really high number.
351  DebuggerBreakOnNaN_StagingOnly( newCostSoFar );
352  if ( IS_NAN( newCostSoFar ) )
353  newCostSoFar = 1e30f;
354 
355  // check if cost functor says this area is a dead-end
356  if ( newCostSoFar < 0.0f )
357  continue;
358 
359  // Safety check against a bogus functor. The cost of the path
360  // A...B, C should always be at least as big as the path A...B.
361  Assert( newCostSoFar >= area->GetCostSoFar() );
362 
363  // And now that we've asserted, let's be a bit more defensive.
364  // Make sure that any jump to a new area incurs some pathfinsing
365  // cost, to avoid us spinning our wheels over insignificant cost
366  // benefit, floating point precision bug, or busted cost functor.
367  float minNewCostSoFar = area->GetCostSoFar() * 1.00001f + 0.00001f;
368  newCostSoFar = Max( newCostSoFar, minNewCostSoFar );
369 
370  // stop if path length limit reached
371  if ( bHaveMaxPathLength )
372  {
373  // keep track of path length so far
374  float deltaLength = ( newArea->GetCenter() - area->GetCenter() ).Length();
375  float newLengthSoFar = area->GetPathLengthSoFar() + deltaLength;
376  if ( newLengthSoFar > maxPathLength )
377  continue;
378 
379  newArea->SetPathLengthSoFar( newLengthSoFar );
380  }
381 
382  if ( ( newArea->IsOpen() || newArea->IsClosed() ) && newArea->GetCostSoFar() <= newCostSoFar )
383  {
384  // this is a worse path - skip it
385  continue;
386  }
387  else
388  {
389  // compute estimate of distance left to go
390  float distSq = ( newArea->GetCenter() - actualGoalPos ).LengthSqr();
391  float newCostRemaining = ( distSq > 0.0 ) ? FastSqrt( distSq ) : 0.0 ;
392 
393  // track closest area to goal in case path fails
394  if ( closestArea && newCostRemaining < closestAreaDist )
395  {
396  *closestArea = newArea;
397  closestAreaDist = newCostRemaining;
398  }
399 
400  newArea->SetCostSoFar( newCostSoFar );
401  newArea->SetTotalCost( newCostSoFar + newCostRemaining );
402 
403  if ( newArea->IsClosed() )
404  {
405  newArea->RemoveFromClosedList();
406  }
407 
408  if ( newArea->IsOpen() )
409  {
410  // area already on open list, update the list order to keep costs sorted
411  newArea->UpdateOnOpenList();
412  }
413  else
414  {
415  newArea->AddToOpenList();
416  }
417 
418  newArea->SetParent( area, how );
419  }
420  }
421 
422  // we have searched this area
423  area->AddToClosedList();
424  }
425 
426  return false;
427 }

434 template< typename CostFunctor >
435 float NavAreaTravelDistance( CNavArea *startArea, CNavArea *endArea, CostFunctor &costFunc, float maxPathLength = 0.0f )
436 {
437  if (startArea == NULL)
438  return -1.0f;
439 
440  if (endArea == NULL)
441  return -1.0f;
442 
443  if (startArea == endArea)
444  return 0.0f;
445 
446  // compute path between areas using given cost heuristic
447  if (NavAreaBuildPath( startArea, endArea, NULL, costFunc, NULL, maxPathLength ) == false)
448  return -1.0f;
449 
450  // compute distance along path
451  float distance = 0.0f;
452  for( CNavArea *area = endArea; area->GetParent(); area = area->GetParent() )
453  {
454  distance += (area->GetCenter() - area->GetParent()->GetCenter()).Length();
455  }
456 
457  return distance;
458 }

474 // helper function
475 inline void AddAreaToOpenList( CNavArea *area, CNavArea *parent, const Vector &startPos, float maxRange )
476 {
477  if (area == NULL)
478  return;
479 
480  if (!area->IsMarked())
481  {
482  area->Mark();
483  area->SetTotalCost( 0.0f );
484  area->SetParent( parent );
485 
486  if (maxRange > 0.0f)
487  {
488  // make sure this area overlaps range
489  Vector closePos;
490  area->GetClosestPointOnArea( startPos, &closePos );
491  if ((closePos - startPos).AsVector2D().IsLengthLessThan( maxRange ))
492  {
493  // compute approximate distance along path to limit travel range, too
494  float distAlong = parent->GetCostSoFar();
495  distAlong += (area->GetCenter() - parent->GetCenter()).Length();
496  area->SetCostSoFar( distAlong );
497 
498  // allow for some fudge due to large size areas
499  if (distAlong <= 1.5f * maxRange)
500  area->AddToOpenList();
501  }
502  }
503  else
504  {
505  // infinite range
506  area->AddToOpenList();
507  }
508  }
509 }

512 /****************************************************************
513  * DEPRECATED: Use filter-based SearchSurroundingAreas below
514  ****************************************************************/
515 #define INCLUDE_INCOMING_CONNECTIONS 0x1
516 #define INCLUDE_BLOCKED_AREAS 0x2
517 #define EXCLUDE_OUTGOING_CONNECTIONS 0x4
518 #define EXCLUDE_ELEVATORS 0x8
519 template < typename Functor >
520 void SearchSurroundingAreas( CNavArea *startArea, const Vector &startPos, Functor &func, float maxRange = -1.0f, unsigned int options = 0, int teamID = TEAM_ANY )
521 {
522  if (startArea == NULL)
523  return;
524 
525  CNavArea::MakeNewMarker();
526  CNavArea::ClearSearchLists();
527 
528  startArea->AddToOpenList();
529  startArea->SetTotalCost( 0.0f );
530  startArea->SetCostSoFar( 0.0f );
531  startArea->SetParent( NULL );
532  startArea->Mark();
533 
534  while( !CNavArea::IsOpenListEmpty() )
535  {
536  // get next area to check
537  CNavArea *area = CNavArea::PopOpenList();
538 
539  // don't use blocked areas
540  if ( area->IsBlocked( teamID ) && !(options & INCLUDE_BLOCKED_AREAS) )
541  continue;
542 
543  // invoke functor on area
544  if (func( area ))
545  {
546  // explore adjacent floor areas
547  for( int dir=0; dir<NUM_DIRECTIONS; ++dir )
548  {
549  int count = area->GetAdjacentCount( (NavDirType)dir );
550  for( int i=0; i<count; ++i )
551  {
552  CNavArea *adjArea = area->GetAdjacentArea( (NavDirType)dir, i );
553  if ( options & EXCLUDE_OUTGOING_CONNECTIONS )
554  {
555  if ( !adjArea->IsConnected( area, NUM_DIRECTIONS ) )
556  {
557  continue; // skip this outgoing connection
558  }
559  }
560 
561  AddAreaToOpenList( adjArea, area, startPos, maxRange );
562  }
563  }
564 
565  // potentially include areas that connect TO this area via a one-way link
566  if (options & INCLUDE_INCOMING_CONNECTIONS)
567  {
568  for( int dir=0; dir<NUM_DIRECTIONS; ++dir )
569  {
570  const NavConnectVector *list = area->GetIncomingConnections( (NavDirType)dir );
571 
572  FOR_EACH_VEC( (*list), it )
573  {
574  NavConnect connect = (*list)[ it ];
575 
576  AddAreaToOpenList( connect.area, area, startPos, maxRange );
577  }
578  }
579  }
580 
581 
582  // explore adjacent areas connected by ladders
583 
584  // check up ladders
585  const NavLadderConnectVector *ladderList = area->GetLadders( CNavLadder::LADDER_UP );
586  if (ladderList)
587  {
588  FOR_EACH_VEC( (*ladderList), it )
589  {
590  const CNavLadder *ladder = (*ladderList)[ it ].ladder;
591 
592  // do not use BEHIND connection, as its very hard to get to when going up a ladder
593  AddAreaToOpenList( ladder->m_topForwardArea, area, startPos, maxRange );
594  AddAreaToOpenList( ladder->m_topLeftArea, area, startPos, maxRange );
595  AddAreaToOpenList( ladder->m_topRightArea, area, startPos, maxRange );
596  }
597  }
598 
599  // check down ladders
600  ladderList = area->GetLadders( CNavLadder::LADDER_DOWN );
601  if (ladderList)
602  {
603  FOR_EACH_VEC( (*ladderList), it )
604  {
605  const CNavLadder *ladder = (*ladderList)[ it ].ladder;
606 
607  AddAreaToOpenList( ladder->m_bottomArea, area, startPos, maxRange );
608  }
609  }
610 
611  if ( (options & EXCLUDE_ELEVATORS) == 0 )
612  {
613  const NavConnectVector &elevatorList = area->GetElevatorAreas();
614  FOR_EACH_VEC( elevatorList, it )
615  {
616  CNavArea *elevatorArea = elevatorList[ it ].area;
617  AddAreaToOpenList( elevatorArea, area, startPos, maxRange );
618  }
619  }
620  }
621  }
622 }

629 class ISearchSurroundingAreasFunctor
630 {
631 public:
632  virtual ~ISearchSurroundingAreasFunctor() { }
633 
638  virtual bool operator() ( CNavArea *area, CNavArea *priorArea, float travelDistanceSoFar ) = 0;
639 
640  // return true if 'adjArea' should be included in the ongoing search
641  virtual bool ShouldSearch( CNavArea *adjArea, CNavArea *currentArea, float travelDistanceSoFar )
642  {
643  return !adjArea->IsBlocked( TEAM_ANY );
644  }
645 
649  virtual void IterateAdjacentAreas( CNavArea *area, CNavArea *priorArea, float travelDistanceSoFar )
650  {
651  // search adjacent outgoing connections
652  for( int dir=0; dir<NUM_DIRECTIONS; ++dir )
653  {
654  int count = area->GetAdjacentCount( (NavDirType)dir );
655  for( int i=0; i<count; ++i )
656  {
657  CNavArea *adjArea = area->GetAdjacentArea( (NavDirType)dir, i );
658 
659  if ( ShouldSearch( adjArea, area, travelDistanceSoFar ) )
660  {
661  IncludeInSearch( adjArea, area );
662  }
663  }
664  }
665  }
666 
667  // Invoked after the search has completed
668  virtual void PostSearch( void ) { }
669 
670  // consider 'area' in upcoming search steps
671  void IncludeInSearch( CNavArea *area, CNavArea *priorArea )
672  {
673  if ( area == NULL )
674  return;
675 
676  if ( !area->IsMarked() )
677  {
678  area->Mark();
679  area->SetTotalCost( 0.0f );
680  area->SetParent( priorArea );
681 
682  // compute approximate travel distance from start area of search
683  if ( priorArea )
684  {
685  float distAlong = priorArea->GetCostSoFar();
686  distAlong += ( area->GetCenter() - priorArea->GetCenter() ).Length();
687  area->SetCostSoFar( distAlong );
688  }
689  else
690  {
691  area->SetCostSoFar( 0.0f );
692  }
693 
694  // adding an area to the open list also marks it
695  area->AddToOpenList();
696  }
697  }

705 inline void SearchSurroundingAreas( CNavArea *startArea, ISearchSurroundingAreasFunctor &func, float travelDistanceLimit = -1.0f )
706 {
707  if ( startArea )
708  {
709  CNavArea::MakeNewMarker();
710  CNavArea::ClearSearchLists();
711 
712  startArea->AddToOpenList();
713  startArea->SetTotalCost( 0.0f );
714  startArea->SetCostSoFar( 0.0f );
715  startArea->SetParent( NULL );
716  startArea->Mark();
717 
718  CUtlVector< CNavArea * > adjVector;
719 
720  while( !CNavArea::IsOpenListEmpty() )
721  {
722  // get next area to check
723  CNavArea *area = CNavArea::PopOpenList();
724 
725  if ( travelDistanceLimit > 0.0f && area->GetCostSoFar() > travelDistanceLimit )
726  continue;
727 
728  if ( func( area, area->GetParent(), area->GetCostSoFar() ) )
729  {
730  func.IterateAdjacentAreas( area, area->GetParent(), area->GetCostSoFar() );
731  }
732  else

735  break;
736  }
737  }
738  }
739 
740  func.PostSearch();
741 }

751 inline void CollectSurroundingAreas( CUtlVector< CNavArea * > *nearbyAreaVector, CNavArea *startArea, float travelDistanceLimit = 1500.0f, float maxStepUpLimit = StepHeight, float maxDropDownLimit = 100.0f )
752 {
753  nearbyAreaVector->RemoveAll();
754 
755  if ( startArea )
756  {
757  CNavArea::MakeNewMarker();
758  CNavArea::ClearSearchLists();
759 
760  startArea->AddToOpenList();
761  startArea->SetTotalCost( 0.0f );
762  startArea->SetCostSoFar( 0.0f );
763  startArea->SetParent( NULL );
764  startArea->Mark();
765 
766  CUtlVector< CNavArea * > adjVector;
767 
768  while( !CNavArea::IsOpenListEmpty() )
769  {
770  // get next area to check
771  CNavArea *area = CNavArea::PopOpenList();
772 
773  if ( travelDistanceLimit > 0.0f && area->GetCostSoFar() > travelDistanceLimit )
774  continue;
775 
776  if ( area->GetParent() )
777  {
778  float deltaZ = area->GetParent()->ComputeAdjacentConnectionHeightChange( area );
779 
780  if ( deltaZ > maxStepUpLimit )
781  continue;
782 
783  if ( deltaZ < -maxDropDownLimit )
784  continue;
785  }
786 
787  nearbyAreaVector->AddToTail( area );
788 
789  // mark here to ensure all marked areas are also valid areas that are in the collection
790  area->Mark();
791 
792  // search adjacent outgoing connections
793  for( int dir=0; dir<NUM_DIRECTIONS; ++dir )
794  {
795  int count = area->GetAdjacentCount( (NavDirType)dir );
796  for( int i=0; i<count; ++i )
797  {
798  CNavArea *adjArea = area->GetAdjacentArea( (NavDirType)dir, i );
799 
800  if ( adjArea->IsBlocked( TEAM_ANY ) )
801  {
802  continue;
803  }
804 
805  if ( !adjArea->IsMarked() )
806  {
807  adjArea->SetTotalCost( 0.0f );
808  adjArea->SetParent( area );
809 
810  // compute approximate travel distance from start area of search
811  float distAlong = area->GetCostSoFar();
812  distAlong += ( adjArea->GetCenter() - area->GetCenter() ).Length();
813  adjArea->SetCostSoFar( distAlong );
814  adjArea->AddToOpenList();
815  }
816  }
817  }
818  }
819  }

828 class FarAwayFunctor
829 {
830 public:
831  float operator() ( CNavArea *area, CNavArea *fromArea, const CNavLadder *ladder )
832  {
833  if (area == fromArea)

864 template< typename CostFunctor >
865 CNavArea *FindMinimumCostArea( CNavArea *startArea, CostFunctor &costFunc )
866 {
867  const float minSize = 150.0f;
868 
869  // collect N low-cost areas of a decent size
870  enum { NUM_CHEAP_AREAS = 32 };
871  struct
872  {
873  CNavArea *area;
874  float cost;
875  }
876  cheapAreaSet[ NUM_CHEAP_AREAS ] = {};
877  int cheapAreaSetCount = 0;
878 
879  FOR_EACH_VEC( TheNavAreas, iter )
880  {
881  CNavArea *area = TheNavAreas[iter];
882 
883  // skip the small areas
884  if ( area->GetSizeX() < minSize || area->GetSizeY() < minSize)
885  continue;
886 
887  // compute cost of this area
888 
889  // HPE_FIX[pfreese]: changed this to only pass three parameters, in accord with the two functors above
890  float cost = costFunc( area, startArea, NULL );
891 
892  if (cheapAreaSetCount < NUM_CHEAP_AREAS)
893  {
894  cheapAreaSet[ cheapAreaSetCount ].area = area;
895  cheapAreaSet[ cheapAreaSetCount++ ].cost = cost;
896  }
897  else
898  {
899  // replace most expensive cost if this is cheaper
900  int expensive = 0;
901  for( int i=1; i<NUM_CHEAP_AREAS; ++i )
902  if (cheapAreaSet[i].cost > cheapAreaSet[expensive].cost)
903  expensive = i;
904 
905  if (cheapAreaSet[expensive].cost > cost)
906  {
907  cheapAreaSet[expensive].area = area;
908  cheapAreaSet[expensive].cost = cost;
909  }
910  }
911  }
912 
913  if (cheapAreaSetCount)
914  {
915  // pick one of the areas at random
916  return cheapAreaSet[ RandomInt( 0, cheapAreaSetCount-1 ) ].area;
917  }
918  else
919  {
920  // degenerate case - no decent sized areas - pick a random area
921  int numAreas = TheNavAreas.Count();
922  int which = RandomInt( 0, numAreas-1 );
923 
924  FOR_EACH_VEC( TheNavAreas, iter )
925  {
926  if (which-- == 0)
927  return TheNavAreas[iter];
928  }
929 
930  }
931  return cheapAreaSet[ RandomInt( 0, cheapAreaSetCount-1 ) ].area;
932 }

936 //
937 // Given a vector of CNavAreas (or derived types), 'inVector', populate 'outVector' with a randomly shuffled set
938 // of 'maxCount' areas that are at least 'minSeparation' travel distance apart from each other.
939 //
940 template< typename T >
941 void SelectSeparatedShuffleSet( int maxCount, float minSeparation, const CUtlVector< T * > &inVector, CUtlVector< T * > *outVector )
942 {
943  if ( !outVector )
944  return;
945 
946  outVector->RemoveAll();
947 
948  CUtlVector< T * > shuffledVector;
949 
950  int i, j;
951 
952  for( i=0; i<inVector.Count(); ++i )
953  {
954  shuffledVector.AddToTail( inVector[i] );
955  }
956 
957  // randomly shuffle the order
958  int n = shuffledVector.Count();
959  while( n > 1 )
960  {
961  int k = RandomInt( 0, n-1 );
962  n--;
963 
964  T *tmp = shuffledVector[n];
965  shuffledVector[n] = shuffledVector[k];
966  shuffledVector[k] = tmp;
967  }
968 
969  // enforce minSeparation between shuffled areas
970  for( i=0; i<shuffledVector.Count(); ++i )
971  {
972  T *area = shuffledVector[i];
973 
974  CUtlVector< CNavArea * > nearVector;
975  CollectSurroundingAreas( &nearVector, area, minSeparation, 2.0f * StepHeight, 2.0f * StepHeight );
976 
977  for( j=0; j<i; ++j )
978  {
979  if ( nearVector.HasElement( (CNavArea *)shuffledVector[j] ) )
980  {
981  // this area is too near an area earlier in the vector
982  break;
983  }
984  }
985 
986  if ( j == i )
987  {
988  // separated from all prior areas
989  outVector->AddToTail( area );
990 
991  if ( outVector->Count() >= maxCount )
992  return;
993  }
994  }
995 }
