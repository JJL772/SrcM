44 //-----------------------------------------------------------------------------
45 // Purpose: Singleton class responsible for managing vgui panel metaclasses
46 // A metaclass is simply an association of panel implementation class with
47 // various initialization data
48 //-----------------------------------------------------------------------------
49 abstract_class IPanelMetaClassMgr
50 {
51 public:
52  // Call this to load up a file containing metaclass definitions
53  virtual void LoadMetaClassDefinitionFile( const char *pFileName ) = 0;
54 
55  // Call this to install a new panel type
56  // MetaClasses will refer to the panel type to create along with
57  // various initialization data
58  virtual void InstallPanelType( const char *pPanelName, IPanelFactory *pFactory ) = 0;
59 
60  // Creates a metaclass panel with the specified parent panel.
61  // Chain name is used as a filter of the metaclass data; if specified,
62  // it recursively iterates through the keyvalue sections and calls
63  // chainKeyValue on sections whose name matches the chain name
64  virtual vgui::Panel *CreatePanelMetaClass( const char *pMetaClassType,
65  int sortorder, void *pInitData, vgui::Panel *pParent, const char *pChainName = NULL ) = 0;
66 
67  // removes a particular panel meta class
68  virtual void DestroyPanelMetaClass( vgui::Panel *pPanel ) = 0;
69 
70 protected:
71  // Don't delete me!
72  virtual ~IPanelMetaClassMgr() {}
73 };

82 //-----------------------------------------------------------------------------
83 // Helper class for simple construction of planel class factories
84 // This class is expected to be a singleton
85 // Note the panel must have a constructor of the following form:
86 // CPanel( vgui::Panel* );
87 // and it must have the following member function:
88 // bool CPanel::Init( KeyValues* pInitData )
89 // which returns true if the panel initialized successfully
90 //-----------------------------------------------------------------------------
91 
92 #include "tier0/memdbgon.h"
93 
94 template< class CPanel, class CInitData >
95 class CPanelFactory : public IPanelFactory
96 {
97 public:
98  CPanelFactory( const char *pTypeName )
99  {
100  // Hook us up baby
101  Assert( pTypeName );
102  PanelMetaClassMgr()->InstallPanelType( pTypeName, this );
103  }
104 
105  // Creation, destruction methods
106  virtual vgui::Panel *Create( const char *pMetaClassName, KeyValues* pKeyValues, void *pVoidInitData, vgui::Panel *pParent )
107  {
108  // NOTE: make sure this matches the panel allocation pattern;
109  // it will break if panels are deleted differently
110  CPanel* pPanel = new CPanel( pParent, pMetaClassName );
111  if (pPanel)
112  {
113  // Set parent before Init; it may be needed there...
114  CInitData* pInitData = (CInitData*)(pVoidInitData);
115  if (!pPanel->Init( pKeyValues, pInitData ))
116  {
117  delete pPanel;
118  pPanel = NULL;
119  }
120  }
121  return pPanel;
122  }
123 };
