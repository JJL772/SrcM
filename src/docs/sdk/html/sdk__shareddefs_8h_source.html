168 // The various states the player can be in during the join game process.
169 enum SDKPlayerState
170 {
171  // Happily running around in the game.
172  // You can't move though if CSGameRules()->IsFreezePeriod() returns true.
173  // This state can jump to a bunch of other states like STATE_PICKINGCLASS or STATE_DEATH_ANIM.
174  STATE_ACTIVE=0,
175 
176  // This is the state you're in when you first enter the server.
177  // It's switching between intro cameras every few seconds, and there's a level info
178  // screen up.
179  STATE_WELCOME, // Show the level intro screen.
180 
181  // During these states, you can either be a new player waiting to join, or
182  // you can be a live player in the game who wants to change teams.
183  // Either way, you can't move while choosing team or class (or while any menu is up).
184 #if defined ( SDK_USE_TEAMS )
185  STATE_PICKINGTEAM, // Choosing team.
186 #endif
187 #if defined ( SDK_USE_PLAYERCLASSES )
188  STATE_PICKINGCLASS, // Choosing class.
189 #endif
190 
191  STATE_DEATH_ANIM, // Playing death anim, waiting for that to finish.
192  STATE_OBSERVER_MODE, // Noclipping around, watching players, etc.
193 
194  NUM_PLAYER_STATES
195 };
