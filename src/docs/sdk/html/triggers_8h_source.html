32  SF_TRIG_PUSH_AFFECT_PLAYER_ON_LADDER = 0x100, // if pushed object is player on a ladder, then this disengages them from the ladder (HL2only)
33  SF_TRIG_TOUCH_DEBRIS = 0x400, // Will touch physics debris objects
