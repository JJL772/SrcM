74 inline void FX_GetSplashLighting( Vector position, Vector *color, float *luminosity )
75 {
76  // Compute our lighting at our position
77  Vector totalColor = engine->GetLightForPoint( position, true );
78 
79  // Get our lighting information
80  UTIL_GetNormalizedColorTintAndLuminosity( totalColor, color, luminosity );
81 
82  // Fake a specular highlight (too dim otherwise)
83  if ( luminosity != NULL )
84  {
85  *luminosity = MIN( 1.0f, (*luminosity) * 4.0f );
86 
87  // Clamp so that we never go completely translucent
88  if ( *luminosity < 0.25f )
89  {
90  *luminosity = 0.25f;
91  }
92  }
