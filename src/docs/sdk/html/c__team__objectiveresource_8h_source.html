233  void GetHillData( int team, int hill, float &flStart, float &flEnd )
234  {
235  if ( hill < TEAM_TRAIN_MAX_HILLS && team < TEAM_TRAIN_MAX_TEAMS )
236  {
237  int index = ( hill * TEAM_TRAIN_FLOATS_PER_HILL ) + ( team * TEAM_TRAIN_MAX_HILLS * TEAM_TRAIN_FLOATS_PER_HILL );
238  if ( index < TEAM_TRAIN_HILLS_ARRAY_SIZE - 1 ) // - 1 because we want to look at 2 entries
239  {
240  flStart = m_flNodeHillData[index];
241  flEnd = m_flNodeHillData[index+1];
242  }
243  }
244  }
