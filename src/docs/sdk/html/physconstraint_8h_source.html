114  {
115  matrix3x4_t xformRef;
116  m_pConstraint->GetConstraintTransform( &xformRef, NULL );
117  IPhysicsObject *pObj = m_pConstraint->GetReferenceObject();
118  if ( pObj && pObj->IsMoveable() )
119  {
120  Vector pos, posWorld;
121  MatrixPosition( xformRef, pos );

124  }
125  }
