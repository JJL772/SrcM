37 struct AvatarImagePair_t
38 {
39  AvatarImagePair_t() { m_iAvatar = 0; }
40  AvatarImagePair_t( CSteamID steamID, int av ) { m_SteamID = steamID; m_iAvatar = av; }
41  bool operator<( const AvatarImagePair_t &rhs ) const
42  {
43  return m_SteamID.ConvertToUint64() < rhs.m_SteamID.ConvertToUint64() ||
44  ( m_SteamID.ConvertToUint64() == rhs.m_SteamID.ConvertToUint64() && m_iAvatar < rhs.m_iAvatar );
45  }
46 
47  CSteamID m_SteamID;
48  int m_iAvatar;
49 };
