160 struct Extent
161 {
162  Vector lo, hi;
163 
164  void Init( void )
165  {
166  lo.Init();
167  hi.Init();
168  }
169 
170  void Init( CBaseEntity *entity )
171  {
172  entity->CollisionProp()->WorldSpaceSurroundingBounds( &lo, &hi );
173  }
174 
175  float SizeX( void ) const { return hi.x - lo.x; }
176  float SizeY( void ) const { return hi.y - lo.y; }
177  float SizeZ( void ) const { return hi.z - lo.z; }
178  float Area( void ) const { return SizeX() * SizeY(); }
179 
180  // Increase bounds to contain the given point
181  void Encompass( const Vector &pos )
182  {
183  for ( int i=0; i<3; ++i )
184  {
185  if ( pos[i] < lo[i] )
186  {
187  lo[i] = pos[i];
188  }
189  else if ( pos[i] > hi[i] )
190  {
191  hi[i] = pos[i];
192  }
193  }
194  }
195 
196  // Increase bounds to contain the given extent
197  void Encompass( const Extent &extent )
198  {
199  Encompass( extent.lo );
200  Encompass( extent.hi );
201  }
202 
203  // return true if 'pos' is inside of this extent
204  bool Contains( const Vector &pos ) const
205  {
206  return (pos.x >= lo.x && pos.x <= hi.x &&
207  pos.y >= lo.y && pos.y <= hi.y &&
208  pos.z >= lo.z && pos.z <= hi.z);
209  }
210 
211  // return true if this extent overlaps the given one
212  bool IsOverlapping( const Extent &other ) const
213  {
214  return (lo.x <= other.hi.x && hi.x >= other.lo.x &&
215  lo.y <= other.hi.y && hi.y >= other.lo.y &&
216  lo.z <= other.hi.z && hi.z >= other.lo.z);
217  }
218 
219  // return true if this extent completely contains the given one
220  bool IsEncompassing( const Extent &other, float tolerance = 0.0f ) const
221  {
222  return (lo.x <= other.lo.x + tolerance && hi.x >= other.hi.x - tolerance &&
223  lo.y <= other.lo.y + tolerance && hi.y >= other.hi.y - tolerance &&
224  lo.z <= other.lo.z + tolerance && hi.z >= other.hi.z - tolerance);
225  }
226 };

312 inline NavDirType AngleToDirection( float angle )
313 {
314  while( angle < 0.0f )
315  angle += 360.0f;
316 
317  while( angle > 360.0f )
318  angle -= 360.0f;
319 
320  if (angle < 45 || angle > 315)
321  return EAST;
322 
323  if (angle >= 45 && angle < 135)
324  return SOUTH;
325 
326  if (angle >= 135 && angle < 225)
327  return WEST;
328 
329  return NORTH;
330 }

409 inline bool IsEntityWalkable( CBaseEntity *entity, unsigned int flags )
410 {
411  if (FClassnameIs( entity, "worldspawn" ))
412  return false;
413 
414  if (FClassnameIs( entity, "player" ))
415  return false;
416 
417  // if we hit a door, assume its walkable because it will open when we touch it
418  if (FClassnameIs( entity, "func_door*" ))
419  {
420 #ifdef PROBLEMATIC // cp_dustbowl doors dont open by touch - they use surrounding triggers
421  if ( !entity->HasSpawnFlags( SF_DOOR_PTOUCH ) )
422  {
423  // this door is not opened by touching it, if it is closed, the area is blocked
424  CBaseDoor *door = (CBaseDoor *)entity;
425  return door->m_toggle_state == TS_AT_TOP;
426  }
427 #endif // _DEBUG
428 
429  return (flags & WALK_THRU_FUNC_DOORS) ? true : false;
430  }
431 
432  if (FClassnameIs( entity, "prop_door*" ))
433  {
434  return (flags & WALK_THRU_PROP_DOORS) ? true : false;
435  }
436 
437  // if we hit a clip brush, ignore it if it is not BRUSHSOLID_ALWAYS
438  if (FClassnameIs( entity, "func_brush" ))
439  {
440  CFuncBrush *brush = (CFuncBrush *)entity;
441  switch ( brush->m_iSolidity )
442  {
443  case CFuncBrush::BRUSHSOLID_ALWAYS:
444  return false;
445  case CFuncBrush::BRUSHSOLID_NEVER:
446  return true;
447  case CFuncBrush::BRUSHSOLID_TOGGLE:
448  return (flags & WALK_THRU_TOGGLE_BRUSHES) ? true : false;
449  }
450  }
451 
452  // if we hit a breakable object, assume its walkable because we will shoot it when we touch it
453  if (FClassnameIs( entity, "func_breakable" ) && entity->GetHealth() && entity->m_takedamage == DAMAGE_YES)
454  return (flags & WALK_THRU_BREAKABLES) ? true : false;
455 
456  if (FClassnameIs( entity, "func_breakable_surf" ) && entity->m_takedamage == DAMAGE_YES)
457  return (flags & WALK_THRU_BREAKABLES) ? true : false;
458 
459  if ( FClassnameIs( entity, "func_playerinfected_clip" ) == true )
460  return true;
461 
462  if ( nav_solid_props.GetBool() && FClassnameIs( entity, "prop_*" ) )
463  return true;
464 
465  return false;
466 }
