381 inline float DistanceToRay( const Vector &pos, const Vector &rayStart, const Vector &rayEnd, float *along = NULL, Vector *pointOnRay = NULL )
382 {
383  Vector to = pos - rayStart;
384  Vector dir = rayEnd - rayStart;
385  float length = dir.NormalizeInPlace();
386 
387  float rangeAlong = DotProduct( dir, to );
388  if (along)
389  {
390  *along = rangeAlong;
391  }
392 
393  float range;
394 
395  if (rangeAlong < 0.0f)
396  {
397  // off start point
398  range = -(pos - rayStart).Length();
399 
400  if (pointOnRay)
401  {
402  *pointOnRay = rayStart;
403  }
404  }
405  else if (rangeAlong > length)
406  {
407  // off end point
408  range = -(pos - rayEnd).Length();
409 
410  if (pointOnRay)
411  {
412  *pointOnRay = rayEnd;
413  }
414  }
415  else // within ray bounds
416  {
417  Vector onRay = rayStart + rangeAlong * dir;
418  range = (pos - onRay).Length();
419 
420  if (pointOnRay)
421  {
422  *pointOnRay = onRay;
423  }
424  }
425 
426  return range;
427 }

446  static const CUtlVector< interfaceName* >& AutoList( void ) { return m_##interfaceName##AutoList; } \
447  private: \

449  };
