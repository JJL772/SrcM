42  struct CritEntry_t
43  {
44  CritEntry_t() :
45  criterianame( UTL_INVAL_SYMBOL ),
46  weight( 0.0f )
47  {
48  value[ 0 ] = 0;
49  }
50 
51  CritEntry_t( const CritEntry_t& src )
52  {
53  criterianame = src.criterianame;
54  value[ 0 ] = 0;
55  weight = src.weight;
56  SetValue( src.value );
57  }
58 
59  CritEntry_t& operator=( const CritEntry_t& src )
60  {
61  if ( this == &src )
62  return *this;
63 
64  criterianame = src.criterianame;
65  weight = src.weight;
66  SetValue( src.value );
67 
68  return *this;
69  }
70 
71  static bool LessFunc( const CritEntry_t& lhs, const CritEntry_t& rhs )
72  {
73  return Q_stricmp( lhs.criterianame.String(), rhs.criterianame.String() ) < 0 ? true : false;
74  }
75 
76  void SetValue( char const *str )
77  {
78  if ( !str )
79  {
80  value[ 0 ] = 0;
81  }
82  else
83  {
84  Q_strncpy( value, str, sizeof( value ) );
85  }
86  }

90  CUtlSymbol criterianame;
91  char value[ 64 ];
92  float weight;
93  };
