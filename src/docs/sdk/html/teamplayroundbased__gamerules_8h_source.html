43 enum gamerules_roundstate_t
44 {
45  // initialize the game, create teams
46  GR_STATE_INIT = 0,
47 
48  //Before players have joined the game. Periodically checks to see if enough players are ready
49  //to start a game. Also reverts to this when there are no active players
50  GR_STATE_PREGAME,
51 
52  //The game is about to start, wait a bit and spawn everyone
53  GR_STATE_STARTGAME,
54 
55  //All players are respawned, frozen in place
56  GR_STATE_PREROUND,
57 
58  //Round is on, playing normally
59  GR_STATE_RND_RUNNING,
60 
61  //Someone has won the round
62  GR_STATE_TEAM_WIN,
63 
64  //Noone has won, manually restart the game, reset scores
65  GR_STATE_RESTART,
66 
67  //Noone has won, restart the game
68  GR_STATE_STALEMATE,
69 
70  //Game is over, showing the scoreboard etc
71  GR_STATE_GAME_OVER,
72 
73  //Game is in a bonus state, transitioned to after a round ends
74  GR_STATE_BONUS,
75 
76  //Game is awaiting the next wave/round of a multi round experience
77  GR_STATE_BETWEEN_RNDS,
78 
79  GR_NUM_ROUND_STATES
80 };
