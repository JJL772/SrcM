304 template<typename T>
305 class CSimpleRingBuffer
306 {
307 public:
308  CSimpleRingBuffer( int startSize = 4 )
309  {
310  m_pElements = 0;
311  m_maxElement = 0;
312  m_firstElement = 0;
313  m_count = 0;
314  m_growSize = 16;
315  EnsureCapacity(startSize);
316  }
317  ~CSimpleRingBuffer()
318  {
319  delete[] m_pElements;
320  m_pElements = NULL;
321  }
322 
323  inline int Count() const { return m_count; }
324 
325  int Head() const { return (m_count>0) ? 0 : InvalidIndex(); }
326 
327  bool IsIdxValid( int i ) const { return (i >= 0 && i < m_count) ? true : false; }
328  bool IsValidIndex(int i) const { return IsIdxValid(i); }
329  static int InvalidIndex() { return -1; }
330 
331  T& operator[]( int i )
332  {
333  Assert( IsIdxValid(i) );
334  i += m_firstElement;
335  i = WrapRange(i);
336  return m_pElements[i];
337  }
338 
339  const T& operator[]( int i ) const
340  {
341  Assert( IsIdxValid(i) );
342  i += m_firstElement;
343  i = WrapRange(i);
344  return m_pElements[i];
345  }
346 
347  void EnsureCapacity( int capSize )
348  {
349  if ( capSize > m_maxElement )
350  {
351  int newMax = m_maxElement + ((capSize+m_growSize-1)/m_growSize) * m_growSize;
352  T *pNew = new T[newMax];
353  for ( int i = 0; i < m_maxElement; i++ )
354  {
355  // ------------
356  // If you wanted to make this a more generic container you'd probably want this code
357  // instead - since FastTransferFrom() is an optimization dependent on types stored
358  // here defining this operation.
359  //pNew[i] = m_pElements[WrapRange(i+m_firstElement)];
360  pNew[i].FastTransferFrom( m_pElements[WrapRange(i+m_firstElement)] );
361  // ------------
362  }
363  m_firstElement = 0;
364  m_maxElement = newMax;
365  delete[] m_pElements;
366  m_pElements = pNew;
367  }
368  }
369 
370  int AddToHead()
371  {
372  EnsureCapacity( m_count + 1 );
373  int i = m_firstElement + m_maxElement - 1;
374  m_count++;
375  i = WrapRange(i);
376  m_firstElement = i;
377  return 0;
378  }
379 
380  int AddToHead( const T &elem )
381  {
382  AddToHead();
383  m_pElements[m_firstElement] = elem;
384  return 0;
385  }
386 
387  int AddToTail()
388  {
389  EnsureCapacity( m_count + 1 );
390  m_count++;
391  return WrapRange(m_firstElement+m_count-1);
392  }
393 
394  void RemoveAll()
395  {
396  m_count = 0;
397  m_firstElement = 0;
398  }
399 
400  void RemoveAtHead()
401  {
402  if ( m_count > 0 )
403  {
404  m_firstElement = WrapRange(m_firstElement+1);
405  m_count--;
406  }
407  }
408 
409  void Truncate( int newLength )
410  {
411  if ( newLength < m_count )
412  {
413  Assert(newLength>=0);
414  m_count = newLength;
415  }
416  }
417 
418 private:
419  inline int WrapRange( int i ) const
420  {
421  return ( i >= m_maxElement ) ? (i - m_maxElement) : i;
422  }
423 
424  T *m_pElements;
425  unsigned short m_maxElement;
426  unsigned short m_firstElement;
427  unsigned short m_count;
428  unsigned short m_growSize;

698 template< typename Type, bool IS_ARRAY >
699 inline void CInterpolatedVarArrayBase<Type, IS_ARRAY>::AddToHead( float changeTime, const Type* values, bool bFlushNewer )
700 {
701  MEM_ALLOC_CREDIT_CLASS();
702  int newslot;
703 
704  if ( bFlushNewer )
705  {
706  // Get rid of anything that has a timestamp after this sample. The server might have
707  // corrected our clock and moved us back, so our current changeTime is less than a
708  // changeTime we added samples during previously.
709  while ( m_VarHistory.Count() )
710  {
711  if ( (m_VarHistory[0].changetime+0.0001f) > changeTime )
712  {
713  m_VarHistory.RemoveAtHead();
714  }
715  else
716  {
717  break;
718  }
719  }
720 
721  newslot = m_VarHistory.AddToHead();
722  }
723  else
724  {
725  newslot = m_VarHistory.AddToHead();
726  for ( int i = 1; i < m_VarHistory.Count(); i++ )
727  {
728  if ( m_VarHistory[i].changetime <= changeTime )
729  break;
730  m_VarHistory[newslot].FastTransferFrom( m_VarHistory[i] );
731  newslot = i;
732  }
733  }
734 
735  CInterpolatedVarEntry *e = &m_VarHistory[ newslot ];
736  e->NewEntry( values, m_nMaxCount, changeTime );
737 }

798 template< typename Type, bool IS_ARRAY >
799 inline bool CInterpolatedVarArrayBase<Type, IS_ARRAY>::GetInterpolationInfo(
800  typename CInterpolatedVarArrayBase<Type, IS_ARRAY>::CInterpolationInfo *pInfo,
801  float currentTime,
802  float interpolation_amount,
803  int *pNoMoreChanges
804  )
805 {
806  Assert( m_pValue );
807 
808  CVarHistory &varHistory = m_VarHistory;
809 
810  float targettime = currentTime - interpolation_amount;
811 
812  pInfo->m_bHermite = false;
813  pInfo->frac = 0;
814  pInfo->oldest = pInfo->older = pInfo->newer = varHistory.InvalidIndex();
815 
816  for ( int i = 0; i < varHistory.Count(); i++ )
817  {
818  pInfo->older = i;
819 
820  float older_change_time = m_VarHistory[ i ].changetime;
821  if ( older_change_time == 0.0f )
822  break;
823 
824  if ( targettime < older_change_time )
825  {
826  pInfo->newer = pInfo->older;
827  continue;
828  }
829 
830  if ( pInfo->newer == varHistory.InvalidIndex() )
831  {
832  // Have it linear interpolate between the newest 2 entries.
833  pInfo->newer = pInfo->older;
834 
835  // Since the time given is PAST all of our entries, then as long
836  // as time continues to increase, we'll be returning the same value.
837  if ( pNoMoreChanges )
838  *pNoMoreChanges = 1;
839  return true;
840  }
841 
842  float newer_change_time = varHistory[ pInfo->newer ].changetime;
843  float dt = newer_change_time - older_change_time;
844  if ( dt > 0.0001f )
845  {
846  pInfo->frac = ( targettime - older_change_time ) / ( newer_change_time - older_change_time );
847  pInfo->frac = MIN( pInfo->frac, 2.0f );
848 
849  int oldestindex = i+1;
850 
851  if ( !(m_fType & INTERPOLATE_LINEAR_ONLY) && varHistory.IsIdxValid(oldestindex) )
852  {
853  pInfo->oldest = oldestindex;
854  float oldest_change_time = varHistory[ oldestindex ].changetime;
855  float dt2 = older_change_time - oldest_change_time;
856  if ( dt2 > 0.0001f )
857  {
858  pInfo->m_bHermite = true;
859  }
860  }
861 
862  // If pInfo->newer is the most recent entry we have, and all 2 or 3 other
863  // entries are identical, then we're always going to return the same value
864  // if currentTime increases.
865  if ( pNoMoreChanges && pInfo->newer == m_VarHistory.Head() )
866  {
867  if ( COMPARE_HISTORY( pInfo->newer, pInfo->older ) )
868  {
869  if ( !pInfo->m_bHermite || COMPARE_HISTORY( pInfo->newer, pInfo->oldest ) )
870  *pNoMoreChanges = 1;
871  }
872  }
873  }
874  return true;
875  }
876 
877  // Didn't find any, return last entry???
878  if ( pInfo->newer != varHistory.InvalidIndex() )
879  {
880  pInfo->older = pInfo->newer;
881  return true;
882  }
883 
884 
885  // This is the single-element case
886  pInfo->newer = pInfo->older;
887  return (pInfo->older != varHistory.InvalidIndex());
888 }
