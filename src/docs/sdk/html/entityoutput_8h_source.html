93 //-----------------------------------------------------------------------------
94 // Purpose: wraps variant_t data handling in convenient, compiler type-checked template
95 //-----------------------------------------------------------------------------
96 template< class Type, fieldtype_t fieldType >
97 class CEntityOutputTemplate : public CBaseEntityOutput
98 {
99 public:
100  //
101  // Sets an initial value without firing the output.
102  //
103  void Init( Type value )
104  {
105  m_Value.Set( fieldType, &value );
106  }
107 
108  //
109  // Sets a value and fires the output.
110  //
111  void Set( Type value, CBaseEntity *pActivator, CBaseEntity *pCaller )
112  {
113  m_Value.Set( fieldType, &value );
114  FireOutput( m_Value, pActivator, pCaller );
115  }
116 
117  //
118  // Returns the current value.
119  //
120  Type Get( void )
121  {
122  return *((Type*)&m_Value);
123  }
124 };
