50 struct FS_LocalToGlobal_t
51 {
52  explicit FS_LocalToGlobal_t() :
53  m_Key( 0 ),
54  m_nCount( 0 ),
55  m_Mapping( 0 )
56  {
57  }
58 
59  explicit FS_LocalToGlobal_t( const flexsettinghdr_t *key ) :
60  m_Key( key ),
61  m_nCount( 0 ),
62  m_Mapping( 0 )
63  {
64  }
65 
66  void SetCount( int count )
67  {
68  Assert( !m_Mapping );
69  Assert( count > 0 );
70  m_nCount = count;
71  m_Mapping = new int[ m_nCount ];
72  Q_memset( m_Mapping, 0, m_nCount * sizeof( int ) );
73  }
74 
75  FS_LocalToGlobal_t( const FS_LocalToGlobal_t& src )
76  {
77  m_Key = src.m_Key;
78  delete m_Mapping;
79  m_Mapping = new int[ src.m_nCount ];
80  Q_memcpy( m_Mapping, src.m_Mapping, src.m_nCount * sizeof( int ) );
81 
82  m_nCount = src.m_nCount;
83  }
84 
85  ~FS_LocalToGlobal_t()
86  {
87  delete m_Mapping;
88  m_nCount = 0;
89  m_Mapping = 0;
90  }
91 
92  const flexsettinghdr_t *m_Key;
93  int m_nCount;
94  int *m_Mapping;
95 };
