129 //-----------------------------------------------------------------------------
130 // Set up the class's sentence information
131 //-----------------------------------------------------------------------------
132 template< class NPC_CLASS >
133 void CAI_Sentence< NPC_CLASS >::Init( NPC_CLASS *pOuter, const char *pGameSound )
134 {
135  SetOuter( pOuter );
136 
137  if ( m_voicePitchMin <= 0 || m_voicePitchMax <= 0 )
138  {
139  // init the sentence parameters using a dummy gamesounds entry
140  CSoundParameters params;
141  if ( GetOuter()->GetParametersForSound( pGameSound, params, NULL ) )
142  {
143  m_sentenceVolume = params.volume;
144  m_sentenceSoundlevel = params.soundlevel;
145  m_voicePitchMin = params.pitchlow;
146  m_voicePitchMax = params.pitchhigh;
147  }
148  }
149 
150  // Select a voice pitch
151  if ( random->RandomInt(0,1) )
152  {
153  SetVoicePitch( random->RandomInt( m_voicePitchMin, m_voicePitchMax ) );
154  }
155  else
156  {
157  SetVoicePitch( 100 );
158  }
159 }
