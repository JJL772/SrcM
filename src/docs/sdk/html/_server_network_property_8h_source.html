75  // This is useful for entities that don't change frequently or that the client
76  // doesn't need updates on very often. If you use this mode, the server will only try to
77  // detect state changes every N seconds, so it will save CPU cycles and bandwidth.
78  //
79  // Note: N must be less than AUTOUPDATE_MAX_TIME_LENGTH.
80  //
81  // Set back to zero to disable the feature.
82  //
83  // This feature works on top of manual mode.
84  // - If you turn it on and manual mode is off, it will autodetect changes every N seconds.
85  // - If you turn it on and manual mode is on, then every N seconds it will only say there
86  // is a change if you've called NetworkStateChanged.
87  void SetUpdateInterval( float N );

185 inline void CServerNetworkProperty::NetworkStateChanged()
186 {
187  // If we're using the timer, then ignore this call.
188  if ( m_TimerEvent.IsRegistered() )
189  {
190  // If we're waiting for a timer event, then queue the change so it happens
191  // when the timer goes off.
192  m_bPendingStateChange = true;
193  }
194  else
195  {
196  if ( m_pPev )
197  m_pPev->StateChanged();
198  }
199 }
