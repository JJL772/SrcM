84 inline float C_AnimationLayer::GetFadeout( float flCurTime )
85 {
86  float s;
87 
88  if (m_flLayerFadeOuttime <= 0.0f)
89  {
90  s = 0;
91  }
92  else
93  {
94  // blend in over 0.2 seconds
95  s = 1.0 - (flCurTime - m_flLayerAnimtime) / m_flLayerFadeOuttime;
96  if (s > 0 && s <= 1.0)
97  {
98  // do a nice spline curve
99  s = 3 * s * s - 2 * s * s * s;
100  }
101  else if ( s > 1.0f )
102  {
103  // Shouldn't happen, but maybe curtime is behind animtime?
104  s = 1.0f;
105  }
106  }
107  return s;
108 }

183 inline void C_AnimationLayer::BlendWeight()
184 {
185  if ( !m_bClientBlend )
186  return;
187 
188  m_flWeight = 1;
189 
190  // blend in?
191  if ( m_flBlendIn != 0.0f )
192  {
193  if (m_flCycle < m_flBlendIn)
194  {
195  m_flWeight = m_flCycle / m_flBlendIn;
196  }
197  }
198 
199  // blend out?
200  if ( m_flBlendOut != 0.0f )
201  {
202  if (m_flCycle > 1.0 - m_flBlendOut)
203  {
204  m_flWeight = (1.0 - m_flCycle) / m_flBlendOut;
205  }
206  }
207 
208  m_flWeight = 3.0 * m_flWeight * m_flWeight - 2.0 * m_flWeight * m_flWeight * m_flWeight;
209  if (m_nSequence == 0)
210  m_flWeight = 0;
211 }
