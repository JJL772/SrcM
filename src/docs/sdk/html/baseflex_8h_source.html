197  void SetCount( int count )
198  {
199  Assert( !m_Mapping );
200  Assert( count > 0 );
201  m_nCount = count;

203  Q_memset( m_Mapping, 0, m_nCount * sizeof( int ) );
204  }
205 
206  FS_LocalToGlobal_t( const FS_LocalToGlobal_t& src )
207  {
208  m_Key = src.m_Key;
209  delete m_Mapping;
210  m_Mapping = new LocalFlexController_t[ src.m_nCount ];
211  Q_memcpy( m_Mapping, src.m_Mapping, src.m_nCount * sizeof( int ) );
212 
213  m_nCount = src.m_nCount;
214  }
215 
216  ~FS_LocalToGlobal_t()

218  delete m_Mapping;
219  m_nCount = 0;
