97 inline float CAnimationLayer::GetFadeout( float flCurTime )
98 {
99  float s;
100 
101  if (m_flLayerFadeOuttime <= 0.0f)
102  {
103  s = 0;
104  }
105  else
106  {
107  // blend in over 0.2 seconds
108  s = 1.0 - (flCurTime - m_flLayerAnimtime) / m_flLayerFadeOuttime;
109  if (s > 0 && s <= 1.0)
110  {
111  // do a nice spline curve
112  s = 3 * s * s - 2 * s * s * s;
113  }
114  else if ( s > 1.0f )
115  {
116  // Shouldn't happen, but maybe curtime is behind animtime?
117  s = 1.0f;
118  }
119  }
120  return s;
121 }
