27 {

35  virtual void PostInit() = 0;

37

42

46  virtual void LevelShutdownPostEntity() = 0;
47  // end of level shutdown
48

50  virtual void OnSave() = 0;
51 
52  // Called during game restore, after the local player has connected and entities have been fully restored
53  virtual void OnRestore() = 0;
54 
55  // Called every frame. It's safe to remove an igamesystem from within this callback.
56  virtual void SafeRemoveIfDesired() = 0;
57 
58  virtual bool IsPerFrame() = 0;
59 
60  // destructor, cleans up automagically....
61  virtual ~IGameSystem();
62 
63  // Client systems can use this to get at the map name
64  static char const* MapName();
65 
66  // These methods are used to add and remove server systems from the
67  // main server loop. The systems are invoked in the order in which
68  // they are added.

70  static void Remove ( IGameSystem* pSys );
71  static void RemoveAll ( );

87 
88 #ifdef CLIENT_DLL

92 #else
93  static void FrameUpdatePreEntityThinkAllSystems();

95  static void PreClientUpdateAllSystems();

99  static CUserCmd *RunCommandUserCmd();
