41  inline bool NextEvent(float &curDelta)
42  {
43  // If this goes off, you didn't call Init().
44  Assert( m_TimeBetweenEvents != -1 );
45 
46  if(curDelta >= m_fNextEvent)

48  curDelta -= m_fNextEvent;

52  }
53  else
54  {

57  }
58  }
