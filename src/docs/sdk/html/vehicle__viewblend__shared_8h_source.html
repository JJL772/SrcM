21 // If we enter the linear part of the remap for curve for any degree of freedom, we can lock
22 // that DOF (stop remapping). This is useful for making flips feel less spastic as we oscillate
23 // randomly between different parts of the remapping curve.
24 struct ViewLockData_t
25 {
26  float flLockInterval; // The duration to lock the view when we lock it for this degree of freedom.
27  // 0 = never lock this degree of freedom.
28 
29  bool bLocked; // True if this DOF was locked because of the above condition.
30 
31  float flUnlockTime; // If this DOF is locked, the time when we will unlock it.
32 
33  float flUnlockBlendInterval; // If this DOF is locked, how long to spend blending out of the locked view when we unlock.
34 };
