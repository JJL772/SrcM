151

153  {
154  m_closeRangeSq = maxRange * maxRange;

156  else
157  {
158  m_closeRangeSq = 999999999.9f;
159  }
160 
161  m_ignore = ignore;
162  }
163 
164  bool operator() ( CBasePlayer *player )
165  {
166  if (player == m_ignore)
167  return true;
168 
169  if (player->IsAlive() && (m_team == TEAM_ANY || player->GetTeamNumber() == m_team))
170  {
171  if ( !m_includeBots && player->IsBot() )
172  return true;
173 
174  Vector to = player->WorldSpaceCenter() - m_spot;
175  float rangeSq = to.LengthSqr();
176  if (rangeSq < m_closeRangeSq)
177  {
178  m_closeRangeSq = rangeSq;
179  m_close = player;
180  }
181  }
182  return true;
183  }
184 
185  CBasePlayer *GetPlayer( void ) const
186  {
187  return m_close;
188  }
189 
190  bool IsCloserThan( float range )
191  {
192  return (m_closeRangeSq < (range * range));
193  }
194 
195  bool IsFartherThan( float range )
196  {
197  return (m_closeRangeSq > (range * range));
198  }
199

201  int m_team;
202  bool m_includeBots;
