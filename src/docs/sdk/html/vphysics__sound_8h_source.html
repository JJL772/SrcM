40  void PlayImpactSounds( soundlist_t &list )
41  {
42  for ( int i = list.Count()-1; i >= 0; --i )
43  {
44  impactsound_t &sound = list.GetElement(i);
45  const surfacedata_t *psurf = physprops->GetSurfaceData( sound.surfaceProps );
46  if ( psurf->sounds.impactHard )
47  {
48  const surfacedata_t *pHit = physprops->GetSurfaceData( sound.surfacePropsHit );
49  unsigned short soundName = psurf->sounds.impactHard;
50  if ( pHit && psurf->sounds.impactSoft )
51  {
52  if ( pHit->audio.hardnessFactor < psurf->audio.hardThreshold ||
53  (psurf->audio.hardVelocityThreshold > 0 && psurf->audio.hardVelocityThreshold > sound.impactSpeed) )
54  {
55  soundName = psurf->sounds.impactSoft;
56  }
57  }
58  const char *pSound = physprops->GetString( soundName );
59 
60  CSoundParameters params;
61  if ( !CBaseEntity::GetParametersForSound( pSound, params, NULL ) )
62  break;
63 
64  if ( sound.volume > 1 )
65  sound.volume = 1;
66  CPASAttenuationFilter filter( sound.origin, params.soundlevel );
67  // JAY: If this entity gets deleted, the sound comes out at the world origin
68  // this sounds bad! Play on ent 0 for now.
69  EmitSound_t ep;
70  ep.m_nChannel = sound.soundChannel;
71  ep.m_pSoundName = params.soundname;
72  ep.m_flVolume = params.volume * sound.volume;
73  ep.m_SoundLevel = params.soundlevel;
74  ep.m_nPitch = params.pitch;
75  ep.m_pOrigin = &sound.origin;
76 
77  CBaseEntity::EmitSound( filter, 0 /*sound.entityIndex*/, ep );
78  }
79  }
80  list.RemoveAll();
81  }

88  // UNDONE: Compare entity or channel somehow?

90

92  if ( surfaceProps == sound.surfaceProps || list.Count() > 4 )
93  {
94  // UNDONE: Store instance volume separate from aggregate volume and compare that?

126  void AddBreakSound( CUtlVector<breaksound_t> &list, const Vector &origin, unsigned short surfaceProps )
127  {
128  const surfacedata_t *psurf = physprops->GetSurfaceData( surfaceProps );
129  if ( !psurf->sounds.breakSound )
130  return;
131 
132  for ( int i = list.Count()-1; i >= 0; --i )
133  {
134  breaksound_t &sound = list.Element(i);
135  // Allow 3 break sounds before you start merging anything.
136  if ( list.Count() > 2 && surfaceProps == sound.surfacePropsBreak )
137  {
138  sound.origin = (sound.origin + origin) * 0.5f;
139  return;
140  }
141  }
142  breaksound_t sound;
143  sound.origin = origin;
144  sound.surfacePropsBreak = surfaceProps;
145  list.AddToTail(sound);
146 
147  }
