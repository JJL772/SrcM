193 class CFireOverlay : public CGlowOverlay
194 {
195 public:
196 
197  //Constructor
198  CFireOverlay( C_FireSmoke *owner )
199  {
200  m_pOwner = owner;
201  m_flScale = 0.0f;
202  m_nGUID = random->RandomInt( -999999, 999999 );
203  }
204 
205  //-----------------------------------------------------------------------------
206  // Purpose: Generate a flicker value
207  // Output : scalar value
208  //-----------------------------------------------------------------------------
209  float GetFlickerScale( void )
210  {
211  float result = 0.0f;
212 
213  float time = Helper_GetTime() + m_nGUID;
214 
215  result = sin( time * 1000.0f );
216  result += 0.5f * sin( time * 2000.0f );
217  result -= 0.5f * cos( time * 8000.0f );
218 
219  return result;
220  }
221 
222  //-----------------------------------------------------------------------------
223  // Purpose: Update the overlay
224  //-----------------------------------------------------------------------------
225  virtual bool Update( void )
226  {
227  if ( m_pOwner == NULL )
228  return false;
229 
230  float scale = m_pOwner->GetScale();
231  float dscale = scale - m_flScale;
232 
233  m_vPos[2] += dscale * FLAME_SOURCE_HEIGHT;
234  m_flScale = scale;
235 
236  scale *= 0.75f;
237 
238  float flickerScale = GetFlickerScale();
239 
240  float newScale = scale + ( scale * flickerScale * 0.1f );
241 
242  m_Sprites[0].m_flHorzSize = ( newScale * 0.2f ) + ( m_Sprites[0].m_flHorzSize * 0.8f );
243  m_Sprites[0].m_flVertSize = m_Sprites[0].m_flHorzSize * 1.5f;
244 
245  float cameraDistance = ( CurrentViewOrigin() - (m_pOwner->GetAbsOrigin())).Length();
246 
247  C_BasePlayer *local = C_BasePlayer::GetLocalPlayer();
248  if ( local )
249  {
250  cameraDistance *= local->GetFOVDistanceAdjustFactor();
251  }
252 
253  if ( cameraDistance > OVERLAY_MAX_VISIBLE_RANGE )
254  cameraDistance = OVERLAY_MAX_VISIBLE_RANGE;
255 
256  float alpha = 1.0f - ( cameraDistance / OVERLAY_MAX_VISIBLE_RANGE );
257 
258  Vector newColor = m_vBaseColors[0] + ( m_vBaseColors[0] * flickerScale * 0.5f );
259  m_Sprites[0].m_vColor = ( newColor * 0.1f ) + ( m_Sprites[0].m_vColor * 0.9f ) * alpha;
260 
261  return true;
262  }
263 
264 public:
265 
266  C_FireSmoke *m_pOwner;
267  Vector m_vBaseColors[MAX_SUN_LAYERS];
268  float m_flScale;
269  int m_nGUID;
270 };
