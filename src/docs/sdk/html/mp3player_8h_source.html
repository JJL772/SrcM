72 // Purpose: A directory has a name, 0 or more subdirectories and 0 or more mp3 files in it
73 //-----------------------------------------------------------------------------
74 struct MP3Dir_t
75 {
76  MP3Dir_t() :
77  m_DirName( UTL_INVAL_SYMBOL ),
78  m_FullDirPath( UTL_INVAL_SYMBOL )
79  {
80  }
81 
82  ~MP3Dir_t()
83  {
84  DeleteSubdirectories();
85  }
86 
87  void DeleteSubdirectories()
88  {
89  int i, c;
90 
91  c = m_Subdirectories.Count();
92  for ( i = c - 1; i >= 0 ; --i )
93  {
94  delete m_Subdirectories[ i ];
95  }
96  m_Subdirectories.RemoveAll();
97  }
98 
99  MP3Dir_t( const MP3Dir_t& src )
100  {
101  m_DirName = src.m_DirName;
102  m_FullDirPath = src.m_FullDirPath;
103 
104  int i, c;
105 
106  c = src.m_Subdirectories.Count();
107  for ( i = 0; i < c; ++i )
108  {
109  MP3Dir_t *subCopy = new MP3Dir_t( *src.m_Subdirectories[ i ] );
110  m_Subdirectories.AddToTail( subCopy );
111  }
112 
113  c = src.m_FilesInDirectory.Count();
114  for ( i = 0; i < c; ++i )
115  {
116  m_FilesInDirectory.AddToTail( src.m_FilesInDirectory[ i ] );
117  }
118  }
119 
120  MP3Dir_t &operator =( const MP3Dir_t& src )
121  {
122  if ( this == &src )
123  {
124  return *this;
125  }
126 
127  m_DirName = src.m_DirName;
128  m_FullDirPath = src.m_FullDirPath;
129 
130  DeleteSubdirectories();
131 
132  m_FilesInDirectory.RemoveAll();
133 
134  int i, c;
135 
136  c = src.m_Subdirectories.Count();
137  for ( i = 0; i < c; ++i )
138  {
139  // make a copy
140  MP3Dir_t *subCopy = new MP3Dir_t( *src.m_Subdirectories[ i ] );
141  m_Subdirectories.AddToTail( subCopy );
142  }
143 
144  c = src.m_FilesInDirectory.Count();
145  for ( i = 0; i < c; ++i )
146  {
147  m_FilesInDirectory.AddToTail( src.m_FilesInDirectory[ i ] );
148  }
149 
150  return *this;
151  }
152 
153  void AddSubDirectory( MP3Dir_t *sub )
154  {
155  m_Subdirectories.AddToTail( sub );
156  }
157 
158  CUtlSymbol m_DirName; // "artist"
159  CUtlSymbol m_FullDirPath; // "artist/album
160  CUtlVector< MP3Dir_t * > m_Subdirectories;
161  CUtlVector< int > m_FilesInDirectory;
162 };
