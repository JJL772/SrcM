57  bits_CAP_WEAPON_RANGE_ATTACK2 = 0x00004000, // can do a weapon range attack 2

62  bits_CAP_INNATE_MELEE_ATTACK1 = 0x00080000, // can do a innate melee attack 1
63  bits_CAP_INNATE_MELEE_ATTACK2 = 0x00100000, // can do a innate melee attack 1
64  bits_CAP_USE_WEAPONS = 0x00200000, // can use weapons (non-innate attacks)
65  //bits_CAP_STRAFE = 0x00400000, // strafe ( walk/run sideways)
66  bits_CAP_ANIMATEDFACE = 0x00800000, // has animated eyes/face
67  bits_CAP_USE_SHOT_REGULATOR = 0x01000000, // Uses the shot regulator for range attack1
68  bits_CAP_FRIENDLY_DMG_IMMUNE = 0x02000000, // don't take damage from npc's that are D_LI
69  bits_CAP_SQUAD = 0x04000000, // can form squads
70  bits_CAP_DUCK = 0x08000000, // cover and reload ducking
71  bits_CAP_NO_HIT_PLAYER = 0x10000000, // don't hit players
72  bits_CAP_AIM_GUN = 0x20000000, // Use arms to aim gun, not just body
73  bits_CAP_NO_HIT_SQUADMATES = 0x40000000, // none
