285 inline bool CBaseGameStats::AppendLump( int nMaxLumpCount, CUtlBuffer &SaveBuffer, unsigned short iLump, unsigned short iLumpCount, size_t nSize, void *pData )
286 {
287  // Verify the lump index.
288  Assert( ( iLump > 0 ) && ( iLump < nMaxLumpCount ) );
289 
290  if ( !( ( iLump > 0 ) && ( iLump < nMaxLumpCount ) ) )
291  return false;
292 
293  // Check to see if we have any elements to save.
294  if ( iLumpCount <= 0 )
295  return false;
296 
297  // Write the lump id and element count.
298  SaveBuffer.PutUnsignedShort( iLump );
299  SaveBuffer.PutUnsignedShort( iLumpCount );
300 
301  size_t nTotalSize = iLumpCount * nSize;
302  SaveBuffer.Put( pData, nTotalSize );
303 
304  return true;
305 }

315 {
316  // Get the lump id and element count.

318  if ( !LoadBuffer.IsValid() )
319  {
320  // check for EOF

323  iLumpCount = LoadBuffer.GetUnsignedShort();
324 
325  if ( bPermissive )
326  return true;
327 
328  // Verify the lump index.
329  Assert( ( iLump > 0 ) && ( iLump < nMaxLumpCount ) );
330  if ( !( ( iLump > 0 ) && ( iLump < nMaxLumpCount ) ) )
331  {
332  return false;
333  }
334

336  if ( iLumpCount <= 0 )
337  return false;
