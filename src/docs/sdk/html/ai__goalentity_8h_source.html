121 inline void CAI_GoalEntity::UpdateActors()
122 {
123  if ( !( m_flags & ACTIVE ) || !( m_flags & RESOLVED_NAME ) )
124  {
125  ResolveNames();
126  m_flags |= RESOLVED_NAME;
127  }
128  else
129  PruneActors();
130 }
