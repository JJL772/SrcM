9 // Initializers are a way to register your object to be initialized at startup time.
10 // They're a good way to have global variables without worrying about dependent
11 // constructors being called. They also make it so init code doesn't depend on the
12 // global objects it's initializing.
13 
14 // To use initializers, just use REGISTER_INITIALIZER to register your global variable like this:
15 // class SomeClass {....}
16 // SomeClass *g_pSomeClassSingleton = NULL;
17 // REGISTER_INITIALIZER(SomeClass, &g_pSomeClassSingleton);
