555  template < typename Functor >
556  bool ForAllAreasOverlappingExtent( Functor &func, const Extent &extent )
557  {
558  if ( !m_grid.Count() )
559  {
560 #if _DEBUG
561  Warning("Query before nav mesh is loaded! %d\n", TheNavAreas.Count() );
562 #endif
563  return true;
564  }
565  static unsigned int searchMarker = RandomInt(0, 1024*1024 );
566  if ( ++searchMarker == 0 )
567  {
568  ++searchMarker;
569  }
570 
571  Extent areaExtent;
572 
573  // get list in cell that contains position
574  int startX = WorldToGridX( extent.lo.x );
575  int endX = WorldToGridX( extent.hi.x );
576  int startY = WorldToGridY( extent.lo.y );
577  int endY = WorldToGridY( extent.hi.y );
578 
579  for( int x = startX; x <= endX; ++x )
580  {
581  for( int y = startY; y <= endY; ++y )
582  {
583  int iGrid = x + y*m_gridSizeX;
584  if ( iGrid >= m_grid.Count() )
585  {
586  ExecuteNTimes( 10, Warning( "** Walked off of the CNavMesh::m_grid in ForAllAreasOverlappingExtent()\n" ) );
587  return true;
588  }
589 
590  NavAreaVector *areaVector = &m_grid[ iGrid ];
591 
592  // find closest area in this cell
593  FOR_EACH_VEC( (*areaVector), it )
594  {
595  CNavArea *area = (*areaVector)[ it ];
596 
597  // skip if we've already visited this area
598  if ( area->m_nearNavSearchMarker == searchMarker )
599  continue;
600 
601  // mark as visited
602  area->m_nearNavSearchMarker = searchMarker;
603  area->GetExtent( &areaExtent );
604 
605  if ( extent.IsOverlapping( areaExtent ) )
606  {
607  if ( func( area ) == false )
608  return false;
609  }
610  }
611  }
612  }
613  return true;
614  }

748  return false;

766  func( startArea );
767  return true;

795  break;
796  }
797  }
798  }
799 
800  return false;
801  }

821 
822  const Vector &adjOrigin = adjArea->GetCorner( NORTH_WEST );
823

825  {

827  break;

869  }
870  }

873  // find Y at east edge intersection
874  float t = ( xMax - start.x ) / ( end.x - start.x );
875  if ( t > 0.0f && t < 1.0f )
876  {
877  float y = start.y + t * ( end.y - start.y );
878  if ( y >= yMin && y <= yMax )
879  {
880  // intersects this edge
881  exit.x = xMax;
882  exit.y = y;
883  edge = EAST;
884  }
885  }
886  }
887 
888  if ( edge == NUM_DIRECTIONS )
889  {
890  if ( to.y < 0.0f )
891  {

893  float t = ( yMin - start.y ) / ( end.y - start.y );
894  if ( t > 0.0f && t < 1.0f )
895  {
896  float x = start.x + t * ( end.x - start.x );
897  if ( x >= xMin && x <= xMax )
898  {
899  // intersects this edge
900  exit.x = x;
901  exit.y = yMin;
902  edge = NORTH;
903  }
904  }
905  }
906  else
907  {
908  // find X at south edge intersection

910  if ( t > 0.0f && t < 1.0f )
911  {
