434 inline CStudioHdr *CBaseAnimating::GetModelPtr( void )
435 {
436  if ( IsDynamicModelLoading() )
437  return NULL;
438 
439 #ifdef _DEBUG
440  if ( !HushAsserts() )
441  {
442  // GetModelPtr() is often called before OnNewModel() so go ahead and set it up first chance.
443  static IDataCacheSection *pModelCache = datacache->FindSection( "ModelData" );
444  AssertOnce( pModelCache->IsFrameLocking() );
445  }
446 #endif
447 
448  if ( !m_pStudioHdr && GetModel() )
449  {
450  LockStudioHdr();
451  }
452  return ( m_pStudioHdr && m_pStudioHdr->IsValid() ) ? m_pStudioHdr : NULL;
453 }
