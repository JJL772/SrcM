25 inline float NormalizeAngle( float angle )

205 inline void AI_Arc_t::SetByLimits( float yawRight, float yawLeft )
206 {
207  // Yaw runs counter-clockwise
208  span = yawLeft - yawRight;
209 
210  if ( span < 0 )
211  span += 360;
212 
213  center = yawRight + span * 0.5;
214 
215  if ( center >= 360 )
216  center -= 360;
217 }
