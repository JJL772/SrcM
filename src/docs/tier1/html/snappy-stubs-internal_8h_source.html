193 // This can be more efficient than UNALIGNED_LOAD64 + UNALIGNED_STORE64
194 // on some platforms, in particular ARM.
195 inline void UnalignedCopy64(const void *src, void *dst) {
196  if (sizeof(void *) == 8) {
197  UNALIGNED_STORE64(dst, UNALIGNED_LOAD64(src));

203  UNALIGNED_STORE32(dst_char + 4, UNALIGNED_LOAD32(src_char + 4));
204  }

351 inline int Bits::Log2Floor(uint32 n) {
352  if (n == 0)
353  return -1;
354  int log = 0;
355  uint32 value = n;
356  for (int i = 4; i >= 0; --i) {
357  int shift = (1 << i);
358  uint32 x = value >> shift;
359  if (x != 0) {
360  value = x;
361  log += shift;
362  }
363  }
364  assert(value == 1);
365  return log;
366 }

368 inline int Bits::FindLSBSetNonZero(uint32 n) {
369  int rc = 31;
370  for (int i = 4, shift = 1 << 4; i >= 0; --i) {
371  const uint32 x = n << shift;
372  if (x != 0) {
373  n = x;
374  rc -= shift;
375  }
376  shift >>= 1;
377  }
378  return rc;
379 }

381 // FindLSBSetNonZero64() is defined in terms of FindLSBSetNonZero().
382 inline int Bits::FindLSBSetNonZero64(uint64 n) {
383  const uint32 bottombits = static_cast<uint32>(n);
384  if (bottombits == 0) {
385  // Bottom bits are zero, so scan in top bits
386  return 32 + FindLSBSetNonZero(static_cast<uint32>(n >> 32));
387  } else {
388  return FindLSBSetNonZero(bottombits);
389  }
390 }

417 inline const char* Varint::Parse32WithLimit(const char* p,
418  const char* l,
419  uint32* OUTPUT) {
420  const unsigned char* ptr = reinterpret_cast<const unsigned char*>(p);
421  const unsigned char* limit = reinterpret_cast<const unsigned char*>(l);
422  uint32 b, result;
423  if (ptr >= limit) return NULL;
424  b = *(ptr++); result = b & 127; if (b < 128) goto done;
425  if (ptr >= limit) return NULL;
426  b = *(ptr++); result |= (b & 127) << 7; if (b < 128) goto done;
427  if (ptr >= limit) return NULL;
428  b = *(ptr++); result |= (b & 127) << 14; if (b < 128) goto done;
429  if (ptr >= limit) return NULL;
430  b = *(ptr++); result |= (b & 127) << 21; if (b < 128) goto done;
431  if (ptr >= limit) return NULL;
432  b = *(ptr++); result |= (b & 127) << 28; if (b < 16) goto done;
433  return NULL; // Value is too long to be a varint32
434  done:
435  *OUTPUT = result;
436  return reinterpret_cast<const char*>(ptr);
437 }

439 inline char* Varint::Encode32(char* sptr, uint32 v) {
440  // Operate on characters as unsigneds
441  unsigned char* ptr = reinterpret_cast<unsigned char*>(sptr);
442  static const int B = 128;
443  if (v < (1<<7)) {
444  *(ptr++) = v;
445  } else if (v < (1<<14)) {
446  *(ptr++) = v | B;
447  *(ptr++) = v>>7;
448  } else if (v < (1<<21)) {
449  *(ptr++) = v | B;
450  *(ptr++) = (v>>7) | B;
451  *(ptr++) = v>>14;
452  } else if (v < (1<<28)) {
453  *(ptr++) = v | B;
454  *(ptr++) = (v>>7) | B;
455  *(ptr++) = (v>>14) | B;
456  *(ptr++) = v>>21;
457  } else {
458  *(ptr++) = v | B;
459  *(ptr++) = (v>>7) | B;
460  *(ptr++) = (v>>14) | B;
461  *(ptr++) = (v>>21) | B;
462  *(ptr++) = v>>28;
463  }
464  return reinterpret_cast<char*>(ptr);
465 }
