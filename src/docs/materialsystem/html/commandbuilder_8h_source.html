212  void SetVertexShaderTextureTransform( int vertexReg, int transformVar )
213  {
214  Vector4D transformation[2];

216  if (pTransformationVar && (pTransformationVar->GetType() == MATERIAL_VAR_TYPE_MATRIX))

219  transformation[0].Init( mat[0][0], mat[0][1], mat[0][2], mat[0][3] );

231  void SetVertexShaderTextureScaledTransform( int vertexReg, int transformVar, int scaleVar )
232  {
233  Vector4D transformation[2];
234  IMaterialVar* pTransformationVar = Param( transformVar );
235  if (pTransformationVar && (pTransformationVar->GetType() == MATERIAL_VAR_TYPE_MATRIX))
236  {
237  const VMatrix &mat = pTransformationVar->GetMatrixValue();
238  transformation[0].Init( mat[0][0], mat[0][1], mat[0][2], mat[0][3] );
239  transformation[1].Init( mat[1][0], mat[1][1], mat[1][2], mat[1][3] );
240  }
241  else
242  {
243  transformation[0].Init( 1.0f, 0.0f, 0.0f, 0.0f );
244  transformation[1].Init( 0.0f, 1.0f, 0.0f, 0.0f );
245  }
246 
247  Vector2D scale( 1, 1 );
248  IMaterialVar* pScaleVar = Param( scaleVar );
249  if (pScaleVar)
250  {
251  if (pScaleVar->GetType() == MATERIAL_VAR_TYPE_VECTOR)
252  pScaleVar->GetVecValue( scale.Base(), 2 );
253  else if (pScaleVar->IsDefined())
254  scale[0] = scale[1] = pScaleVar->GetFloatValue();
255  }
256 
257  // Apply the scaling
258  transformation[0][0] *= scale[0];
259  transformation[0][1] *= scale[1];
260  transformation[1][0] *= scale[0];
261  transformation[1][1] *= scale[1];
262  transformation[0][3] *= scale[0];
263  transformation[1][3] *= scale[1];
264  SetVertexShaderConstant( vertexReg, transformation[0].Base(), 2 );
265  }
