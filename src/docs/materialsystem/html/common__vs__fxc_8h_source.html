184 //-----------------------------------------------------------------------------------
185 // Decompress a normal from two-component compressed format
186 // We expect this data to come from a signed SHORT2 stream in the range of -32768..32767
187 //
188 // -32678 and 0 are invalid encodings
189 // w contains the sign to use in the cross product when generating a binormal
190 void _DecompressShort2Tangent( float2 inputTangent, out float4 outputTangent )
191 {
192  float2 ztSigns = sign( inputTangent ); // sign bits for z and tangent (+1 or -1)
193  float2 xyAbs = abs( inputTangent ); // 1..32767
194  outputTangent.xy = (xyAbs - 16384.0f) / 16384.0f; // x and y
195  outputTangent.z = ztSigns.x * sqrt( saturate( 1.0f - dot( outputTangent.xy, outputTangent.xy ) ) );
196  outputTangent.w = ztSigns.y;
197 }

199 //-----------------------------------------------------------------------------------
200 // Same code as _DecompressShort2Tangent, just one returns a float4, one a float3
201 void _DecompressShort2Normal( float2 inputNormal, out float3 outputNormal )
202 {
203  float4 result;
204  _DecompressShort2Tangent( inputNormal, result );
205  outputNormal = result.xyz;
206 }

210 void _DecompressShort2NormalTangent( float2 inputNormal, float2 inputTangent, out float3 outputNormal, out float4 outputTangent )
211 {

218 //=======================================================================================
219 // Decompress a normal and tangent from four-component compressed format
220 // We expect this data to come from an unsigned UBYTE4 stream in the range of 0..255
221 // The final vTangent.w contains the sign to use in the cross product when generating a binormal
222 void _DecompressUByte4NormalTangent( float4 inputNormal,
223  out float3 outputNormal, // {nX, nY, nZ}
224  out float4 outputTangent ) // {tX, tY, tZ, sign of binormal}
225 {
226  float fOne = 1.0f;
227 
228  float4 ztztSignBits = ( inputNormal - 128.0f ) < 0; // sign bits for zs and binormal (1 or 0) set-less-than (slt) asm instruction
229  float4 xyxyAbs = abs( inputNormal - 128.0f ) - ztztSignBits; // 0..127
230  float4 xyxySignBits = ( xyxyAbs - 64.0f ) < 0; // sign bits for xs and ys (1 or 0)
231  float4 normTan = (abs( xyxyAbs - 64.0f ) - xyxySignBits) / 63.0f; // abs({nX, nY, tX, tY})
232  outputNormal.xy = normTan.xy; // abs({nX, nY, __, __})
233  outputTangent.xy = normTan.zw; // abs({tX, tY, __, __})
234 
235  float4 xyxySigns = 1 - 2*xyxySignBits; // Convert sign bits to signs
236  float4 ztztSigns = 1 - 2*ztztSignBits; // ( [1,0] -> [-1,+1] )
237 
238  outputNormal.z = 1.0f - outputNormal.x - outputNormal.y; // Project onto x+y+z=1
239  outputNormal.xyz = normalize( outputNormal.xyz ); // Normalize onto unit sphere
240  outputNormal.xy *= xyxySigns.xy; // Restore x and y signs
241  outputNormal.z *= ztztSigns.x; // Restore z sign
242 
243  outputTangent.z = 1.0f - outputTangent.x - outputTangent.y; // Project onto x+y+z=1
244  outputTangent.xyz = normalize( outputTangent.xyz ); // Normalize onto unit sphere
245  outputTangent.xy *= xyxySigns.zw; // Restore x and y signs
246  outputTangent.z *= ztztSigns.z; // Restore z sign
247  outputTangent.w = ztztSigns.w; // Binormal sign
248 }

251 //-----------------------------------------------------------------------------------
252 // Decompress just a normal from four-component compressed format (same as above)
253 // We expect this data to come from an unsigned UBYTE4 stream in the range of 0..255
254 // [ When compiled, this works out to approximately 17 asm instructions ]
255 void _DecompressUByte4Normal( float4 inputNormal,
256  out float3 outputNormal) // {nX, nY, nZ}
257 {
258  float fOne = 1.0f;
259 
260  float2 ztSigns = ( inputNormal.xy - 128.0f ) < 0; // sign bits for zs and binormal (1 or 0) set-less-than (slt) asm instruction
261  float2 xyAbs = abs( inputNormal.xy - 128.0f ) - ztSigns; // 0..127
262  float2 xySigns = ( xyAbs - 64.0f ) < 0; // sign bits for xs and ys (1 or 0)
263  outputNormal.xy = ( abs( xyAbs - 64.0f ) - xySigns ) / 63.0f; // abs({nX, nY})
264 
265  outputNormal.z = 1.0f - outputNormal.x - outputNormal.y; // Project onto x+y+z=1
266  outputNormal.xyz = normalize( outputNormal.xyz ); // Normalize onto unit sphere
267 
268  outputNormal.xy *= lerp( fOne.xx, -fOne.xx, xySigns ); // Restore x and y signs
269  outputNormal.z *= lerp( fOne.x, -fOne.x, ztSigns.x ); // Restore z sign
270 }

315 //-----------------------------------------------------------------------------
316 // Methods to sample morph data from a vertex texture
317 // NOTE: vMorphTargetTextureDim.x = width, cVertexTextureDim.y = height, cVertexTextureDim.z = # of float4 fields per vertex
318 // For position + normal morph for example, there will be 2 fields.
319 //-----------------------------------------------------------------------------
320 float4 SampleMorphDelta( sampler2D vt, const float3 vMorphTargetTextureDim, const float4 vMorphSubrect, const float flVertexID, const float flField )
321 {
322  float flColumn = floor( flVertexID / vMorphSubrect.w );
323 
324  float4 t;
325  t.x = vMorphSubrect.x + vMorphTargetTextureDim.z * flColumn + flField + 0.5f;
326  t.y = vMorphSubrect.y + flVertexID - flColumn * vMorphSubrect.w + 0.5f;
327  t.xy /= vMorphTargetTextureDim.xy;
328  t.z = t.w = 0.f;
329 
330  return tex2Dlod( vt, t );
331 }

335 {

528 float WaterFog( const float3 worldPos, const float3 projPos )
529 {
530  float4 tmp;
531 
532  tmp.xy = cEyePosWaterZ.wz - worldPos.z;
533 
534  // tmp.x is the distance from the water surface to the vert
535  // tmp.y is the distance from the eye position to the vert
536 
537  // if $tmp.x < 0, then set it to 0
538  // This is the equivalent of moving the vert to the water surface if it's above the water surface
539 
540  tmp.x = max( 0.0f, tmp.x );
541 
542  // $tmp.w = $tmp.x / $tmp.y
543  tmp.w = tmp.x / tmp.y;
544 
545  tmp.w *= projPos.z;
546 
547  // $tmp.w is now the distance that we see through water.
548 
549  return max( cFogMaxDensity, ( -tmp.w * cOOFogRange + cFogOne ) );
550 }

599 float4 DecompressBoneWeights( const float4 weights )
600 {
601  float4 result = weights;
602 
603  if ( COMPRESSED_VERTS )
604  {
605  // Decompress from SHORT2 to float. In our case, [-1, +32767] -> [0, +1]
606  // NOTE: we add 1 here so we can divide by 32768 - which is exact (divide by 32767 is not).
607  // This avoids cracking between meshes with different numbers of bone weights.
608  // We use SHORT2 instead of SHORT2N for a similar reason - the GPU's conversion
609  // from [-32768,+32767] to [-1,+1] is imprecise in the same way.
610  result += 1;
611  result /= 32768;
612  }
613 
614  return result;

750 // The following "internal" routines are called "privately" by other routines in this file which
751 // handle the particular flavor of vs20 control flow appropriate to the original caller
752 float VertexAttenInternal( const float3 worldPos, int lightNum )
753 {
754  float result = 0.0f;
755 
756  // Get light direction
757  float3 lightDir = cLightInfo[lightNum].pos - worldPos;
758 
759  // Get light distance squared.
760  float lightDistSquared = dot( lightDir, lightDir );
761 
762  // Get 1/lightDistance
763  float ooLightDist = rsqrt( lightDistSquared );
764 
765  // Normalize light direction
766  lightDir *= ooLightDist;
767 
768  float3 vDist;
769 # if defined( _X360 )
770  {
771  //X360 dynamic compile hits an internal compiler error using dst(), this is the breakdown of how dst() works from the 360 docs.
772  vDist.x = 1;
773  vDist.y = lightDistSquared * ooLightDist;
774  vDist.z = lightDistSquared;
775  //flDist.w = ooLightDist;
776  }
777 # else
778  {
779  vDist = dst( lightDistSquared, ooLightDist );
780  }
781 # endif
782 
783  float flDistanceAtten = 1.0f / dot( cLightInfo[lightNum].atten.xyz, vDist );
784 
785  // Spot attenuation
786  float flCosTheta = dot( cLightInfo[lightNum].dir.xyz, -lightDir );
787  float flSpotAtten = (flCosTheta - cLightInfo[lightNum].spotParams.z) * cLightInfo[lightNum].spotParams.w;
788  flSpotAtten = max( 0.0001f, flSpotAtten );
789  flSpotAtten = pow( flSpotAtten, cLightInfo[lightNum].spotParams.x );
790  flSpotAtten = saturate( flSpotAtten );
791 
792  // Select between point and spot
793  float flAtten = lerp( flDistanceAtten, flDistanceAtten * flSpotAtten, cLightInfo[lightNum].dir.w );
794 
795  // Select between above and directional (no attenuation)
796  result = lerp( flAtten, 1.0f, cLightInfo[lightNum].color.w );
797 
798  return result;
799 }

801 float CosineTermInternal( const float3 worldPos, const float3 worldNormal, int lightNum, bool bHalfLambert )
802 {
803  // Calculate light direction assuming this is a point or spot
804  float3 lightDir = normalize( cLightInfo[lightNum].pos - worldPos );
805 
806  // Select the above direction or the one in the structure, based upon light type
807  lightDir = lerp( lightDir, -cLightInfo[lightNum].dir, cLightInfo[lightNum].color.w );
808 
809  // compute N dot L
810  float NDotL = dot( worldNormal, lightDir );
811 
812  if ( !bHalfLambert )
813  {
814  NDotL = max( 0.0f, NDotL );
815  }
816  else // Half-Lambert
817  {
818  NDotL = NDotL * 0.5 + 0.5;
819  NDotL = NDotL * NDotL;
820  }
821  return NDotL;
822 }
