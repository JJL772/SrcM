340 FORCEINLINE void SetFlashLightColorFromState( FlashlightState_t const &state, IShaderDynamicAPI *pShaderAPI, int nPSRegister=28, bool bFlashlightNoLambert=false )
341 {
342  // Old code
343  //float flToneMapScale = ( pShaderAPI->GetToneMappingScaleLinear() ).x;
344  //float flFlashlightScale = 1.0f / flToneMapScale;
345 
346  // Fix to old code to keep flashlight from ever getting brighter than 1.0
347  //float flToneMapScale = ( pShaderAPI->GetToneMappingScaleLinear() ).x;
348  //if ( flToneMapScale < 1.0f )
349  // flToneMapScale = 1.0f;
350  //float flFlashlightScale = 1.0f / flToneMapScale;
351 
352  // Force flashlight to 25% bright always
353  float flFlashlightScale = 0.25f;
354 
355  if ( !g_pHardwareConfig->GetHDREnabled() )
356  {
357  // Non-HDR path requires 2.0 flashlight
358  flFlashlightScale = 2.0f;
359  }
360 
361  // DX10 requires some hackery due to sRGB/blend ordering change from DX9
362  if ( g_pHardwareConfig->UsesSRGBCorrectBlending() )
363  {
364  flFlashlightScale *= 2.5f; // Magic number that works well on the NVIDIA 8800
365  }
366 
367  // Generate pixel shader constant
368  float const *pFlashlightColor = state.m_Color;
369  float vPsConst[4] = { flFlashlightScale * pFlashlightColor[0], flFlashlightScale * pFlashlightColor[1], flFlashlightScale * pFlashlightColor[2], pFlashlightColor[3] };
370  vPsConst[3] = bFlashlightNoLambert ? 2.0f : 0.0f; // This will be added to N.L before saturate to force a 1.0 N.L term
371 
372  // Red flashlight for testing
373  //vPsConst[0] = 0.5f; vPsConst[1] = 0.0f; vPsConst[2] = 0.0f;
374 
375  pShaderAPI->SetPixelShaderConstant( nPSRegister, ( float * )vPsConst );
376 }
