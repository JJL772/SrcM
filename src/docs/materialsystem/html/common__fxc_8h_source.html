52 HALF3 CalcReflectionVectorNormalized( HALF3 normal, HALF3 eyeVector )
53 {
54  // FIXME: might be better of normalizing with a normalizing cube map and
55  // get rid of the dot( normal, normal )
56  // compute reflection vector r = 2 * ((n dot v)/(n dot n)) n - v
57  return 2.0 * ( dot( normal, eyeVector ) / dot( normal, normal ) ) * normal - eyeVector;
58 }

60 HALF3 CalcReflectionVectorUnnormalized( HALF3 normal, HALF3 eyeVector )
61 {
62  // FIXME: might be better of normalizing with a normalizing cube map and
63  // get rid of the dot( normal, normal )
64  // compute reflection vector r = 2 * ((n dot v)/(n dot n)) n - v
65  // multiply all values through by N.N. uniformly scaling reflection vector won't affect result
66  // since it is used in a cubemap lookup
67  return (2.0*(dot( normal, eyeVector ))*normal) - (dot( normal, normal )*eyeVector);
68 }

152 float4 CompressHDR( float3 input )
153 {
154  // FIXME: want to use min so that we clamp to white, but what happens if we
155  // have an albedo component that's less than 1/MAX_HDR_OVERBRIGHT?
156  // float fMax = max( max( color.r, color.g ), color.b );
157  float4 output;
158  float fMax = min( min( input.r, input.g ), input.b );
159  if( fMax > 1.0f )
160  {
161  float oofMax = 1.0f / fMax;
162  output.rgb = oofMax * input.rgb;
163  output.a = min( fMax / MAX_HDR_OVERBRIGHT, 1.0f );
164  }
165  else
166  {
167  output.rgb = input.rgb;
168  output.a = 0.0f;
169  }
170  return output;
171 }

217 // These twofunctions use the XBox 360's exact piecewise linear algorithm
218 float X360GammaToLinear( float fl360GammaValue )
219 {
220  float flLinearValue;
221 
222  fl360GammaValue = saturate( fl360GammaValue );
223  if ( fl360GammaValue < ( 96.0f / 255.0f ) )
224  {
225  if ( fl360GammaValue < ( 64.0f / 255.0f ) )
226  {
227  flLinearValue = fl360GammaValue * 255.0f;
228  }
229  else
230  {
231  flLinearValue = fl360GammaValue * ( 255.0f * 2.0f ) - 64.0f;
232  flLinearValue += floor( flLinearValue * ( 1.0f / 512.0f ) );
233  }
234  }
235  else
236  {
237  if( fl360GammaValue < ( 192.0f / 255.0f ) )
238  {
239  flLinearValue = fl360GammaValue * ( 255.0f * 4.0f ) - 256.0f;
240  flLinearValue += floor( flLinearValue * ( 1.0f / 256.0f ) );
241  }
242  else
243  {
244  flLinearValue = fl360GammaValue * ( 255.0f * 8.0f ) - 1024.0f;
245  flLinearValue += floor( flLinearValue * ( 1.0f / 128.0f ) );
246  }
247  }
248 
249  flLinearValue *= 1.0f / 1023.0f;
250 
251  flLinearValue = saturate( flLinearValue );
252  return flLinearValue;
253 }

255 float X360LinearToGamma( float flLinearValue )
256 {
257  float fl360GammaValue;
258 
259  flLinearValue = saturate( flLinearValue );
260  if ( flLinearValue < ( 128.0f / 1023.0f ) )
261  {
262  if ( flLinearValue < ( 64.0f / 1023.0f ) )
263  {
264  fl360GammaValue = flLinearValue * ( 1023.0f * ( 1.0f / 255.0f ) );
265  }
266  else
267  {
268  fl360GammaValue = flLinearValue * ( ( 1023.0f / 2.0f ) * ( 1.0f / 255.0f ) ) + ( 32.0f / 255.0f );
269  }
270  }
271  else
272  {
273  if ( flLinearValue < ( 512.0f / 1023.0f ) )
274  {
275  fl360GammaValue = flLinearValue * ( ( 1023.0f / 4.0f ) * ( 1.0f / 255.0f ) ) + ( 64.0f / 255.0f );
276  }
277  else
278  {
279  fl360GammaValue = flLinearValue * ( ( 1023.0f /8.0f ) * ( 1.0f / 255.0f ) ) + ( 128.0f /255.0f ); // 1.0 -> 1.0034313725490196078431372549016
280  if ( fl360GammaValue > 1.0f )
281  {
282  fl360GammaValue = 1.0f;
283  }
284  }
285  }
286 
287  fl360GammaValue = saturate( fl360GammaValue );
288  return fl360GammaValue;
