127 // Optional analysis settings passed to TalkBackGetAnalysis().
128 typedef struct
129 {
130  // Set this field to sizeof(TALKBACK_ANALYSIS_SETTINGS) before using the
131  // structure.
132  long fSize;
133  // Frame rate for analysis. This only matters if you will be using *AtFrame
134  // functions.
135  //
136  // Default value: 30 (frames per second).
137  long fFrameRate;
138  // Set this to 1 to optimize for flipbook output, 0 to do analysis normally.
139  //
140  // Default value: 0 (normal analysis).
141  long fOptimizeForFlipbook;
142  // Set this to -1 to seed the random number generator with the current time.
143  // Any other number will be used directly for the random number seed, which
144  // is useful if you want repeatable speech gestures. This value does not
145  // influence lip-synching at all.
146  //
147  // Default value: -1 (use current time).
148  long fRandomSeed;
149  // Path to the configuration (.INI) file with phoneme-to-speech-target
150  // mapping. Set this to NULL to use the default mapping.
151  //
152  // Default value: NULL (use default mapping).
153  char const *fConfigFile;
154 } TALKBACK_ANALYSIS_SETTINGS;

188 // TalkBack error codes. Use TalkBackGetErrorString() to return text
189 // descriptions for these codes.
190 enum
191 {
192  // Windows convention: set this bit to indicate an application-defined error
193  // code.
194  BIT29 = (1 << 29),
195  // Success (not an error).
196  TALKBACK_NOERR = 0,
197  // The first error code: useful for iterating through the error codes.
198  TALKBACK_ERROR_FIRST = 4201 | BIT29,
199  // Generic error.
200  TALKBACK_ERROR = TALKBACK_ERROR_FIRST,
201  // TalkBackStartupLibrary() failed [internal error] or was never called.
202  TALKBACK_STARTUP_FAILED_ERR,
203  // TalkBackShutdownLibrary() failed, either because
204  // TalkBackStartupLibrary() was never called or because
205  // TalkBackShutdownLibrary() has already been called.
206  TALKBACK_SHUTDOWN_FAILED_ERR,
207  // The TalkBack data files could not be found [invalid path or missing
208  // files].
209  TALKBACK_CORE_DATA_NOT_FOUND_ERR,
210  // One or more of the parameters are NULL.
211  TALKBACK_NULL_PARAMETER_ERR,
212  // One or more of the parameters is invalid.
213  TALKBACK_INVALID_PARAMETER_ERR,
214  // The analysis object pointer is invalid.
215  TALKBACK_INVALID_ANALYSIS_ERR,
216  // Analysis failed [the sound file cannot be analyzed or an internal error
217  // occurred].
218  TALKBACK_ANALYSIS_FAILED_ERR,
219  // One or more of the indices (track, key, frame, word, phoneme) are
220  // invalid (out of range).
221  TALKBACK_INVALID_INDEX_ERR,
222  // The time parameter is invalid (out of range).
223  TALKBACK_INVALID_TIME_ERR,
224  // A serious internal error occurred in TalkBack; please alert LIPSinc by
225  // sending mail with a description of how the error was triggered to
226  // talkback-support@LIPSinc.com.
227  TALKBACK_INTERNAL_ERR,
228  // Could not open the specified sound file.
229  TALKBACK_COULD_NOT_LOAD_SOUND_ERR,
230  // TalkBackStartupLibrary() has not been called.
231  TALKBACK_STARTUP_NOT_CALLED,
232  // The configuration file specified in the TALKBACK_ANALYSIS_SETTINGS
233  // structure is invalid.
234  TALKBACK_CONFIG_PARSE_ERROR,
235  // The last error code: useful for iterating through the error codes.
236  TALKBACK_ERROR_LAST = TALKBACK_CONFIG_PARSE_ERROR
237 };
