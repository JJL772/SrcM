121  FORCEINLINE bool IsValid( void ) const

129  }
130 
131  FORCEINLINE void Zero( void )
132  {
133  m_vecLighting.Init( 0, 0, 0 );
134  m_flDirectSunAmount = 0.0;
135  }
136 
137  FORCEINLINE void Scale( float m_flScale )
138  {
139  m_vecLighting *= m_flScale;
140  m_flDirectSunAmount *= m_flScale;
141  }
142 
143  FORCEINLINE void AddWeighted( LightingValue_t const &src, float flWeight )
144  {
145  m_vecLighting += flWeight * src.m_vecLighting;
146  m_flDirectSunAmount += flWeight * src.m_flDirectSunAmount;
147  }

149  FORCEINLINE void AddWeighted( Vector const &src, float flWeight )
150  {
151  m_vecLighting += flWeight * src;
